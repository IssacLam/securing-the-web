46 static int s2n_ecdhe_server_key_recv(struct s2n_connection *conn)
47 {
48  struct s2n_hash_state signature_hash;
49  struct s2n_stuffer *in = &conn->handshake.io;
50  struct s2n_blob ecdhparams;
51  struct s2n_blob signature;
52  uint16_t signature_length;
53 
54  /* Read server ECDH params and calculate their hash */
55  GUARD(s2n_ecc_read_ecc_params(&conn->secure.server_ecc_params, in, &ecdhparams));
56 
57  GUARD(s2n_hash_init(&signature_hash, conn->secure.signature_digest_alg));
58 
59  if (conn->actual_protocol_version == S2N_TLS12) {
60  uint8_t hash_algorithm;
61  uint8_t signature_algorithm;
62 
63  GUARD(s2n_stuffer_read_uint8(in, &hash_algorithm));
64  GUARD(s2n_stuffer_read_uint8(in, &signature_algorithm));
65 
66  if (signature_algorithm != TLS_SIGNATURE_ALGORITHM_RSA) {
67  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
68  }
69 
70  switch (hash_algorithm) {
71  case TLS_HASH_ALGORITHM_MD5:
72  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_MD5));
73  break;
74  case TLS_HASH_ALGORITHM_SHA1:
75  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA1));
76  break;
77  case TLS_HASH_ALGORITHM_SHA224:
78  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA224));
79  break;
80  case TLS_HASH_ALGORITHM_SHA256:
81  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA256));
82  break;
83  case TLS_HASH_ALGORITHM_SHA384:
84  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA384));
85  break;
86  case TLS_HASH_ALGORITHM_SHA512:
87  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA512));
88  break;
89  default:
90  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
91  }
92  }
93 
94  GUARD(s2n_hash_update(&signature_hash, conn->secure.client_random, S2N_TLS_RANDOM_DATA_LEN));
95  GUARD(s2n_hash_update(&signature_hash, conn->secure.server_random, S2N_TLS_RANDOM_DATA_LEN));
96  GUARD(s2n_hash_update(&signature_hash, ecdhparams.data, ecdhparams.size));
97 
98  /* Verify the signature */
99  GUARD(s2n_stuffer_read_uint16(in, &signature_length));
100  signature.size = signature_length;
101  signature.data = s2n_stuffer_raw_read(in, signature.size);
102  notnull_check(signature.data);
103 
104  gt_check(signature_length, 0);
105 
106  if (s2n_rsa_verify(&conn->secure.server_rsa_public_key, &signature_hash, &signature) < 0) {
107  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
108  }
109 
110  /* We don't need the key any more, so free it */
111  GUARD(s2n_rsa_public_key_free(&conn->secure.server_rsa_public_key));
112 
113  return 0;
114 }

116 static int s2n_dhe_server_key_recv(struct s2n_connection *conn)
117 {
118  struct s2n_hash_state signature_hash;
119  struct s2n_stuffer *in = &conn->handshake.io;
120  struct s2n_blob p, g, Ys, serverDHparams, signature;
121  uint16_t p_length;
122  uint16_t g_length;
123  uint16_t Ys_length;
124  uint16_t signature_length;
125 
126  /* Keep a copy to the start of the whole structure for the signature check */
127  serverDHparams.data = s2n_stuffer_raw_read(in, 0);
128  notnull_check(serverDHparams.data);
129 
130  /* Read each of the three elements in */
131  GUARD(s2n_stuffer_read_uint16(in, &p_length));
132  p.size = p_length;
133  p.data = s2n_stuffer_raw_read(in, p.size);
134  notnull_check(p.data);
135 
136  GUARD(s2n_stuffer_read_uint16(in, &g_length));
137  g.size = g_length;
138  g.data = s2n_stuffer_raw_read(in, g.size);
139  notnull_check(g.data);
140 
141  GUARD(s2n_stuffer_read_uint16(in, &Ys_length));
142  Ys.size = Ys_length;
143  Ys.data = s2n_stuffer_raw_read(in, Ys.size);
144  notnull_check(Ys.data);
145 
146  /* Now we know the total size of the structure */
147  serverDHparams.size = 2 + p_length + 2 + g_length + 2 + Ys_length;
148 
149  GUARD(s2n_hash_init(&signature_hash, conn->secure.signature_digest_alg));
150 
151  if (conn->actual_protocol_version == S2N_TLS12) {
152  uint8_t hash_algorithm;
153  uint8_t signature_algorithm;
154 
155  GUARD(s2n_stuffer_read_uint8(in, &hash_algorithm));
156  GUARD(s2n_stuffer_read_uint8(in, &signature_algorithm));
157 
158  if (signature_algorithm != TLS_SIGNATURE_ALGORITHM_RSA) {
159  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
160  }
161 
162  switch (hash_algorithm) {
163  case TLS_HASH_ALGORITHM_MD5:
164  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_MD5));
165  break;
166  case TLS_HASH_ALGORITHM_SHA1:
167  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA1));
168  break;
169  case TLS_HASH_ALGORITHM_SHA224:
170  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA224));
171  break;
172  case TLS_HASH_ALGORITHM_SHA256:
173  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA256));
174  break;
175  case TLS_HASH_ALGORITHM_SHA384:
176  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA384));
177  break;
178  case TLS_HASH_ALGORITHM_SHA512:
179  GUARD(s2n_hash_init(&signature_hash, S2N_HASH_SHA512));
180  break;
181  default:
182  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
183  }
184  }
185 
186  GUARD(s2n_hash_update(&signature_hash, conn->secure.client_random, S2N_TLS_RANDOM_DATA_LEN));
187  GUARD(s2n_hash_update(&signature_hash, conn->secure.server_random, S2N_TLS_RANDOM_DATA_LEN));
188  GUARD(s2n_hash_update(&signature_hash, serverDHparams.data, serverDHparams.size));
189 
190  GUARD(s2n_stuffer_read_uint16(in, &signature_length));
191  signature.size = signature_length;
192  signature.data = s2n_stuffer_raw_read(in, signature.size);
193  notnull_check(signature.data);
194 
195  gt_check(signature_length, 0);
196 
197  if (s2n_rsa_verify(&conn->secure.server_rsa_public_key, &signature_hash, &signature) < 0) {
198  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
199  }
200 
201  /* We don't need the key any more, so free it */
202  GUARD(s2n_rsa_public_key_free(&conn->secure.server_rsa_public_key));
203 
204  /* Copy the DH details */
205  GUARD(s2n_dh_p_g_Ys_to_dh_params(&conn->secure.server_dh_params, &p, &g, &Ys));
206 
207  return 0;
208 }

221 static int s2n_ecdhe_server_key_send(struct s2n_connection *conn)
222 {
223  struct s2n_blob signature;
224  struct s2n_stuffer *out = &conn->handshake.io;
225  struct s2n_hash_state signature_hash;
226  struct s2n_blob ecdhparams;
227 
228  GUARD(s2n_hash_init(&signature_hash, conn->secure.signature_digest_alg));
229 
230  /* Generate an ephemeral key and */
231  GUARD(s2n_ecc_generate_ephemeral_key(&conn->secure.server_ecc_params));
232 
233  /* Write it out and calcualte the hash */
234  GUARD(s2n_ecc_write_ecc_params(&conn->secure.server_ecc_params, out, &ecdhparams));
235 
236  if (conn->actual_protocol_version == S2N_TLS12) {
237  GUARD(s2n_stuffer_write_uint8(out, TLS_HASH_ALGORITHM_SHA1));
238  GUARD(s2n_stuffer_write_uint8(out, TLS_SIGNATURE_ALGORITHM_RSA));
239  }
240 
241  /* Add the random data to the hash */
242  GUARD(s2n_hash_update(&signature_hash, conn->secure.client_random, S2N_TLS_RANDOM_DATA_LEN));
243  GUARD(s2n_hash_update(&signature_hash, conn->secure.server_random, S2N_TLS_RANDOM_DATA_LEN));
244  GUARD(s2n_hash_update(&signature_hash, ecdhparams.data, ecdhparams.size));
245 
246  signature.size = s2n_rsa_private_encrypted_size(&conn->config->cert_and_key_pairs->private_key);
247  GUARD(s2n_stuffer_write_uint16(out, signature.size));
248 
249  signature.data = s2n_stuffer_raw_write(out, signature.size);
250  notnull_check(signature.data);
251 
252  if (s2n_rsa_sign(&conn->config->cert_and_key_pairs->private_key, &signature_hash, &signature) < 0) {
253  S2N_ERROR(S2N_ERR_DH_FAILED_SIGNING);
254  }
255 
256  return 0;
257 }

275  GUARD(s2n_stuffer_write_uint8(out, TLS_HASH_ALGORITHM_SHA1));
276  GUARD(s2n_stuffer_write_uint8(out, TLS_SIGNATURE_ALGORITHM_RSA));
277  }
