33 static int s2n_stuffer_read_n_bits_hex(struct s2n_stuffer *stuffer, uint8_t n, uint64_t *u)
34 {
35  uint8_t hex_data[16];
36  struct s2n_blob b = { .data = hex_data, .size = n / 4 };
37 
38  GUARD(s2n_stuffer_read(stuffer, &b));
39 
40  /* Start with u = 0 */
41  *u = 0;
42 
43  for (int i = 0; i < b.size; i++) {
44  *u <<= 4;
45  if (b.data[i] >= '0' && b.data[i] <= '9') {
46  *u |= b.data[i] - '0';
47  } else if (b.data[i] >= 'a' && b.data[i] <= 'f') {
48  *u |= b.data[i] - 'a' + 10;
49  } else if (b.data[i] >= 'A' && b.data[i] <= 'F') {
50  *u |= b.data[i] - 'A' + 10;
51  } else {
52  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
53  }
54  }
55 
56  return 0;
57 }

126 static int s2n_stuffer_write_n_bits_hex(struct s2n_stuffer *stuffer, uint8_t n, uint64_t u)
127 {
128  uint8_t hex_data[16] = { 0 };
129  struct s2n_blob b = { .data = hex_data, .size = n / 4 };
130 
131  lte_check(n, 64);
132 
133  for (int i = b.size; i > 0; i--) {
134  b.data[i - 1] = hex[u & 0x0f];
135  u >>= 4;
136  }
137 
138  GUARD(s2n_stuffer_write(stuffer, &b));
139 
140  return 0;
141 }

163 int s2n_stuffer_alloc_ro_from_hex_string(struct s2n_stuffer *stuffer, const char *str)
164 {
165  if (strlen(str) % 2) {
166  S2N_ERROR(S2N_ERR_SIZE_MISMATCH);
167  }
168 
169  GUARD(s2n_stuffer_alloc(stuffer, strlen(str) / 2));
170 
171  for (int i = 0; i < strlen(str); i += 2) {
172  uint8_t u = 0;
173 
174  if (str[i] >= '0' && str[i] <= '9') {
175  u = str[i] - '0';
176  } else if (str[i] >= 'a' && str[i] <= 'f') {
177  u = str[i] - 'a' + 10;
178  } else if (str[i] >= 'A' && str[i] <= 'F') {
179  u = str[i] - 'A' + 10;
180  } else {
181  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
182  }
183  u <<= 4;
184 
185  if (str[i + 1] >= '0' && str[i + 1] <= '9') {
186  u |= str[i + 1] - '0';
187  } else if (str[i + 1] >= 'a' && str[i + 1] <= 'f') {
188  u |= str[i + 1] - 'a' + 10;
189  } else if (str[i + 1] >= 'A' && str[i + 1] <= 'F') {
190  u |= str[i + 1] - 'A' + 10;
191  } else {
192  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
193  }
194 
195  GUARD(s2n_stuffer_write_uint8(stuffer, u));
196  }
197 
198  return 0;
199 }
