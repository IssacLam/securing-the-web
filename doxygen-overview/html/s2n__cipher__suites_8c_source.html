65 struct s2n_cipher_suite *s2n_cipher_suite_match(uint8_t cipher_suite[S2N_TLS_CIPHER_SUITE_LEN])
66 {
67  int low = 0;
68  int top = (sizeof(s2n_all_cipher_suites) / sizeof(struct s2n_cipher_suite)) - 1;
69 
70  /* Perform a textbook binary search */
71  while (low <= top) {
72  /* Check in the middle */
73  int mid = low + ((top - low) / 2);
74  int m = memcmp(s2n_all_cipher_suites[mid].value, cipher_suite, 2);
75 
76  if (m == 0) {
77  return &s2n_all_cipher_suites[mid];
78  } else if (m > 0) {
79  top = mid - 1;
80  } else if (m < 0) {
81  low = mid + 1;
82  }
83  }
84 
85  return NULL;
86 }

99 static int s2n_wire_ciphers_contain(uint8_t * match, uint8_t * wire, uint32_t count, uint32_t cipher_suite_len)
100 {
101  for (int i = 0; i < count; i++) {
102  uint8_t *theirs = wire + (i * cipher_suite_len) + (cipher_suite_len - S2N_TLS_CIPHER_SUITE_LEN);
103 
104  if (!memcmp(match, theirs, S2N_TLS_CIPHER_SUITE_LEN)) {
105  return 1;
106  }
107  }
108 
109  return 0;
110 }

112 static int s2n_set_cipher_as_server(struct s2n_connection *conn, uint8_t * wire, uint32_t count, uint32_t cipher_suite_len)
113 {
114  uint8_t renegotiation_info_scsv[S2N_TLS_CIPHER_SUITE_LEN] = { TLS_EMPTY_RENEGOTIATION_INFO_SCSV };
115  struct s2n_cipher_suite *higher_vers_match = NULL;
116 
117  /* RFC 7507 - If client is attempting to negotiate a TLS Version that is lower than the highest supported server
118  * version, and the client cipher list contains TLS_FALLBACK_SCSV, then the server must abort the connection since
119  * TLS_FALLBACK_SCSV should only be present when the client previously failed to negotiate a higher TLS version.
120  */
121  if (conn->client_protocol_version < S2N_TLS12) {
122  uint8_t fallback_scsv[S2N_TLS_CIPHER_SUITE_LEN] = { TLS_FALLBACK_SCSV };
123  if (s2n_wire_ciphers_contain(fallback_scsv, wire, count, cipher_suite_len)) {
124  conn->closed = 1;
125  S2N_ERROR(S2N_ERR_FALLBACK_DETECTED);
126  }
127  }
128 
129  /* RFC5746 Section 3.6: A server must check if TLS_EMPTY_RENEGOTIATION_INFO_SCSV is included */
130  if (s2n_wire_ciphers_contain(renegotiation_info_scsv, wire, count, cipher_suite_len)) {
131  conn->secure_renegotiation = 1;
132  }
133 
134  /* s2n supports only server order */
135  for (int i = 0; i < conn->config->cipher_preferences->count; i++) {
136  uint8_t *ours = conn->config->cipher_preferences->wire_format + (i * S2N_TLS_CIPHER_SUITE_LEN);
137 
138  if (s2n_wire_ciphers_contain(ours, wire, count, cipher_suite_len)) {
139  /* We have a match */
140  struct s2n_cipher_suite *match = s2n_cipher_suite_match(ours);
141 
142  /* This should never happen */
143  if (match == NULL) {
144  S2N_ERROR(S2N_ERR_CIPHER_NOT_SUPPORTED);
145  }
146 
147  /* Don't choose DHE key exchange if it's not configured. */
148  if (conn->config->dhparams == NULL && match->key_exchange_alg == &s2n_dhe) {
149  continue;
150  }
151  /* Don't choose EC ciphers if the curve was not agreed upon. */
152  if (conn->secure.server_ecc_params.negotiated_curve == NULL && (match->key_exchange_alg->flags & S2N_KEY_EXCHANGE_ECC)) {
153  continue;
154  }
155 
156  /* Don't immediately choose a cipher the client shouldn't be able to support */
157  if (conn->client_protocol_version < match->minimum_required_tls_version) {
158  higher_vers_match = match;
159  continue;
160  }
161 
162  conn->secure.cipher_suite = match;
163  return 0;
164  }
165  }
166 
167  /* Settle for a cipher with a higher required proto version, if it was set */
168  if (higher_vers_match != NULL) {
169  conn->secure.cipher_suite = higher_vers_match;
170  return 0;
171  }
172 
173  S2N_ERROR(S2N_ERR_CIPHER_NOT_SUPPORTED);
174 }
