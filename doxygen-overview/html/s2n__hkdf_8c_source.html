43 static int s2n_hkdf_expand(s2n_hmac_algorithm alg, const struct s2n_blob *pseudo_rand_key, const struct s2n_blob *info, struct s2n_blob *output)
44 {
45  uint8_t prev[MAX_DIGEST_SIZE] = { 0 };
46 
47  uint32_t done_len = 0;
48  uint32_t hash_len = s2n_hmac_digest_size(alg);
49  uint32_t total_rounds = output->size / hash_len;
50  if (output->size % hash_len) {
51  total_rounds++;
52  }
53 
54  if (total_rounds > MAX_HKDF_ROUNDS || total_rounds <= 0) {
55  S2N_ERROR(S2N_ERR_HKDF_OUTPUT_SIZE);
56  }
57 
58  struct s2n_hmac_state hmac;
59 
60  uint32_t cat_len;
61  for (uint32_t curr_round = 1; curr_round <= total_rounds; curr_round++) {
62 
63  GUARD(s2n_hmac_init(&hmac, alg, pseudo_rand_key->data, pseudo_rand_key->size));
64  if (curr_round != 1) {
65  GUARD(s2n_hmac_update(&hmac, prev, hash_len));
66  }
67  GUARD(s2n_hmac_update(&hmac, info->data, info->size));
68  GUARD(s2n_hmac_update(&hmac, &curr_round, 1));
69  GUARD(s2n_hmac_digest(&hmac, prev, hash_len));
70 
71  cat_len = hash_len;
72  if (done_len + hash_len > output->size) {
73  cat_len = output->size - done_len;
74  }
75 
76  memcpy_check(output->data + done_len, prev, cat_len);
77 
78  done_len += cat_len;
79  }
80 
81  return 0;
82 }
