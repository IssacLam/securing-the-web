26 static int s2n_aead_cipher_aes_gcm_encrypt(struct s2n_session_key *key, struct s2n_blob *iv, struct s2n_blob *aad, struct s2n_blob *in, struct s2n_blob *out)
27 {
28  gte_check(in->size, S2N_TLS_GCM_TAG_LEN + S2N_TLS_GCM_EXPLICIT_IV_LEN);
29  gte_check(out->size, in->size);
30  eq_check(iv->size, S2N_TLS_GCM_IV_LEN);
31  eq_check(aad->size, S2N_TLS_GCM_AAD_LEN);
32 
33  /* Initialize the IV */
34  if (EVP_EncryptInit_ex(&key->native_format.evp_cipher_ctx, NULL, NULL, NULL, iv->data) != 1) {
35  S2N_ERROR(S2N_ERR_KEY_INIT);
36  }
37 
38  /* Adjust our buffer pointers to account for the explicit IV and TAG lengths */
39  int in_len = in->size - (S2N_TLS_GCM_EXPLICIT_IV_LEN + S2N_TLS_GCM_TAG_LEN);
40  uint8_t *in_data = in->data + S2N_TLS_GCM_EXPLICIT_IV_LEN;
41  uint8_t *out_data = out->data + S2N_TLS_GCM_EXPLICIT_IV_LEN;
42  uint8_t *tag_data = out->data + out->size - S2N_TLS_GCM_TAG_LEN;
43 
44  int out_len;
45  /* Specify the AAD */
46  if (EVP_EncryptUpdate(&key->native_format.evp_cipher_ctx, NULL, &out_len, aad->data, aad->size) != 1) {
47  S2N_ERROR(S2N_ERR_ENCRYPT);
48  }
49 
50  /* Encrypt the data */
51  if (EVP_EncryptUpdate(&key->native_format.evp_cipher_ctx, out_data, &out_len, in_data, in_len) != 1) {
52  S2N_ERROR(S2N_ERR_ENCRYPT);
53  }
54 
55  /* Finalize */
56  if (EVP_EncryptFinal_ex(&key->native_format.evp_cipher_ctx, out_data, &out_len) != 1) {
57  S2N_ERROR(S2N_ERR_ENCRYPT);
58  }
59 
60  /* write the tag */
61  if (EVP_CIPHER_CTX_ctrl(&key->native_format.evp_cipher_ctx, EVP_CTRL_GCM_GET_TAG, S2N_TLS_GCM_TAG_LEN, tag_data) != 1) {
62  S2N_ERROR(S2N_ERR_ENCRYPT);
63  }
64 
65  return 0;
66 }
