104  /* An alert may be partially written in the outbound
105  * direction, so we keep this as a small 2 byte queue.
106  *
107  * We keep seperate queues for alerts generated by
108  * readers (a response to an alert from a peer) and writers (an
109  * intentional shutdown) so that the s2n reader and writer
110  * can be seperate duplex I/O threads.
111  */
112  uint8_t reader_alert_out_data[S2N_ALERT_LENGTH];

119 
120  /* Our handshake state machine */
121  struct s2n_handshake handshake;
122 
123  uint16_t max_fragment_length;
124 
125  /* Keep some accounting on each connection */
126  uint64_t wire_bytes_in;
127  uint64_t wire_bytes_out;
128 
129  /* Is the connection open or closed ? We use C's only
130  * atomic type as both the reader and the writer threads
131  * may declare a connection closed.
132  *
133  * A connection can be gracefully closed or hard-closed.
134  * When gracefully closed the reader or the writer mark
135  * the connection as closing, and then the writer will
136  * send an alert message before closing the connection
137  * and marking it as closed.
138  *
139  * A hard-close goes straight to closed with no alert
140  * message being sent.
141  */
142  sig_atomic_t closing;
143  sig_atomic_t closed;
144 
145  /* TLS extension data */
146  char server_name[256];
147  char application_protocol[256];
148  /* s2n does not support renegotiation.
149  * RFC5746 Section 4.3 suggests servers implement a minimal version of the
150  * renegotiation_info extension even if renegotiation is not supported.
151  * Some clients may fail the handshake if a corresponding renegotiation_info
152  * extension is not sent back by the server.
153  */
154  unsigned int secure_renegotiation:1;
155 
156  /* OCSP stapling response data */
157  s2n_status_request_type status_type;
158  struct s2n_blob status_response;
