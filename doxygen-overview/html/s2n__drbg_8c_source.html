27 static int s2n_drbg_block_encrypt(EVP_CIPHER_CTX * ctx, uint8_t in[S2N_DRBG_BLOCK_SIZE], uint8_t out[S2N_DRBG_BLOCK_SIZE])

31  S2N_ERROR(S2N_ERR_DRBG);
32  }

38 static int s2n_drbg_bits(struct s2n_drbg *drbg, struct s2n_blob *out)
39 {
40  struct s2n_blob value = {.data = drbg->v,.size = sizeof(drbg->v) };
41  int block_aligned_size = out->size - (out->size % S2N_DRBG_BLOCK_SIZE);
42 
43  /* Per NIST SP800-90A 10.2.1.2: */
44  for (int i = 0; i < block_aligned_size; i += S2N_DRBG_BLOCK_SIZE) {
45  GUARD(s2n_increment_sequence_number(&value));
46  GUARD(s2n_drbg_block_encrypt(&drbg->ctx, drbg->v, out->data + i));
47  drbg->bytes_used += S2N_DRBG_BLOCK_SIZE;
48  }
49 
50  if (out->size <= block_aligned_size) {
51  return 0;
52  }
53 
54  uint8_t spare_block[S2N_DRBG_BLOCK_SIZE];
55  GUARD(s2n_increment_sequence_number(&value));
56  GUARD(s2n_drbg_block_encrypt(&drbg->ctx, drbg->v, spare_block));
57  drbg->bytes_used += S2N_DRBG_BLOCK_SIZE;
58 
59  memcpy_check(out->data + block_aligned_size, spare_block, out->size - block_aligned_size);
60 
61  return 0;
62 }

64 static int s2n_drbg_update(struct s2n_drbg *drbg, struct s2n_blob *provided_data)
65 {
66  uint8_t temp[32];
67  struct s2n_blob temp_blob = {.data = temp,.size = sizeof(temp) };
68 
69  eq_check(provided_data->size, sizeof(temp));
70 
71  GUARD(s2n_drbg_bits(drbg, &temp_blob));
72 
73  /* XOR in the provided data */
74  for (int i = 0; i < provided_data->size; i++) {
75  temp[i] ^= provided_data->data[i];
76  }
77 
78  /* Update the key and value */
79  if (EVP_EncryptInit_ex(&drbg->ctx, EVP_aes_128_ecb(), NULL, temp, NULL) != 1) {
80  S2N_ERROR(S2N_ERR_DRBG);
81  }
82 
83  memcpy_check(drbg->v, temp + S2N_DRBG_BLOCK_SIZE, S2N_DRBG_BLOCK_SIZE);
84 
85  return 0;
86 }

88 int s2n_drbg_seed(struct s2n_drbg *drbg, struct s2n_blob *ps)
89 {
90  uint8_t seed[32];
91  struct s2n_blob blob = {.data = seed,.size = sizeof(seed) };
92  lte_check(ps->size, sizeof(seed));
93 
94  if (drbg->entropy_generator) {
95  GUARD(drbg->entropy_generator(&blob));
96  } else {
97  GUARD(s2n_get_urandom_data(&blob));
98  }
99 
100  for (int i = 0; i < ps->size; i++) {
101  blob.data[i] ^= ps->data[i];
102  }
103 
104  GUARD(s2n_drbg_update(drbg, &blob));
105 
106  drbg->bytes_used = 0;
107  drbg->generation += 1;
108 
109  return 0;
110 }

112 int s2n_drbg_instantiate(struct s2n_drbg *drbg, struct s2n_blob *personalization_string)
113 {
114  struct s2n_blob value = {.data = drbg->v,.size = sizeof(drbg->v) };
115  uint8_t ps_prefix[32];
116  struct s2n_blob ps = {.data = ps_prefix,.size = sizeof(ps_prefix) };
117 
118  /* Start off with zerod data, per 10.2.1.3.1 item 4 */
119  GUARD(s2n_blob_zero(&value));
120 
121  /* Start off with zerod key, per 10.2.1.3.1 item 5 */
122  (void)EVP_CIPHER_CTX_init(&drbg->ctx);
123  if (EVP_EncryptInit_ex(&drbg->ctx, EVP_aes_128_ecb(), NULL, drbg->v, NULL) != 1) {
124  S2N_ERROR(S2N_ERR_DRBG);
125  }
126 
127  /* Copy the personalization string */
128  GUARD(s2n_blob_zero(&ps));
129 
130  memcpy_check(ps.data, personalization_string->data, MIN(ps.size, personalization_string->size));
131 
132  /* Seed / update the DRBG */
133  GUARD(s2n_drbg_seed(drbg, &ps));
134 
135  /* After initial seeding, pivot to RDRAND if available and not overridden */
136  if (drbg->entropy_generator == NULL && s2n_cpu_supports_rdrand()) {
137  drbg->entropy_generator = s2n_get_rdrand_data;
138  }
139  return 0;
140 }

142 int s2n_drbg_generate(struct s2n_drbg *drbg, struct s2n_blob *blob)
143 {
144  uint8_t all_zeros[32] = { 0 };
145  struct s2n_blob zeros = {.data = all_zeros,.size = sizeof(all_zeros) };
146  if (blob->size > S2N_DRBG_GENERATE_LIMIT) {
147  S2N_ERROR(S2N_ERR_DRBG_REQUEST_SIZE);
148  }
149 
150  GUARD(s2n_drbg_seed(drbg, &zeros));
151  GUARD(s2n_drbg_bits(drbg, blob));
152  GUARD(s2n_drbg_update(drbg, &zeros));
153 
154  return 0;
155 }
