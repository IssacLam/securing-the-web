118 int s2n_conn_set_handshake_type(struct s2n_connection *conn)
119 {
120  if (s2n_is_caching_enabled(conn->config)) {
121  if (!s2n_resume_from_cache(conn)) {
122  conn->handshake.handshake_type = RESUME;
123  return 0;
124  }
125 
126  if (conn->mode == S2N_SERVER) {
127  struct s2n_blob session_id = {.data = conn->session_id,.size = S2N_TLS_SESSION_ID_MAX_LEN };
128 
129  /* Generate a new session id */
130  GUARD(s2n_get_public_random_data(&session_id));
131  conn->session_id_len = S2N_TLS_SESSION_ID_MAX_LEN;
132  }
133  }
134 
135  if (conn->secure.cipher_suite->key_exchange_alg->flags & S2N_KEY_EXCHANGE_EPH) {
136  conn->handshake.handshake_type = FULL_WITH_PFS;
137 
138  if (s2n_server_can_send_ocsp(conn)) {
139  conn->handshake.handshake_type = FULL_WITH_PFS_WITH_STATUS;
140  }
141  } else {
142  conn->handshake.handshake_type = FULL_NO_PFS;
143 
144  if (s2n_server_can_send_ocsp(conn)) {
145  conn->handshake.handshake_type = FULL_NO_PFS_WITH_STATUS;
146  }
147  }
148 
149  return 0;
150 }

162 /* Writing is relatively straight forward, simply write each message out as a record,
163  * we may fragment a message across multiple records, but we never coalesce multiple
164  * messages into single records.
165  * Precondition: secure outbound I/O has already been flushed
166  */
167 static int handshake_write_io(struct s2n_connection *conn)
168 {
169  uint8_t record_type = ACTIVE_STATE(conn).record_type;
170  s2n_blocked_status blocked = S2N_NOT_BLOCKED;
171  int max_payload_size;
172 
173  /* Populate handshake.io with header/payload for the current state, once.
174  * Check wiped instead of s2n_stuffer_data_available to differentiate between the initial call
175  * to handshake_write_io and a repeated call after an EWOULDBLOCK.
176  */
177  if (conn->handshake.io.wiped == 1) {
178  if (record_type == TLS_HANDSHAKE) {
179  GUARD(s2n_handshake_write_header(conn, ACTIVE_STATE(conn).message_type));
180  }
181  GUARD(ACTIVE_STATE(conn).handler[conn->mode] (conn));
182  if (record_type == TLS_HANDSHAKE) {
183  GUARD(s2n_handshake_finish_header(conn));
184  }
185  }
186 
187  /* Write the handshake data to records in fragment sized chunks */
188  struct s2n_blob out;
189  while (s2n_stuffer_data_available(&conn->handshake.io) > 0) {
190 
191  GUARD((max_payload_size = s2n_record_max_write_payload_size(conn)));
192  out.size = MIN(s2n_stuffer_data_available(&conn->handshake.io), max_payload_size);
193 
194  out.data = s2n_stuffer_raw_read(&conn->handshake.io, out.size);
195  notnull_check(out.data);
196 
197  /* Make the actual record */
198  GUARD(s2n_record_write(conn, record_type, &out));
199 
200  /* MD5 and SHA sum the handshake data too */
201  if (record_type == TLS_HANDSHAKE) {
202  GUARD(s2n_conn_update_handshake_hashes(conn, &out));
203  }
204 
205  /* Actually send the record. We could block here. Assume the caller will call flush before coming back. */
206  GUARD(s2n_flush(conn, &blocked));
207  }
208 
209  /* We're done sending the last record, reset everything */
210  GUARD(s2n_stuffer_wipe(&conn->out));
211  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
212 
213  /* Advance the state machine */
214  conn->handshake.message_number++;
215 
216  return 0;
217 }
218 
219 /*
220  * Returns:
221  * 1 - more data is needed to complete the handshake message.
222  * 0 - we read the whole handshake message.
223  * -1 - error processing the handshake message.
224  */
225 static int read_full_handshake_message(struct s2n_connection *conn, uint8_t * message_type)
226 {
227  uint32_t current_handshake_data = s2n_stuffer_data_available(&conn->handshake.io);
228  if (current_handshake_data < TLS_HANDSHAKE_HEADER_LENGTH) {
229  /* The message may be so badly fragmented that we don't even read the full header, take
230  * what we can and then continue to the next record read iteration.
231  */
232  if (s2n_stuffer_data_available(&conn->in) < (TLS_HANDSHAKE_HEADER_LENGTH - current_handshake_data)) {
233  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, s2n_stuffer_data_available(&conn->in)));
234  return 1;
235  }
236 
237  /* Get the remainder of the header */
238  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, (TLS_HANDSHAKE_HEADER_LENGTH - current_handshake_data)));
239  }
240 
241  uint32_t handshake_message_length;
242  GUARD(s2n_handshake_parse_header(conn, message_type, &handshake_message_length));
243 
244  if (handshake_message_length > S2N_MAXIMUM_HANDSHAKE_MESSAGE_LENGTH) {
245  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
246  }
247 
248  uint32_t bytes_to_take = handshake_message_length - s2n_stuffer_data_available(&conn->handshake.io);
249  bytes_to_take = MIN(bytes_to_take, s2n_stuffer_data_available(&conn->in));
250 
251  /* If the record is handshake data, add it to the handshake buffer */
252  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, bytes_to_take));
253 
254  /* If we have the whole handshake message, then success */
255  if (s2n_stuffer_data_available(&conn->handshake.io) == handshake_message_length) {
256  struct s2n_blob handshake;
257  handshake.data = conn->handshake.io.blob.data;
258  handshake.size = TLS_HANDSHAKE_HEADER_LENGTH + handshake_message_length;
259 
260  notnull_check(handshake.data);
261 
262  /* MD5 and SHA sum the handshake data too */
263  GUARD(s2n_conn_update_handshake_hashes(conn, &handshake));
264 
265  return 0;
266  }
267 
268  /* We don't have the whole message, so we'll need to go again */
269  GUARD(s2n_stuffer_reread(&conn->handshake.io));
270 
271  return 1;
272 }

274 /* Reading is a little more complicated than writing as the TLS RFCs allow content
275  * types to be interleaved at the record layer. We may get an alert message
276  * during the handshake phase, or messages of types that we don't support (e.g.
277  * HEARTBEAT messages), or during renegotiations we may even get application
278  * data messages that need to be handled by the application. The latter is punted
279  * for now (s2n does support renegotiations).
280  */
281 static int handshake_read_io(struct s2n_connection *conn)
282 {
283  uint8_t record_type;
284  int isSSLv2;
285 
286  GUARD(s2n_read_full_record(conn, &record_type, &isSSLv2));
287 
288  if (isSSLv2) {
289  if (ACTIVE_MESSAGE(conn) != CLIENT_HELLO) {
290  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
291  }
292 
293  /* Add the message to our handshake hashes */
294  struct s2n_blob hashed = {.data = conn->header_in.blob.data + 2,.size = 3 };
295  GUARD(s2n_conn_update_handshake_hashes(conn, &hashed));
296 
297  hashed.data = conn->in.blob.data;
298  hashed.size = s2n_stuffer_data_available(&conn->in);
299  GUARD(s2n_conn_update_handshake_hashes(conn, &hashed));
300 
301  /* Handle an SSLv2 client hello */
302  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, s2n_stuffer_data_available(&conn->in)));
303  GUARD(s2n_sslv2_client_hello_recv(conn));
304  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
305 
306  /* We're done with the record, wipe it */
307  GUARD(s2n_stuffer_wipe(&conn->header_in));
308  GUARD(s2n_stuffer_wipe(&conn->in));
309  conn->in_status = ENCRYPTED;
310 
311  /* Advance the state machine */
312  conn->handshake.message_number++;
313  }
314 
315  /* Now we have a record, but it could be a partial fragment of a message, or it might
316  * contain several messages.
317  */
318  if (record_type == TLS_APPLICATION_DATA) {
319  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
320  } else if (record_type == TLS_CHANGE_CIPHER_SPEC) {
321  if (s2n_stuffer_data_available(&conn->in) != 1) {
322  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
323  }
324 
325  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, s2n_stuffer_data_available(&conn->in)));
326  GUARD(ACTIVE_STATE(conn).handler[conn->mode] (conn));
327  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
328 
329  /* We're done with the record, wipe it */
330  GUARD(s2n_stuffer_wipe(&conn->header_in));
331  GUARD(s2n_stuffer_wipe(&conn->in));
332  conn->in_status = ENCRYPTED;
333 
334  /* Advance the state machine */
335  conn->handshake.message_number++;
336 
337  return 0;
338  } else if (record_type != TLS_HANDSHAKE) {
339  if (record_type == TLS_ALERT) {
340  GUARD(s2n_process_alert_fragment(conn));
341  }
342 
343  /* Ignore record types that we don't support */
344 
345  /* We're done with the record, wipe it */
346  GUARD(s2n_stuffer_wipe(&conn->header_in));
347  GUARD(s2n_stuffer_wipe(&conn->in));
348  conn->in_status = ENCRYPTED;
349  return 0;
350  }
351 
352  /* Record is a handshake message */
353  while (s2n_stuffer_data_available(&conn->in)) {
354  int r;
355  uint8_t handshake_message_type;
356  GUARD((r = read_full_handshake_message(conn, &handshake_message_type)));
357 
358  /* Do we need more data? */
359  if (r == 1) {
360  /* Break out of this inner loop, but since we're not changing the state, the
361  * outer loop in s2n_handshake_io() will read another record.
362  */
363  GUARD(s2n_stuffer_wipe(&conn->header_in));
364  GUARD(s2n_stuffer_wipe(&conn->in));
365  conn->in_status = ENCRYPTED;
366  return 0;
367  }
368 
369  if (handshake_message_type != ACTIVE_STATE(conn).message_type) {
370  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
371  }
372 
373  /* Call the relevant handler */
374  r = ACTIVE_STATE(conn).handler[conn->mode] (conn);
375  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
376 
377  if (r < 0) {
378  GUARD(s2n_connection_kill(conn));
379 
380  return r;
381  }
382 
383  /* Advance the state machine */
384  conn->handshake.message_number++;
385  }
386 
387  /* We're done with the record, wipe it */
388  GUARD(s2n_stuffer_wipe(&conn->header_in));
389  GUARD(s2n_stuffer_wipe(&conn->in));
390  conn->in_status = ENCRYPTED;
391 
392  return 0;
393 }

395 int s2n_negotiate(struct s2n_connection *conn, s2n_blocked_status * blocked)
396 {
397  char this = 'S';
398  if (conn->mode == S2N_CLIENT) {
399  this = 'C';
400  }
401 
402  while (ACTIVE_STATE(conn).writer != 'B') {
403 
404  /* Flush any pending I/O or alert messages */
405  GUARD(s2n_flush(conn, blocked));
406 
407  if (ACTIVE_STATE(conn).writer == this) {
408  *blocked = S2N_BLOCKED_ON_WRITE;
409  GUARD(handshake_write_io(conn));
410  } else {
411  *blocked = S2N_BLOCKED_ON_READ;
412  if (handshake_read_io(conn) < 0) {
413  if (s2n_errno != S2N_ERR_BLOCKED && s2n_is_caching_enabled(conn->config) && conn->session_id_len) {
414  conn->config->cache_delete(conn->config->cache_delete_data, conn->session_id, conn->session_id_len);
415  }
416 
417  return -1;
418  }
419  }
420 
421  /* If the handshake has just ended, free up memory */
422  if (ACTIVE_STATE(conn).writer == 'B') {
423  GUARD(s2n_stuffer_resize(&conn->handshake.io, 0));
424  }
425  }
426 
427  *blocked = S2N_NOT_BLOCKED;
428 
429  return 0;
430 }
