36 int s2n_read_full_record(struct s2n_connection *conn, uint8_t *record_type, int *isSSLv2)
37 {
38  int r;
39 
40  *isSSLv2 = 0;
41 
42  /* If the record has already been decrypted, then leave it alone */
43  if (conn->in_status == PLAINTEXT) {
44  /* Only application data packets count as plaintext */
45  *record_type = TLS_APPLICATION_DATA;
46  return 0;
47  }
48 
49  /* Read the record until we at least have a header */
50  while (s2n_stuffer_data_available(&conn->header_in) < S2N_TLS_RECORD_HEADER_LENGTH) {
51  r = s2n_stuffer_recv_from_fd(&conn->header_in, conn->readfd, S2N_TLS_RECORD_HEADER_LENGTH - s2n_stuffer_data_available(&conn->header_in));
52  if (r == 0) {
53  return -2;
54  }
55  if (r < 0) {
56  return -1;
57  }
58  conn->wire_bytes_in += r;
59  }
60 
61  uint16_t fragment_length;
62 
63  /* If the first bit is set then this is an SSLv2 record */
64  if (conn->header_in.blob.data[0] & 0x80) {
65  conn->header_in.blob.data[0] &= 0x7f;
66  *isSSLv2 = 1;
67 
68  if (s2n_sslv2_record_header_parse(conn, record_type, &conn->client_protocol_version, &fragment_length) < 0) {
69  GUARD(s2n_connection_kill(conn));
70  return -1;
71  }
72  } else {
73  if (s2n_record_header_parse(conn, record_type, &fragment_length) < 0) {
74  GUARD(s2n_connection_kill(conn));
75  return -1;
76  }
77  }
78 
79  /* Read enough to have the whole record */
80  while (s2n_stuffer_data_available(&conn->in) < fragment_length) {
81  r = s2n_stuffer_recv_from_fd(&conn->in, conn->readfd, fragment_length - s2n_stuffer_data_available(&conn->in));
82  if (r == 0) {
83  return -2;
84  }
85  if (r < 0) {
86  return -1;
87  }
88  conn->wire_bytes_in += r;
89  }
90 
91  if (*isSSLv2) {
92  return 0;
93  }
94 
95  /* Decrypt and parse the record */
96  if (s2n_record_parse(conn) < 0) {
97  GUARD(s2n_connection_kill(conn));
98 
99  return -1;
100  }
101 
102  return 0;
103 }

105 ssize_t s2n_recv(struct s2n_connection *conn, void *buf, ssize_t size, s2n_blocked_status *blocked)
106 {
107  ssize_t bytes_read = 0;
108  struct s2n_blob out = {.data = (uint8_t *) buf };
109 
110  if (conn->closed) {
111  return 0;
112  }
113 
114  *blocked = S2N_BLOCKED_ON_READ;
115 
116  while (size && !conn->closed) {
117  int isSSLv2 = 0;
118  uint8_t record_type;
119  errno = 0;
120  int r = s2n_read_full_record(conn, &record_type, &isSSLv2);
121  if (r < 0) {
122  if (errno == EWOULDBLOCK) {
123  if (bytes_read) {
124  return bytes_read;
125  }
126  return -1;
127  }
128  if (r == -2) {
129  conn->closed = 1;
130  GUARD(s2n_connection_wipe(conn));
131  *blocked = S2N_NOT_BLOCKED;
132  return bytes_read;
133  }
134  return -1;
135  }
136 
137  if (isSSLv2) {
138  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
139  }
140 
141  if (record_type != TLS_APPLICATION_DATA) {
142  if (record_type == TLS_ALERT) {
143  GUARD(s2n_process_alert_fragment(conn));
144  GUARD(s2n_flush(conn, blocked));
145  }
146 
147  GUARD(s2n_stuffer_wipe(&conn->header_in));
148  GUARD(s2n_stuffer_wipe(&conn->in));
149  conn->in_status = ENCRYPTED;
150  continue;
151  }
152 
153  out.size = size;
154  if (out.size > s2n_stuffer_data_available(&conn->in)) {
155  out.size = s2n_stuffer_data_available(&conn->in);
156  }
157 
158  GUARD(s2n_stuffer_erase_and_read(&conn->in, &out));
159  bytes_read += out.size;
160 
161  out.data += out.size;
162  size -= out.size;
163 
164  /* Are we ready for more encrypted data? */
165  if (s2n_stuffer_data_available(&conn->in) == 0) {
166  GUARD(s2n_stuffer_wipe(&conn->header_in));
167  GUARD(s2n_stuffer_wipe(&conn->in));
168  conn->in_status = ENCRYPTED;
169  }
170 
171  /* If we've read some data, return it */
172  if (bytes_read) {
173  break;
174  }
175  }
176 
177  if (s2n_stuffer_data_available(&conn->in) == 0) {
178  *blocked = S2N_NOT_BLOCKED;
179  }
180 
181  return bytes_read;
182 }
