33 int s2n_flush(struct s2n_connection *conn, s2n_blocked_status *blocked)
34 {
35  int w;
36 
37  *blocked = S2N_BLOCKED_ON_WRITE;
38 
39  /* Write any data that's already pending */
40  WRITE:
41  while (s2n_stuffer_data_available(&conn->out)) {
42  GUARD(w = s2n_stuffer_send_to_fd(&conn->out, conn->writefd, s2n_stuffer_data_available(&conn->out)));
43  conn->wire_bytes_out += w;
44  }
45  if (conn->closing) {
46  conn->closed = 1;
47  GUARD(s2n_connection_wipe(conn));
48  }
49  GUARD(s2n_stuffer_rewrite(&conn->out));
50 
51  /* If there's an alert pending out, send that */
52  if (s2n_stuffer_data_available(&conn->reader_alert_out) == 2) {
53  struct s2n_blob alert;
54  alert.data = conn->reader_alert_out.blob.data;
55  alert.size = 2;
56  GUARD(s2n_record_write(conn, TLS_ALERT, &alert));
57  GUARD(s2n_stuffer_rewrite(&conn->reader_alert_out));
58  conn->closing = 1;
59 
60  /* Actually write it ... */
61  goto WRITE;
62  }
63 
64  /* Do the same for writer driven alerts */
65  if (s2n_stuffer_data_available(&conn->writer_alert_out) == 2) {
66  struct s2n_blob alert;
67  alert.data = conn->writer_alert_out.blob.data;
68  alert.size = 2;
69  GUARD(s2n_record_write(conn, TLS_ALERT, &alert));
70  GUARD(s2n_stuffer_rewrite(&conn->writer_alert_out));
71  conn->closing = 1;
72 
73  /* Actually write it ... */
74  goto WRITE;
75  }
76 
77  *blocked = S2N_NOT_BLOCKED;
78 
79  return 0;
80 }

82 ssize_t s2n_send(struct s2n_connection *conn, void *buf, ssize_t size, s2n_blocked_status *blocked)
83 {
84  struct s2n_blob in = {.data = buf };
85  ssize_t bytes_written = 0;
86  int max_payload_size;
87  int w;
88 
89  if (conn->closed) {
90  S2N_ERROR(S2N_ERR_CLOSED);
91  }
92 
93  /* Flush any pending I/O */
94  GUARD(s2n_flush(conn, blocked));
95 
96  *blocked = S2N_BLOCKED_ON_WRITE;
97 
98  GUARD((max_payload_size = s2n_record_max_write_payload_size(conn)));
99 
100  /* TLS 1.0 and SSLv3 are vulnerable to the so-called Beast attack. Work
101  * around this by splitting messages into one byte records, and then
102  * the remainder can follow as usual.
103  */
104  int cbcHackUsed = 0;
105 
106  /* Now write the data we were asked to send this round */
107  while (size) {
108  in.size = size;
109  if (in.size > max_payload_size) {
110  in.size = max_payload_size;
111  }
112 
113  if (conn->actual_protocol_version < S2N_TLS11 && conn->active.cipher_suite->cipher->type == S2N_CBC) {
114  if (in.size > 1 && cbcHackUsed == 0) {
115  in.size = 1;
116  cbcHackUsed = 1;
117  }
118  }
119 
120  /* Write and encrypt the record */
121  GUARD(s2n_stuffer_rewrite(&conn->out));
122  GUARD(s2n_record_write(conn, TLS_APPLICATION_DATA, &in));
123 
124  bytes_written += in.size;
125 
126  /* Send it */
127  while (s2n_stuffer_data_available(&conn->out)) {
128  errno = 0;
129  w = s2n_stuffer_send_to_fd(&conn->out, conn->writefd, s2n_stuffer_data_available(&conn->out));
130  if (w < 0) {
131  if (errno == EWOULDBLOCK) {
132  return bytes_written;
133  }
134  return -1;
135  }
136  conn->wire_bytes_out += w;
137  }
138 
139  in.data += in.size;
140  size -= in.size;
141  }
142 
143  *blocked = S2N_NOT_BLOCKED;
144 
145  return bytes_written;
146 }
