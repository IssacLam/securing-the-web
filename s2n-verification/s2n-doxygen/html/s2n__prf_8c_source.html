35 static int s2n_sslv3_prf(union s2n_prf_working_space *ws, struct s2n_blob *secret, struct s2n_blob *seed_a, struct s2n_blob *seed_b, struct s2n_blob *out)
36 {
37  struct s2n_hash_state *md5 = &ws->ssl3.md5;
38  struct s2n_hash_state *sha1 = &ws->ssl3.sha1;
39 
40  uint32_t outputlen = out->size;
41  uint8_t *output = out->data;
42  uint8_t iteration = 1;
43 
44  uint8_t A = 'A';
45  while (outputlen) {
46  GUARD(s2n_hash_init(sha1, S2N_HASH_SHA1));
47 
48  for (int i = 0; i < iteration; i++) {
49  GUARD(s2n_hash_update(sha1, &A, 1));
50  }
51 
52  GUARD(s2n_hash_update(sha1, secret->data, secret->size));
53  GUARD(s2n_hash_update(sha1, seed_a->data, seed_a->size));
54 
55  if (seed_b) {
56  GUARD(s2n_hash_update(sha1, seed_b->data, seed_b->size));
57  }
58 
59  GUARD(s2n_hash_digest(sha1, ws->ssl3.sha1_digest, sizeof(ws->ssl3.sha1_digest)));
60  GUARD(s2n_hash_init(md5, S2N_HASH_MD5));
61  GUARD(s2n_hash_update(md5, secret->data, secret->size));
62  GUARD(s2n_hash_update(md5, ws->ssl3.sha1_digest, sizeof(ws->ssl3.sha1_digest)));
63  GUARD(s2n_hash_digest(md5, ws->ssl3.md5_digest, sizeof(ws->ssl3.md5_digest)));
64 
65  uint32_t bytes_to_copy = outputlen;
66  if (bytes_to_copy > sizeof(ws->ssl3.md5_digest)) {
67  bytes_to_copy = sizeof(ws->ssl3.md5_digest);
68  }
69 
70  memcpy_check(output, ws->ssl3.md5_digest, bytes_to_copy);
71 
72  outputlen -= bytes_to_copy;
73  output += bytes_to_copy;
74 
75  /* Increment the letter */
76  A++;
77  iteration++;
78  }
79 
80  return 0;
81 }

83 static int s2n_p_hash(union s2n_prf_working_space *ws, s2n_hmac_algorithm alg, struct s2n_blob *secret,
84  struct s2n_blob *label, struct s2n_blob *seed_a, struct s2n_blob *seed_b, struct s2n_blob *out)
85 {
86  struct s2n_hmac_state *hmac = &ws->tls.hmac;
87  uint32_t digest_size = s2n_hmac_digest_size(alg);
88 
89  /* First compute hmac(secret + A(0)) */
90  GUARD(s2n_hmac_init(hmac, alg, secret->data, secret->size));
91  GUARD(s2n_hmac_update(hmac, label->data, label->size));
92  GUARD(s2n_hmac_update(hmac, seed_a->data, seed_a->size));
93 
94  if (seed_b) {
95  GUARD(s2n_hmac_update(hmac, seed_b->data, seed_b->size));
96  }
97  GUARD(s2n_hmac_digest(hmac, ws->tls.digest0, digest_size));
98 
99  uint32_t outputlen = out->size;
100  uint8_t *output = out->data;
101 
102  while (outputlen) {
103  /* Now compute hmac(secret + A(N - 1) + seed) */
104  GUARD(s2n_hmac_reset(hmac));
105  GUARD(s2n_hmac_update(hmac, ws->tls.digest0, digest_size));
106 
107  /* Add the label + seed and compute this round's A */
108  GUARD(s2n_hmac_update(hmac, label->data, label->size));
109  GUARD(s2n_hmac_update(hmac, seed_a->data, seed_a->size));
110  if (seed_b) {
111  GUARD(s2n_hmac_update(hmac, seed_b->data, seed_b->size));
112  }
113  GUARD(s2n_hmac_digest(hmac, ws->tls.digest1, digest_size));
114 
115  uint32_t bytes_to_xor = outputlen;
116  if (bytes_to_xor > digest_size) {
117  bytes_to_xor = digest_size;
118  }
119 
120  for (int i = 0; i < bytes_to_xor; i++) {
121  *output ^= ws->tls.digest1[i];
122  output++;
123  outputlen--;
124  }
125 
126  /* Stash a digest of A(N), in A(N), for the next round */
127  GUARD(s2n_hmac_reset(hmac));
128  GUARD(s2n_hmac_update(hmac, ws->tls.digest0, digest_size));
129  GUARD(s2n_hmac_digest(hmac, ws->tls.digest0, digest_size));
130  }
131 
132  return 0;
133 }

135 static int s2n_prf(struct s2n_connection *conn, struct s2n_blob *secret, struct s2n_blob *label, struct s2n_blob *seed_a, struct s2n_blob *seed_b, struct s2n_blob *out)
136 {
137  if (conn->actual_protocol_version == S2N_SSLv3) {
138  return s2n_sslv3_prf(&conn->prf_space, secret, seed_a, seed_b, out);
139  }
140 
141  /* We zero the out blob because p_hash works by XOR'ing with the existing
142  * buffer. This is a little convuloted but means we can avoid dynamic memory
143  * allocation. When we call p_hash once (in the TLS1.2 case) it will produce
144  * the right values. When we call it twice in the regular case, the two
145  * outputs will be XORd just ass the TLS 1.0 and 1.1 RFCs require.
146  */
147  GUARD(s2n_blob_zero(out));
148 
149  if (conn->actual_protocol_version == S2N_TLS12) {
150  return s2n_p_hash(&conn->prf_space, S2N_HMAC_SHA256, secret, label, seed_a, seed_b, out);
151  }
152 
153  struct s2n_blob half_secret = {.data = secret->data,.size = (secret->size + 1) / 2 };
154 
155  GUARD(s2n_p_hash(&conn->prf_space, S2N_HMAC_MD5, &half_secret, label, seed_a, seed_b, out));
156  half_secret.data += secret->size - half_secret.size;
157  GUARD(s2n_p_hash(&conn->prf_space, S2N_HMAC_SHA1, &half_secret, label, seed_a, seed_b, out));
158 
159  return 0;
160 }

263 
264  GUARD(s2n_hash_digest(&conn->handshake.client_md5, md5_digest, MD5_DIGEST_LENGTH));
265  GUARD(s2n_hash_digest(&conn->handshake.client_sha1, sha_digest, SHA_DIGEST_LENGTH));
266  md5.data = md5_digest;
267  md5.size = MD5_DIGEST_LENGTH;
268  sha.data = sha_digest;

312 int s2n_prf_key_expansion(struct s2n_connection *conn)
313 {
314  struct s2n_blob client_random = {.data = conn->pending.client_random,.size = sizeof(conn->pending.client_random) };
315  struct s2n_blob server_random = {.data = conn->pending.server_random,.size = sizeof(conn->pending.server_random) };
316  struct s2n_blob master_secret = {.data = conn->pending.master_secret,.size = sizeof(conn->pending.master_secret) };
317  struct s2n_blob label, out;
318  uint8_t key_expansion_label[] = "key expansion";
319  uint8_t key_block[128];
320 
321  label.data = key_expansion_label;
322  label.size = sizeof(key_expansion_label) - 1;
323  out.data = key_block;
324  out.size = sizeof(key_block);
325 
326  struct s2n_stuffer key_material;
327  GUARD(s2n_prf(conn, &master_secret, &label, &server_random, &client_random, &out));
328  GUARD(s2n_stuffer_init(&key_material, &out));
329  GUARD(s2n_stuffer_write(&key_material, &out));
330 
331  /* What's our hmac algorithm? */
332  s2n_hmac_algorithm hmac_alg = conn->pending.cipher_suite->hmac_alg;
333  if (conn->actual_protocol_version == S2N_SSLv3) {
334  if (hmac_alg == S2N_HMAC_SHA1) {
335  hmac_alg = S2N_HMAC_SSLv3_SHA1;
336  } else if (hmac_alg == S2N_HMAC_MD5) {
337  hmac_alg = S2N_HMAC_SSLv3_MD5;
338  } else {
339  S2N_ERROR(S2N_ERR_HMAC_INVALID_ALGORITHM);
340  }
341  }
342 
343  /* Check that we have a valid MAC and key size */
344  int mac_size;
345  GUARD((mac_size = s2n_hmac_digest_size(hmac_alg)));
346 
347  /* Seed the client MAC */
348  uint8_t *client_write_mac_key = s2n_stuffer_raw_read(&key_material, mac_size);
349  notnull_check(client_write_mac_key);
350  GUARD(s2n_hmac_init(&conn->pending.client_record_mac, hmac_alg, client_write_mac_key, mac_size));
351 
352  /* Seed the server MAC */
353  uint8_t *server_write_mac_key = s2n_stuffer_raw_read(&key_material, mac_size);
354  notnull_check(server_write_mac_key);
355  GUARD(s2n_hmac_init(&conn->pending.server_record_mac, hmac_alg, server_write_mac_key, mac_size));
356 
357  /* Make the client key */
358  struct s2n_blob client_key;
359  client_key.size = conn->pending.cipher_suite->cipher->key_material_size;
360  client_key.data = s2n_stuffer_raw_read(&key_material, client_key.size);
361  notnull_check(client_key.data);
362  if (conn->mode == S2N_CLIENT) {
363  GUARD(conn->pending.cipher_suite->cipher->get_encryption_key(&conn->pending.client_key, &client_key));
364  } else {
365  GUARD(conn->pending.cipher_suite->cipher->get_decryption_key(&conn->pending.client_key, &client_key));
366  }
367 
368  /* Make the server key */
369  struct s2n_blob server_key;
370  server_key.size = conn->pending.cipher_suite->cipher->key_material_size;
371  server_key.data = s2n_stuffer_raw_read(&key_material, server_key.size);
372  notnull_check(server_key.data);
373  if (conn->mode == S2N_SERVER) {
374  GUARD(conn->pending.cipher_suite->cipher->get_encryption_key(&conn->pending.server_key, &server_key));
375  } else {
376  GUARD(conn->pending.cipher_suite->cipher->get_decryption_key(&conn->pending.server_key, &server_key));
377  }
378 
379  /* TLS >= 1.1 has no implicit IVs for non AEAD ciphers */
380  if (conn->actual_protocol_version > S2N_TLS10 &&
381  conn->pending.cipher_suite->cipher->type != S2N_AEAD) {
382  return 0;
383  }
384 
385  if (conn->pending.cipher_suite->cipher->type == S2N_AEAD) {
386  /* Generate the IVs */
387  struct s2n_blob client_implicit_iv;
388  client_implicit_iv.data = conn->pending.client_implicit_iv;
389  client_implicit_iv.size = conn->pending.cipher_suite->cipher->io.aead.fixed_iv_size;
390  GUARD(s2n_stuffer_read(&key_material, &client_implicit_iv));
391 
392  struct s2n_blob server_implicit_iv;
393  server_implicit_iv.data = conn->pending.server_implicit_iv;
394  server_implicit_iv.size = conn->pending.cipher_suite->cipher->io.aead.fixed_iv_size;
395  GUARD(s2n_stuffer_read(&key_material, &server_implicit_iv));
396  } else if (conn->pending.cipher_suite->cipher->type == S2N_CBC) {
397  /* Generate the IVs */
398  struct s2n_blob client_implicit_iv;
399  client_implicit_iv.data = conn->pending.client_implicit_iv;
400  client_implicit_iv.size = conn->pending.cipher_suite->cipher->io.cbc.block_size;
401  GUARD(s2n_stuffer_read(&key_material, &client_implicit_iv));
402 
403  struct s2n_blob server_implicit_iv;
404  server_implicit_iv.data = conn->pending.server_implicit_iv;
405  server_implicit_iv.size = conn->pending.cipher_suite->cipher->io.cbc.block_size;
406  GUARD(s2n_stuffer_read(&key_material, &server_implicit_iv));
407  }
408 
409  return 0;
410 }
