80 int s2n_stuffer_resize(struct s2n_stuffer *stuffer, const uint32_t size)
81 {
82  if (stuffer->growable == 0) {
83  S2N_ERROR(S2N_ERR_RESIZE_STATIC_STUFFER);
84  }
85  if (stuffer->tainted == 1) {
86  S2N_ERROR(S2N_ERR_RESIZE_TAINTED_STUFFER);
87  }
88  if (size == stuffer->blob.size) {
89  return 0;
90  }
91  if (size < stuffer->blob.size) {
92  GUARD(s2n_stuffer_wipe_n(stuffer, stuffer->blob.size - size));
93  }
94 
95  GUARD(s2n_realloc(&stuffer->blob, size));
96 
97  stuffer->blob.size = size;
98 
99  return 0;
100 }

117 int s2n_stuffer_wipe_n(struct s2n_stuffer *stuffer, const uint32_t size)
118 {
119  uint32_t n = size;
120  if (stuffer->write_cursor < n) {
121  n = stuffer->write_cursor;
122  }
123 
124  /* Use '0' instead of 0 precisely to prevent C string compatibility */
125  memset_check(stuffer->blob.data + stuffer->write_cursor - n, '0', n);
126  stuffer->write_cursor -= n;
127 
128  if (stuffer->write_cursor == 0) {
129  stuffer->wiped = 1;
130  }
131  if (stuffer->write_cursor < stuffer->read_cursor) {
132  stuffer->read_cursor = stuffer->write_cursor;
133  }
134 
135  return 0;
136 }

197 int s2n_stuffer_skip_write(struct s2n_stuffer *stuffer, const uint32_t n)
198 {
199  if (s2n_stuffer_space_remaining(stuffer) < n) {
200  if (stuffer->growable) {
201  /* Always grow a stuffer by at least 1k */
202  uint32_t growth = n;
203  if (growth < 1024) {
204  growth = 1024;
205  }
206  GUARD(s2n_stuffer_resize(stuffer, stuffer->blob.size + growth));
207  } else {
208  S2N_ERROR(S2N_ERR_STUFFER_IS_FULL);
209  }
210  }
211 
212  stuffer->write_cursor += n;
213  stuffer->wiped = 0;
214  return 0;
215 }
