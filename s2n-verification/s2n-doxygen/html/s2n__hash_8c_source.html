115 int s2n_hash_digest(struct s2n_hash_state *state, void *out, uint32_t size)
116 {
117  int r;
118  switch (state->alg) {
119  case S2N_HASH_NONE:
120  r = 1;
121  break;
122  case S2N_HASH_MD5:
123  eq_check(size, MD5_DIGEST_LENGTH);
124  r = MD5_Final(out, &state->hash_ctx.md5);
125  break;
126  case S2N_HASH_SHA1:
127  eq_check(size, SHA_DIGEST_LENGTH);
128  r = SHA1_Final(out, &state->hash_ctx.sha1);
129  break;
130  case S2N_HASH_SHA224:
131  eq_check(size, SHA224_DIGEST_LENGTH);
132  r = SHA224_Final(out, &state->hash_ctx.sha224);
133  break;
134  case S2N_HASH_SHA256:
135  eq_check(size, SHA256_DIGEST_LENGTH);
136  r = SHA256_Final(out, &state->hash_ctx.sha256);
137  break;
138  case S2N_HASH_SHA384:
139  eq_check(size, SHA384_DIGEST_LENGTH);
140  r = SHA384_Final(out, &state->hash_ctx.sha384);
141  break;
142  case S2N_HASH_SHA512:
143  eq_check(size, SHA512_DIGEST_LENGTH);
144  r = SHA512_Final(out, &state->hash_ctx.sha512);
145  break;
146  case S2N_HASH_MD5_SHA1:
147  eq_check(size, MD5_DIGEST_LENGTH + SHA_DIGEST_LENGTH);
148  r = SHA1_Final(((uint8_t *) out) + MD5_DIGEST_LENGTH, &state->hash_ctx.md5_sha1.sha1);
149  r &= MD5_Final(out, &state->hash_ctx.md5_sha1.md5);
150  break;
151  default:
152  S2N_ERROR(S2N_ERR_HASH_INVALID_ALGORITHM);
153  }
154 
155  if (r == 0) {
156  S2N_ERROR(S2N_ERR_HASH_DIGEST_FAILED);
157  }
158 
159  return 0;
160 }
