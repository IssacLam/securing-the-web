55 int main(int argc, char * const *argv)
56 {
57  struct addrinfo hints, *ai_list, *ai;
58  int r, sockfd = 0;
59  /* Optional args */
60  const char *alpn_protocols = NULL;
61  const char *server_name = NULL;
62  s2n_status_request_type type = S2N_STATUS_REQUEST_NONE;
63  /* required args */
64  const char *host = NULL;
65  const char *port = "443";
66 
67  static struct option long_options[] = {
68  { "alpn", required_argument, 0, 'a' },
69  { "help", no_argument, 0, 'h' },
70  { "name", required_argument, 0, 'n' },
71  { "status", no_argument, 0, 's' },
72  };
73  while (1) {
74  int option_index = 0;
75  int c = getopt_long (argc, argv, "a:hn:s", long_options, &option_index);
76  if (c == -1) {
77  break;
78  }
79  switch (c) {
80  case 'a':
81  alpn_protocols = optarg;
82  break;
83  case 'h':
84  usage();
85  break;
86  case 'n':
87  server_name = optarg;
88  break;
89  case 's':
90  type = S2N_STATUS_REQUEST_OCSP;
91  break;
92  case '?':
93  default:
94  usage();
95  break;
96  }
97  }
98 
99  if (optind < argc) {
100  host = argv[optind++];
101  }
102  if (optind < argc) {
103  port = argv[optind++];
104  }
105 
106  if (!host) {
107  usage();
108  }
109 
110  if (!server_name) {
111  server_name = host;
112  }
113 
114  memset(&hints, 0, sizeof(hints));
115 
116  hints.ai_family = AF_UNSPEC;
117  hints.ai_socktype = SOCK_STREAM;
118 
119  if (signal(SIGPIPE, SIG_IGN) == SIG_ERR) {
120  fprintf(stderr, "Error disabling SIGPIPE\n");
121  exit(1);
122  }
123 
124  if ((r = getaddrinfo(host, port, &hints, &ai_list)) != 0) {
125  fprintf(stderr, "error: %s\n", gai_strerror(r));
126  exit(1);
127  }
128 
129  int connected = 0;
130  for (ai = ai_list; ai != NULL; ai = ai->ai_next) {
131  if ((sockfd = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol)) == -1) {
132  continue;
133  }
134 
135  if (connect(sockfd, ai->ai_addr, ai->ai_addrlen) == -1) {
136  close(sockfd);
137  continue;
138  }
139 
140  connected = 1;
141  /* connect() succeeded */
142  break;
143  }
144 
145  freeaddrinfo(ai_list);
146 
147  if (connected == 0) {
148  fprintf(stderr, "Failed to connect to %s:%s\n", argv[1], port);
149  close(sockfd);
150  exit(1);
151  }
152 
153  if (s2n_init() < 0) {
154  fprintf(stderr, "Error running s2n_init(): '%s'\n", s2n_strerror(s2n_errno, "EN"));
155  }
156 
157  struct s2n_config *config = s2n_config_new();
158  if (config == NULL) {
159  fprintf(stderr, "Error getting new config: '%s'\n", s2n_strerror(s2n_errno, "EN"));
160  exit(1);
161  }
162 
163  if (s2n_config_set_status_request_type(config, type) < 0) {
164  fprintf(stderr, "Error setting status request type: '%s'\n", s2n_strerror(s2n_errno, "EN"));
165  exit(1);
166  }
167 
168  if (alpn_protocols) {
169  /* Count the number of commas, this tells us how many protocols there
170  are in the list */
171  const char *ptr = alpn_protocols;
172  int protocol_count = 1;
173  while (*ptr) {
174  if (*ptr == ',') {
175  protocol_count++;
176  }
177  ptr++;
178  }
179 
180  char **protocols = malloc(sizeof(char *) * protocol_count);
181  if (!protocols) {
182  fprintf(stderr, "Error allocating memory\n");
183  exit(1);
184  }
185 
186  const char *next = alpn_protocols;
187  int index = 0;
188  int length = 0;
189  ptr = alpn_protocols;
190  while (*ptr) {
191  if (*ptr == ',') {
192  protocols[index] = malloc(length + 1);
193  if (!protocols[index]) {
194  fprintf(stderr, "Error allocating memory\n");
195  exit(1);
196  }
197  memcpy(protocols[index], next, length);
198  protocols[index][length] = '\0';
199  length = 0;
200  index++;
201  ptr++;
202  next = ptr;
203  } else {
204  length++;
205  ptr++;
206  }
207  }
208  if (ptr != next) {
209  protocols[index] = malloc(length + 1);
210  if (!protocols[index]) {
211  fprintf(stderr, "Error allocating memory\n");
212  exit(1);
213  }
214  memcpy(protocols[index], next, length);
215  protocols[index][length] = '\0';
216  }
217  if (s2n_config_set_protocol_preferences(config, (const char * const *)protocols, protocol_count) < 0) {
218  fprintf(stderr, "Failed to set protocol preferences: '%s'\n", s2n_strerror(s2n_errno, "EN"));
219  exit(1);
220  }
221  while(protocol_count) {
222  protocol_count--;
223  free(protocols[protocol_count]);
224  }
225  free(protocols);
226  }
227 
228  struct s2n_connection *conn = s2n_connection_new(S2N_CLIENT);
229 
230  if (conn == NULL) {
231  fprintf(stderr, "Error getting new connection: '%s'\n", s2n_strerror(s2n_errno, "EN"));
232  exit(1);
233  }
234 
235  printf("Connected to %s:%s\n", host, port);
236 
237  if (s2n_connection_set_config(conn, config) < 0) {
238  fprintf(stderr, "Error setting configuration: '%s'\n", s2n_strerror(s2n_errno, "EN"));
239  exit(1);
240  }
241 
242  if (s2n_set_server_name(conn, server_name) < 0) {
243  fprintf(stderr, "Error setting server name: '%s'\n", s2n_strerror(s2n_errno, "EN"));
244  exit(1);
245  }
246 
247  if (s2n_connection_set_fd(conn, sockfd) < 0) {
248  fprintf(stderr, "Error setting file descriptor: '%s'\n", s2n_strerror(s2n_errno, "EN"));
249  exit(1);
250  }
251 
252  /* See echo.c */
253  echo(conn, sockfd);
254 
255  if (s2n_connection_free(conn) < 0) {
256  fprintf(stderr, "Error freeing connection: '%s'\n", s2n_strerror(s2n_errno, "EN"));
257  exit(1);
258  }
259 
260  if (s2n_config_free(config) < 0) {
261  fprintf(stderr, "Error freeing configuration: '%s'\n", s2n_strerror(s2n_errno, "EN"));
262  exit(1);
263  }
264 
265  if (s2n_cleanup() < 0) {
266  fprintf(stderr, "Error running s2n_cleanup(): '%s'\n", s2n_strerror(s2n_errno, "EN"));
267  }
268 
269  return 0;
270 }
