52 int s2n_ecc_write_ecc_params(struct s2n_ecc_params *server_ecc_params, struct s2n_stuffer *out, struct s2n_blob *written)
53 {
54  uint8_t point_len;
55  struct s2n_blob point;
56 
57  /* Remember when the written data starts */
58  written->data = s2n_stuffer_raw_write(out, 0);
59  notnull_check(written->data);
60 
61  GUARD(s2n_stuffer_write_uint8(out, TLS_EC_CURVE_TYPE_NAMED));
62  GUARD(s2n_stuffer_write_uint16(out, server_ecc_params->negotiated_curve->iana_id));
63 
64  /* Precalculate point length */
65  GUARD(s2n_ecc_calculate_point_length(EC_KEY_get0_public_key(server_ecc_params->ec_key), EC_KEY_get0_group(server_ecc_params->ec_key), &point_len));
66 
67  /* Write point length */
68  GUARD(s2n_stuffer_write_uint8(out, point_len));
69 
70  /* Write the point */
71  point.data = s2n_stuffer_raw_write(out, point_len);
72  point.size = point_len;
73  notnull_check(point.data);
74  GUARD(s2n_ecc_write_point_data_snug(EC_KEY_get0_public_key(server_ecc_params->ec_key), EC_KEY_get0_group(server_ecc_params->ec_key), &point));
75 
76  written->size = 3 + (1 + point_len);
77 
78  return 0;
79 }

81 int s2n_ecc_read_ecc_params(struct s2n_ecc_params *server_ecc_params, struct s2n_stuffer *in, struct s2n_blob *read)
82 {
83  uint8_t curve_type;
84  uint8_t point_length;
85  struct s2n_blob point_blob, curve_blob;
86  EC_POINT *point;
87 
88  /* Remember where we started reading the data */
89  read->data = s2n_stuffer_raw_read(in, 0);
90  notnull_check(read->data);
91 
92  /* Read the curve */
93  GUARD(s2n_stuffer_read_uint8(in, &curve_type));
94  if (curve_type != TLS_EC_CURVE_TYPE_NAMED) {
95  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
96  }
97  curve_blob.data = s2n_stuffer_raw_read(in, 2);
98  curve_blob.size = 2;
99  /* Verify that the client supports the server curve */
100  if (s2n_ecc_find_supported_curve(&curve_blob, &server_ecc_params->negotiated_curve) != 0) {
101  S2N_ERROR(S2N_ERR_ECDHE_UNSUPPORTED_CURVE);
102  }
103  /* Create a key to store the server public point */
104  server_ecc_params->ec_key = EC_KEY_new_by_curve_name(server_ecc_params->negotiated_curve->libcrypto_nid);
105  if (server_ecc_params->ec_key == NULL) {
106  S2N_ERROR(S2N_ERR_ECDHE_UNSUPPORTED_CURVE);
107  }
108 
109  /* Read the point */
110  GUARD(s2n_stuffer_read_uint8(in, &point_length));
111  point_blob.size = point_length;
112  point_blob.data = s2n_stuffer_raw_read(in, point_blob.size);
113  notnull_check(point_blob.data);
114 
115  /* Parse and store the server public point */
116  point = s2n_ecc_blob_to_point(&point_blob, server_ecc_params->ec_key);
117  if (point == NULL) {
118  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
119  }
120  if (EC_KEY_set_public_key(server_ecc_params->ec_key, point) != 1) {
121  EC_POINT_free(point);
122  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
123  }
124  EC_POINT_free(point);
125 
126  read->size = 3 + (1 + point_length);
127 
128  return 0;
129 }

131 int s2n_ecc_compute_shared_secret_as_server(struct s2n_ecc_params *server_ecc_params, struct s2n_stuffer *Yc_in, struct s2n_blob *shared_key)
132 {

137

139  client_public_blob.size = client_public_len;
140  client_public_blob.data = s2n_stuffer_raw_read(Yc_in, client_public_blob.size);
141  notnull_check(client_public_blob.data);
142 
143  /* Parse the client public */

146  S2N_ERROR(S2N_ERR_BAD_MESSAGE);

151  EC_POINT_free(client_public);

155 int s2n_ecc_compute_shared_secret_as_client(struct s2n_ecc_params *server_ecc_params, struct s2n_stuffer *Yc_out, struct s2n_blob *shared_key)
156 {
157  EC_KEY *client_key;
158 
159  /* Generate the client key. Don't forget to free it. */
160  notnull_check(server_ecc_params->negotiated_curve);
161  client_key = s2n_ecc_generate_own_key(server_ecc_params->negotiated_curve);
162  if (client_key == NULL) {
163  S2N_ERROR(S2N_ERR_ECDHE_GEN_KEY);
164  }
165 
166  /* Compute the shared secret */
167  if (s2n_ecc_compute_shared_secret(client_key, EC_KEY_get0_public_key(server_ecc_params->ec_key), shared_key) != 0) {
168  EC_KEY_free(client_key);
169  S2N_ERROR(S2N_ERR_ECDHE_SHARED_SECRET);
170  }
171 
172  /* Write the client public to Yc */
173  if (s2n_ecc_write_point_with_length(EC_KEY_get0_public_key(client_key), EC_KEY_get0_group(client_key), Yc_out) != 0) {
174  EC_KEY_free(client_key);
175  S2N_ERROR(S2N_ERR_ECDHE_SERIALIZING);
176  }
177  EC_KEY_free(client_key);
178 
179  return 0;
180 }

204 static EC_POINT *s2n_ecc_blob_to_point(struct s2n_blob *blob, const EC_KEY *ec_key)
205 {
206  const EC_GROUP *group = EC_KEY_get0_group(ec_key);
207  EC_POINT *point = EC_POINT_new(group);
208  if (point == NULL) {
209  S2N_ERROR_PTR(S2N_ERR_ECDHE_UNSUPPORTED_CURVE);
210  }
211  if (EC_POINT_oct2point(group, point, blob->data, blob->size, NULL) != 1) {
212  EC_POINT_free(point);
213  S2N_ERROR_PTR(S2N_ERR_BAD_MESSAGE);
214  }
215  return point;
216 }

218 static int s2n_ecc_calculate_point_length(const EC_POINT *point, const EC_GROUP *group, uint8_t *length)
219 {
220  size_t ret = EC_POINT_point2oct(group, point, POINT_CONVERSION_UNCOMPRESSED, NULL, 0, NULL);
221  if (ret == 0) {
222  S2N_ERROR(S2N_ERR_ECDHE_SERIALIZING);
223  }
224  if (ret > UINT8_MAX) {
225  S2N_ERROR(S2N_ERR_ECDHE_SERIALIZING);
226  }
227  *length = (uint8_t) ret;
228  return 0;

258 static int s2n_ecc_compute_shared_secret(EC_KEY *own_key, const EC_POINT *peer_public, struct s2n_blob *shared_secret)
259 {
260  int field_degree;
261  int shared_secret_size;
262 
263  field_degree = EC_GROUP_get_degree(EC_KEY_get0_group(own_key));
264  if (field_degree <= 0) {
265  S2N_ERROR(S2N_ERR_ECDHE_SHARED_SECRET);
266  }
267 
268  shared_secret_size = (field_degree + 7) / 8;
269  GUARD(s2n_alloc(shared_secret, shared_secret_size));
270 
271  if (ECDH_compute_key(shared_secret->data, shared_secret_size, peer_public, own_key, NULL) != shared_secret_size) {
272  S2N_ERROR(S2N_ERR_ECDHE_SHARED_SECRET);
273  }
274 
275  return 0;
276 }

278 int s2n_ecc_find_supported_curve(struct s2n_blob *iana_ids, const struct s2n_ecc_named_curve **found)
279 {
280  struct s2n_stuffer iana_ids_in;
281 
282  GUARD(s2n_stuffer_init(&iana_ids_in, iana_ids));
283  GUARD(s2n_stuffer_write(&iana_ids_in, iana_ids));
284  for (int i = 0; i < sizeof(s2n_ecc_supported_curves) / sizeof(s2n_ecc_supported_curves[0]); i++) {
285  const struct s2n_ecc_named_curve *supported_curve = &s2n_ecc_supported_curves[i];
286  for (int j = 0; j < iana_ids->size / 2; j++) {
287  uint16_t iana_id;
288  GUARD(s2n_stuffer_read_uint16(&iana_ids_in, &iana_id));
289 
290  if (supported_curve->iana_id == iana_id) {
291  *found = supported_curve;
292  return 0;
293  }
294  }
295  GUARD(s2n_stuffer_reread(&iana_ids_in));
296  }
297 
298  /* Nothing found */
299  S2N_ERROR(S2N_ERR_ECDHE_UNSUPPORTED_CURVE);
300 }
