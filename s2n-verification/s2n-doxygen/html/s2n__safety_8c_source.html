40 int s2n_constant_time_equals(const uint8_t *a, const uint8_t *b, uint32_t len)
41 {
42 // __CPROVER_assume(a != NULL && b != NULL && len > 0);
43  uint8_t xor = 0;
44  for (int i = 0; i < len; i++) {
45  xor |= a[i] ^ b[i];
46  }
47 // __CPROVER_assert((xor > 0 && __CPROVER_exists { int j; (j >= 0 && j < len) ==> a[j] != b[j]}) ||
48 // ( xor == 0 && __CPROVER_forall { int j; (j >= 0 && j < len) ==> a[j] == b[j] }), "ERROR: s2n_constant_time_equals");

52 int s2n_constant_time_copy_or_dont(uint8_t *a, const uint8_t *b, uint32_t len, uint8_t dont)
53 {
54 // __CPROVER_assume(a != NULL && b != NULL && len > 0 && dont >= 0);
55  uint8_t mask = ~(0xff << ((!dont) * 8));
56 
57  __CPROVER_assert((dont > 0 && mask == 0x00) || (dont == 0 && mask == 0xff), "ERROR: mask");
58  /* dont = 0 : mask = 0xff */
59  /* dont > 0 : mask = 0x00 */
60 
61  for (int i = 0; i < len; i++) {
62  uint8_t old = a[i];
63  uint8_t diff = (old ^ b[i]) & mask;
64  a[i] = old ^ diff;
65  }
66 // __CPROVER_assert(dont > 0 || (dont == 0 && __CPROVER_forall { int j; (j >= 0 && j < len) ==> a[j] == b[j]}), "ERROR: s2n_constant_time_copy_or_dont");
67  return 0;
68 }
