105 const char *s2n_strerror(int error, const char *lang)
106 {
107  const char *no_such_language = "Language is not supported for error translation";
108  const char *no_such_error = "Internal s2n error";
109 
110  if (lang == NULL) {
111  lang = "EN";
112  }
113 
114  if (strcasecmp(lang, "EN")) {
115  return no_such_language;
116  }
117 
118  for (int i = 0; i < (sizeof(EN) / sizeof(struct s2n_error_translation)); i++) {
119  if (EN[i].errno_value == error) {
120  return EN[i].str;
121  }
122  }
123 
124  return no_such_error;
125 }
