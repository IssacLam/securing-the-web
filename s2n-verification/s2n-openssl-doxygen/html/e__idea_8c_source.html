99  const unsigned char *iv, int enc)
100 {
101  if (!enc) {
102  if (EVP_CIPHER_CTX_mode(ctx) == EVP_CIPH_OFB_MODE)
103  enc = 1;
104  else if (EVP_CIPHER_CTX_mode(ctx) == EVP_CIPH_CFB_MODE)
105  enc = 1;
106  }
107  if (enc)
108  idea_set_encrypt_key(key, ctx->cipher_data);
109  else {
110  IDEA_KEY_SCHEDULE tmp;
111 
112  idea_set_encrypt_key(key, &tmp);
113  idea_set_decrypt_key(&tmp, ctx->cipher_data);
114  OPENSSL_cleanse((unsigned char *)&tmp, sizeof(IDEA_KEY_SCHEDULE));
115  }
116  return 1;
117 }
