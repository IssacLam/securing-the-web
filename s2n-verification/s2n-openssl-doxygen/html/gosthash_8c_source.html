63 /* Adds blocks of N bytes modulo 2**(8*n). Returns carry*/
64 static int add_blocks(int n, byte * left, const byte * right)
65 {
66  int i;
67  int carry = 0;
68  int sum;
69  for (i = 0; i < n; i++) {
70  sum = (int)left[i] + (int)right[i] + carry;
71  left[i] = sum & 0xff;
72  carry = sum >> 8;
73  }

75 }
