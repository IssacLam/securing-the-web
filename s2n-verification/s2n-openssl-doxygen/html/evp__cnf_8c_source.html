74 static int alg_module_init(CONF_IMODULE *md, const CONF *cnf)
75 {
76  int i;
77  const char *oid_section;
78  STACK_OF(CONF_VALUE) *sktmp;
79  CONF_VALUE *oval;
80  oid_section = CONF_imodule_get_value(md);
81  if (!(sktmp = NCONF_get_section(cnf, oid_section))) {
82  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_ERROR_LOADING_SECTION);
83  return 0;
84  }
85  for (i = 0; i < sk_CONF_VALUE_num(sktmp); i++) {
86  oval = sk_CONF_VALUE_value(sktmp, i);
87  if (!strcmp(oval->name, "fips_mode")) {
88  int m;
89  if (!X509V3_get_value_bool(oval, &m)) {
90  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_INVALID_FIPS_MODE);
91  return 0;
92  }
93  if (m > 0) {
94 #ifdef OPENSSL_FIPS
95  if (!FIPS_mode() && !FIPS_mode_set(1)) {
96  EVPerr(EVP_F_ALG_MODULE_INIT,
97  EVP_R_ERROR_SETTING_FIPS_MODE);
98  return 0;
99  }
100 #else
101  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_FIPS_MODE_NOT_SUPPORTED);
102  return 0;
103 #endif
104  }
105  } else {
106  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_UNKNOWN_OPTION);
107  ERR_add_error_data(4, "name=", oval->name,
108  ", value=", oval->value);
109  }
110 
111  }
112  return 1;
113 }
