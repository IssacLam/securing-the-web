114  const unsigned char *in, size_t inl)

116  while (inl >= EVP_MAXCHUNK) {

119  &data(ctx)->inw, &data(ctx)->outw, ctx->encrypt);
120  inl -= EVP_MAXCHUNK;
121  in += EVP_MAXCHUNK;
122  out += EVP_MAXCHUNK;
123  }
124  if (inl)
125  DES_xcbc_encrypt(in, out, (long)inl, &data(ctx)->ks,
126  (DES_cblock *)&(ctx->iv[0]),

128  return 1;
