63 int BN_bn2mpi(const BIGNUM *a, unsigned char *d)
64 {
65  int bits;
66  int num = 0;
67  int ext = 0;
68  long l;
69 
70  bits = BN_num_bits(a);
71  num = (bits + 7) / 8;
72  if (bits > 0) {
73  ext = ((bits & 0x07) == 0);
74  }
75  if (d == NULL)
76  return (num + 4 + ext);
77 
78  l = num + ext;
79  d[0] = (unsigned char)(l >> 24) & 0xff;
80  d[1] = (unsigned char)(l >> 16) & 0xff;
81  d[2] = (unsigned char)(l >> 8) & 0xff;
82  d[3] = (unsigned char)(l) & 0xff;
83  if (ext)
84  d[4] = 0;
85  num = BN_bn2bin(a, &(d[4 + ext]));
86  if (a->neg)
87  d[4] |= 0x80;
88  return (num + 4 + ext);
89 }
