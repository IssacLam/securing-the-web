66 int RSA_sign_ASN1_OCTET_STRING(int type,
67  const unsigned char *m, unsigned int m_len,
68  unsigned char *sigret, unsigned int *siglen,
69  RSA *rsa)
70 {
71  ASN1_OCTET_STRING sig;
72  int i, j, ret = 1;
73  unsigned char *p, *s;
74 
75  sig.type = V_ASN1_OCTET_STRING;
76  sig.length = m_len;
77  sig.data = (unsigned char *)m;
78 
79  i = i2d_ASN1_OCTET_STRING(&sig, NULL);
80  j = RSA_size(rsa);
81  if (i > (j - RSA_PKCS1_PADDING_SIZE)) {
82  RSAerr(RSA_F_RSA_SIGN_ASN1_OCTET_STRING,
83  RSA_R_DIGEST_TOO_BIG_FOR_RSA_KEY);
84  return (0);
85  }
86  s = (unsigned char *)OPENSSL_malloc((unsigned int)j + 1);
87  if (s == NULL) {
88  RSAerr(RSA_F_RSA_SIGN_ASN1_OCTET_STRING, ERR_R_MALLOC_FAILURE);
89  return (0);
90  }
91  p = s;
92  i2d_ASN1_OCTET_STRING(&sig, &p);
93  i = RSA_private_encrypt(i, s, sigret, rsa, RSA_PKCS1_PADDING);
94  if (i <= 0)
95  ret = 0;
96  else
97  *siglen = i;
98 
99  OPENSSL_cleanse(s, (unsigned int)j + 1);
100  OPENSSL_free(s);
101  return (ret);
102 }

113 
114  if (siglen != (unsigned int)RSA_size(rsa)) {
115  RSAerr(RSA_F_RSA_VERIFY_ASN1_OCTET_STRING,
116  RSA_R_WRONG_SIGNATURE_LENGTH);
117  return (0);
118  }
119 
120  s = (unsigned char *)OPENSSL_malloc((unsigned int)siglen);
121  if (s == NULL) {
122  RSAerr(RSA_F_RSA_VERIFY_ASN1_OCTET_STRING, ERR_R_MALLOC_FAILURE);
123  goto err;
124  }

126 
127  if (i <= 0)

129

132  if (sig == NULL)

134 
135  if (((unsigned int)sig->length != m_len) ||
136  (memcmp(m, sig->data, m_len) != 0)) {
137  RSAerr(RSA_F_RSA_VERIFY_ASN1_OCTET_STRING, RSA_R_BAD_SIGNATURE);
