130 int ASN1_sign(i2d_of_void *i2d, X509_ALGOR *algor1, X509_ALGOR *algor2,
131  ASN1_BIT_STRING *signature, char *data, EVP_PKEY *pkey,
132  const EVP_MD *type)
133 {
134  EVP_MD_CTX ctx;
135  unsigned char *p, *buf_in = NULL, *buf_out = NULL;
136  int i, inl = 0, outl = 0, outll = 0;
137  X509_ALGOR *a;
138 
139  EVP_MD_CTX_init(&ctx);
140  for (i = 0; i < 2; i++) {
141  if (i == 0)
142  a = algor1;
143  else
144  a = algor2;
145  if (a == NULL)
146  continue;
147  if (type->pkey_type == NID_dsaWithSHA1) {
148  /*
149  * special case: RFC 2459 tells us to omit 'parameters' with
150  * id-dsa-with-sha1
151  */
152  ASN1_TYPE_free(a->parameter);
153  a->parameter = NULL;
154  } else if ((a->parameter == NULL) ||
155  (a->parameter->type != V_ASN1_NULL)) {
156  ASN1_TYPE_free(a->parameter);
157  if ((a->parameter = ASN1_TYPE_new()) == NULL)
158  goto err;
159  a->parameter->type = V_ASN1_NULL;
160  }
161  ASN1_OBJECT_free(a->algorithm);
162  a->algorithm = OBJ_nid2obj(type->pkey_type);
163  if (a->algorithm == NULL) {
164  ASN1err(ASN1_F_ASN1_SIGN, ASN1_R_UNKNOWN_OBJECT_TYPE);
165  goto err;
166  }
167  if (a->algorithm->length == 0) {
168  ASN1err(ASN1_F_ASN1_SIGN,
169  ASN1_R_THE_ASN1_OBJECT_IDENTIFIER_IS_NOT_KNOWN_FOR_THIS_MD);
170  goto err;
171  }
172  }
173  inl = i2d(data, NULL);
174  buf_in = (unsigned char *)OPENSSL_malloc((unsigned int)inl);
175  outll = outl = EVP_PKEY_size(pkey);
176  buf_out = (unsigned char *)OPENSSL_malloc((unsigned int)outl);
177  if ((buf_in == NULL) || (buf_out == NULL)) {
178  outl = 0;
179  ASN1err(ASN1_F_ASN1_SIGN, ERR_R_MALLOC_FAILURE);
180  goto err;
181  }
182  p = buf_in;
183 
184  i2d(data, &p);
185  if (!EVP_SignInit_ex(&ctx, type, NULL)
186  || !EVP_SignUpdate(&ctx, (unsigned char *)buf_in, inl)
187  || !EVP_SignFinal(&ctx, (unsigned char *)buf_out,
188  (unsigned int *)&outl, pkey)) {
189  outl = 0;
190  ASN1err(ASN1_F_ASN1_SIGN, ERR_R_EVP_LIB);
191  goto err;
192  }
193  if (signature->data != NULL)
194  OPENSSL_free(signature->data);
195  signature->data = buf_out;
196  buf_out = NULL;
197  signature->length = outl;
198  /*
199  * In the interests of compatibility, I'll make sure that the bit string
200  * has a 'not-used bits' value of 0
201  */
202  signature->flags &= ~(ASN1_STRING_FLAG_BITS_LEFT | 0x07);
203  signature->flags |= ASN1_STRING_FLAG_BITS_LEFT;
204  err:
205  EVP_MD_CTX_cleanup(&ctx);
206  if (buf_in != NULL) {
207  OPENSSL_cleanse((char *)buf_in, (unsigned int)inl);
208  OPENSSL_free(buf_in);
209  }
210  if (buf_out != NULL) {
211  OPENSSL_cleanse((char *)buf_out, outll);
212  OPENSSL_free(buf_out);
213  }
214  return (outl);
215 }

220  X509_ALGOR *algor2, ASN1_BIT_STRING *signature, void *asn,
221  EVP_PKEY *pkey, const EVP_MD *type)
222 {
223  EVP_MD_CTX ctx;

226  EVP_MD_CTX_cleanup(&ctx);
227  return 0;
228  }

230 }

232 int ASN1_item_sign_ctx(const ASN1_ITEM *it,
233  X509_ALGOR *algor1, X509_ALGOR *algor2,
234  ASN1_BIT_STRING *signature, void *asn, EVP_MD_CTX *ctx)
235 {
236  const EVP_MD *type;
237  EVP_PKEY *pkey;
238  unsigned char *buf_in = NULL, *buf_out = NULL;
239  size_t inl = 0, outl = 0, outll = 0;
240  int signid, paramtype;
241  int rv;
242 
243  type = EVP_MD_CTX_md(ctx);
244  pkey = EVP_PKEY_CTX_get0_pkey(ctx->pctx);
245 
246  if (!type || !pkey) {
247  ASN1err(ASN1_F_ASN1_ITEM_SIGN_CTX, ASN1_R_CONTEXT_NOT_INITIALISED);
248  return 0;
249  }
250 
251  if (pkey->ameth->item_sign) {
252  rv = pkey->ameth->item_sign(ctx, it, asn, algor1, algor2, signature);
253  if (rv == 1)
254  outl = signature->length;
255  /*-
256  * Return value meanings:
257  * <=0: error.
258  * 1: method does everything.
259  * 2: carry on as normal.
260  * 3: ASN1 method sets algorithm identifiers: just sign.
261  */
262  if (rv <= 0)
263  ASN1err(ASN1_F_ASN1_ITEM_SIGN_CTX, ERR_R_EVP_LIB);
264  if (rv <= 1)
265  goto err;
266  } else
267  rv = 2;
268 
269  if (rv == 2) {
270  if (type->flags & EVP_MD_FLAG_PKEY_METHOD_SIGNATURE) {
271  if (!pkey->ameth ||
272  !OBJ_find_sigid_by_algs(&signid,
273  EVP_MD_nid(type),
274  pkey->ameth->pkey_id)) {
275  ASN1err(ASN1_F_ASN1_ITEM_SIGN_CTX,
276  ASN1_R_DIGEST_AND_KEY_TYPE_NOT_SUPPORTED);
277  return 0;
278  }
279  } else
280  signid = type->pkey_type;
281 
282  if (pkey->ameth->pkey_flags & ASN1_PKEY_SIGPARAM_NULL)
283  paramtype = V_ASN1_NULL;
284  else
285  paramtype = V_ASN1_UNDEF;
286 
287  if (algor1)
288  X509_ALGOR_set0(algor1, OBJ_nid2obj(signid), paramtype, NULL);
289  if (algor2)
290  X509_ALGOR_set0(algor2, OBJ_nid2obj(signid), paramtype, NULL);
291 
292  }
293 
294  inl = ASN1_item_i2d(asn, &buf_in, it);
295  outll = outl = EVP_PKEY_size(pkey);
296  buf_out = OPENSSL_malloc((unsigned int)outl);
297  if ((buf_in == NULL) || (buf_out == NULL)) {
298  outl = 0;
299  ASN1err(ASN1_F_ASN1_ITEM_SIGN_CTX, ERR_R_MALLOC_FAILURE);
300  goto err;
301  }
302 
303  if (!EVP_DigestSignUpdate(ctx, buf_in, inl)
304  || !EVP_DigestSignFinal(ctx, buf_out, &outl)) {
305  outl = 0;
306  ASN1err(ASN1_F_ASN1_ITEM_SIGN_CTX, ERR_R_EVP_LIB);
307  goto err;
308  }
309  if (signature->data != NULL)
310  OPENSSL_free(signature->data);
311  signature->data = buf_out;
312  buf_out = NULL;
313  signature->length = outl;
314  /*
315  * In the interests of compatibility, I'll make sure that the bit string
316  * has a 'not-used bits' value of 0
317  */
318  signature->flags &= ~(ASN1_STRING_FLAG_BITS_LEFT | 0x07);
319  signature->flags |= ASN1_STRING_FLAG_BITS_LEFT;
320  err:
321  EVP_MD_CTX_cleanup(ctx);
322  if (buf_in != NULL) {
323  OPENSSL_cleanse((char *)buf_in, (unsigned int)inl);
324  OPENSSL_free(buf_in);
325  }
326  if (buf_out != NULL) {
327  OPENSSL_cleanse((char *)buf_out, outll);
328  OPENSSL_free(buf_out);
329  }
330  return (outl);
331 }
