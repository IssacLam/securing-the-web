67 static int do_sigver_init(EVP_MD_CTX *ctx, EVP_PKEY_CTX **pctx,
68  const EVP_MD *type, ENGINE *e, EVP_PKEY *pkey,
69  int ver)
70 {
71  if (ctx->pctx == NULL)
72  ctx->pctx = EVP_PKEY_CTX_new(pkey, e);
73  if (ctx->pctx == NULL)
74  return 0;
75 
76  if (!(ctx->pctx->pmeth->flags & EVP_PKEY_FLAG_SIGCTX_CUSTOM)) {
77 
78  if (type == NULL) {
79  int def_nid;
80  if (EVP_PKEY_get_default_digest_nid(pkey, &def_nid) > 0)
81  type = EVP_get_digestbynid(def_nid);
82  }
83 
84  if (type == NULL) {
85  EVPerr(EVP_F_DO_SIGVER_INIT, EVP_R_NO_DEFAULT_DIGEST);
86  return 0;
87  }
88  }
89 
90  if (ver) {
91  if (ctx->pctx->pmeth->verifyctx_init) {
92  if (ctx->pctx->pmeth->verifyctx_init(ctx->pctx, ctx) <= 0)
93  return 0;
94  ctx->pctx->operation = EVP_PKEY_OP_VERIFYCTX;
95  } else if (EVP_PKEY_verify_init(ctx->pctx) <= 0)
96  return 0;
97  } else {
98  if (ctx->pctx->pmeth->signctx_init) {
99  if (ctx->pctx->pmeth->signctx_init(ctx->pctx, ctx) <= 0)
100  return 0;
101  ctx->pctx->operation = EVP_PKEY_OP_SIGNCTX;
102  } else if (EVP_PKEY_sign_init(ctx->pctx) <= 0)
103  return 0;
104  }
105  if (EVP_PKEY_CTX_set_signature_md(ctx->pctx, type) <= 0)
106  return 0;
107  if (pctx)
108  *pctx = ctx->pctx;
109  if (ctx->pctx->pmeth->flags & EVP_PKEY_FLAG_SIGCTX_CUSTOM)
110  return 1;
111  if (!EVP_DigestInit_ex(ctx, type, e))
112  return 0;
113  return 1;
114 }

128 int EVP_DigestSignFinal(EVP_MD_CTX *ctx, unsigned char *sigret,
129  size_t *siglen)
130 {
131  int sctx, r = 0;
132  EVP_PKEY_CTX *pctx = ctx->pctx;
133  if (pctx->pmeth->flags & EVP_PKEY_FLAG_SIGCTX_CUSTOM) {
134  EVP_PKEY_CTX *dctx;
135  if (!sigret)
136  return pctx->pmeth->signctx(pctx, sigret, siglen, ctx);
137  dctx = EVP_PKEY_CTX_dup(ctx->pctx);
138  if (!dctx)
139  return 0;
140  r = dctx->pmeth->signctx(dctx, sigret, siglen, ctx);
141  EVP_PKEY_CTX_free(dctx);
142  return r;
143  }
144  if (pctx->pmeth->signctx)
145  sctx = 1;
146  else
147  sctx = 0;
148  if (sigret) {
149  EVP_MD_CTX tmp_ctx;
150  unsigned char md[EVP_MAX_MD_SIZE];
151  unsigned int mdlen;
152  EVP_MD_CTX_init(&tmp_ctx);
153  if (!EVP_MD_CTX_copy_ex(&tmp_ctx, ctx))
154  return 0;
155  if (sctx)
156  r = tmp_ctx.pctx->pmeth->signctx(tmp_ctx.pctx,
157  sigret, siglen, &tmp_ctx);
158  else
159  r = EVP_DigestFinal_ex(&tmp_ctx, md, &mdlen);
160  EVP_MD_CTX_cleanup(&tmp_ctx);
161  if (sctx || !r)
162  return r;
163  if (EVP_PKEY_sign(ctx->pctx, sigret, siglen, md, mdlen) <= 0)
164  return 0;
165  } else {
166  if (sctx) {
167  if (pctx->pmeth->signctx(pctx, sigret, siglen, ctx) <= 0)
168  return 0;
169  } else {
170  int s = EVP_MD_size(ctx->digest);
171  if (s < 0 || EVP_PKEY_sign(pctx, sigret, siglen, NULL, s) <= 0)
172  return 0;
173  }
174  }
175  return 1;
176 }

178 int EVP_DigestVerifyFinal(EVP_MD_CTX *ctx, const unsigned char *sig,
179  size_t siglen)
180 {
181  EVP_MD_CTX tmp_ctx;
182  unsigned char md[EVP_MAX_MD_SIZE];
183  int r;
184  unsigned int mdlen;
185  int vctx;
186 
187  if (ctx->pctx->pmeth->verifyctx)
188  vctx = 1;
189  else
190  vctx = 0;
191  EVP_MD_CTX_init(&tmp_ctx);
192  if (!EVP_MD_CTX_copy_ex(&tmp_ctx, ctx))
193  return -1;
194  if (vctx) {
195  r = tmp_ctx.pctx->pmeth->verifyctx(tmp_ctx.pctx,
196  sig, siglen, &tmp_ctx);
197  } else
198  r = EVP_DigestFinal_ex(&tmp_ctx, md, &mdlen);
199  EVP_MD_CTX_cleanup(&tmp_ctx);
200  if (vctx || !r)
201  return r;
202  return EVP_PKEY_verify(ctx->pctx, sig, siglen, md, mdlen);
203 }
