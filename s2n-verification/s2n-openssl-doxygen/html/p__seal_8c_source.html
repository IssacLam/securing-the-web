69 int EVP_SealInit(EVP_CIPHER_CTX *ctx, const EVP_CIPHER *type,
70  unsigned char **ek, int *ekl, unsigned char *iv,
71  EVP_PKEY **pubk, int npubk)
72 {
73  unsigned char key[EVP_MAX_KEY_LENGTH];
74  int i;
75 
76  if (type) {
77  EVP_CIPHER_CTX_init(ctx);
78  if (!EVP_EncryptInit_ex(ctx, type, NULL, NULL, NULL))
79  return 0;
80  }
81  if ((npubk <= 0) || !pubk)
82  return 1;
83  if (EVP_CIPHER_CTX_rand_key(ctx, key) <= 0)
84  return 0;
85  if (EVP_CIPHER_CTX_iv_length(ctx)
86  && RAND_bytes(iv, EVP_CIPHER_CTX_iv_length(ctx)) <= 0)
87  return 0;
88 
89  if (!EVP_EncryptInit_ex(ctx, NULL, NULL, key, iv))
90  return 0;
91 
92  for (i = 0; i < npubk; i++) {
93  ekl[i] =
94  EVP_PKEY_encrypt_old(ek[i], key, EVP_CIPHER_CTX_key_length(ctx),
95  pubk[i]);
96  if (ekl[i] <= 0)
97  return (-1);
98  }
99  return (npubk);
100 }
