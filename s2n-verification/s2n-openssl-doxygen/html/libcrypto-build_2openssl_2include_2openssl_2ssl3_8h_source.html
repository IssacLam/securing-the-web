321 /* Maximum compression overhead: defined by SSL/TLS standards */

324 /*
325  * The standards give a maximum encryption overhead of 1024 bytes. In
326  * practice the value is lower than this. The overhead is the maximum number
327  * of padding bytes (256) plus the mac size.
328  */
329 # define SSL3_RT_MAX_ENCRYPTED_OVERHEAD (256 + SSL3_RT_MAX_MD_SIZE)
330 
331 /*
332  * OpenSSL currently only uses a padding length of at most one block so the
333  * send overhead is smaller.
334  */
335 
336 # define SSL3_RT_SEND_MAX_ENCRYPTED_OVERHEAD \
337  (SSL_RT_MAX_CIPHER_BLOCK_SIZE + SSL3_RT_MAX_MD_SIZE)

410  * rw
