67 /* Override the default free and new methods */
68 static int rsa_cb(int operation, ASN1_VALUE **pval, const ASN1_ITEM *it,
69  void *exarg)
70 {
71  if (operation == ASN1_OP_NEW_PRE) {
72  *pval = (ASN1_VALUE *)RSA_new();
73  if (*pval)
74  return 2;
75  return 0;
76  } else if (operation == ASN1_OP_FREE_PRE) {
77  RSA_free((RSA *)*pval);
78  *pval = NULL;
79  return 2;
80  }
81  return 1;
82 }
