83 int ECDSA_sign_ex(int type, const unsigned char *dgst, int dlen, unsigned char
84  *sig, unsigned int *siglen, const BIGNUM *kinv,
85  const BIGNUM *r, EC_KEY *eckey)
86 {
87  ECDSA_SIG *s;
88  RAND_seed(dgst, dlen);
89  s = ECDSA_do_sign_ex(dgst, dlen, kinv, r, eckey);
90  if (s == NULL) {
91  *siglen = 0;
92  return 0;
93  }
94  *siglen = i2d_ECDSA_SIG(s, &sig);
95  ECDSA_SIG_free(s);
96  return 1;
97 }
