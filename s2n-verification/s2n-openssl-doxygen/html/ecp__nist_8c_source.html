131 int ec_GFp_nist_group_set_curve(EC_GROUP *group, const BIGNUM *p,
132  const BIGNUM *a, const BIGNUM *b, BN_CTX *ctx)
133 {
134  int ret = 0;
135  BN_CTX *new_ctx = NULL;
136  BIGNUM *tmp_bn;
137 
138  if (ctx == NULL)
139  if ((ctx = new_ctx = BN_CTX_new()) == NULL)
140  return 0;
141 
142  BN_CTX_start(ctx);
143  if ((tmp_bn = BN_CTX_get(ctx)) == NULL)
144  goto err;
145 
146  if (BN_ucmp(BN_get0_nist_prime_192(), p) == 0)
147  group->field_mod_func = BN_nist_mod_192;
148  else if (BN_ucmp(BN_get0_nist_prime_224(), p) == 0)
149  group->field_mod_func = BN_nist_mod_224;
150  else if (BN_ucmp(BN_get0_nist_prime_256(), p) == 0)
151  group->field_mod_func = BN_nist_mod_256;
152  else if (BN_ucmp(BN_get0_nist_prime_384(), p) == 0)
153  group->field_mod_func = BN_nist_mod_384;
154  else if (BN_ucmp(BN_get0_nist_prime_521(), p) == 0)
155  group->field_mod_func = BN_nist_mod_521;
156  else {
157  ECerr(EC_F_EC_GFP_NIST_GROUP_SET_CURVE, EC_R_NOT_A_NIST_PRIME);
158  goto err;
159  }
160 
161  ret = ec_GFp_simple_group_set_curve(group, p, a, b, ctx);
162 
163  err:
164  BN_CTX_end(ctx);
165  if (new_ctx != NULL)
166  BN_CTX_free(new_ctx);
167  return ret;
168 }

170 int ec_GFp_nist_field_mul(const EC_GROUP *group, BIGNUM *r, const BIGNUM *a,
171  const BIGNUM *b, BN_CTX *ctx)
172 {
173  int ret = 0;
174  BN_CTX *ctx_new = NULL;
175 
176  if (!group || !r || !a || !b) {
177  ECerr(EC_F_EC_GFP_NIST_FIELD_MUL, ERR_R_PASSED_NULL_PARAMETER);
178  goto err;
179  }
180  if (!ctx)
181  if ((ctx_new = ctx = BN_CTX_new()) == NULL)
182  goto err;
183 
184  if (!BN_mul(r, a, b, ctx))
185  goto err;
186  if (!group->field_mod_func(r, r, &group->field, ctx))
187  goto err;
188 
189  ret = 1;
190  err:
191  if (ctx_new)
192  BN_CTX_free(ctx_new);
193  return ret;
194 }
