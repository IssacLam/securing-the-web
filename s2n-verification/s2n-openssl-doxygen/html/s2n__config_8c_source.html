165 int s2n_config_free_cert_chain_and_key(struct s2n_config *config)
166 {
167  struct s2n_blob b = {
168  .data = (uint8_t *) config->cert_and_key_pairs,
169  .size = sizeof(struct s2n_cert_chain_and_key)
170  };
171 
172  /* If there were cert and key pairs set, walk the chain and free the certs */
173  if (config->cert_and_key_pairs) {
174  struct s2n_cert_chain *node = config->cert_and_key_pairs->head;
175  while (node) {
176  struct s2n_blob n = {
177  .data = (uint8_t *)node,
178  .size = sizeof(struct s2n_cert_chain)
179  };
180  /* Free the cert */
181  GUARD(s2n_free(&node->cert));
182  /* Advance to next */
183  node = node->next;
184  /* Free the node */
185  GUARD(s2n_free(&n));
186  }
187  GUARD(s2n_rsa_private_key_free(&config->cert_and_key_pairs->private_key));
188  GUARD(s2n_free(&config->cert_and_key_pairs->ocsp_status));
189  }
190 
191  GUARD(s2n_free(&b));
192  return 0;
193 }

234 int s2n_config_set_protocol_preferences(struct s2n_config *config, const char * const *protocols, int protocol_count)
235 {
236  struct s2n_stuffer protocol_stuffer;
237 
238  GUARD(s2n_free(&config->application_protocols));
239 
240  if (protocols == NULL || protocol_count == 0) {
241  /* NULL value indicates no prference, so nothing to do */
242  return 0;
243  }
244 
245  GUARD(s2n_stuffer_growable_alloc(&protocol_stuffer, 256));
246  for (int i = 0; i < protocol_count; i++) {
247  size_t length = strlen(protocols[i]);
248  uint8_t protocol[255];
249 
250  if (length > 255 || (s2n_stuffer_data_available(&protocol_stuffer) + length + 1) > 65535) {
251  return S2N_ERR_APPLICATION_PROTOCOL_TOO_LONG;
252  }
253  memcpy_check(protocol, protocols[i], length);
254  GUARD(s2n_stuffer_write_uint8(&protocol_stuffer, length));
255  GUARD(s2n_stuffer_write_bytes(&protocol_stuffer, protocol, length));
256  }
257 
258  uint32_t size = s2n_stuffer_data_available(&protocol_stuffer);
259  /* config->application_protocols blob now owns this data */
260  config->application_protocols.size = size;
261  config->application_protocols.data = s2n_stuffer_raw_read(&protocol_stuffer, size);
262  notnull_check(config->application_protocols.data);
263 
264  return 0;
265 }

274 int s2n_config_add_cert_chain_and_key_with_status(struct s2n_config *config,
275  char *cert_chain_pem, char *private_key_pem, const uint8_t *status, uint32_t length)
276 {
277  struct s2n_stuffer chain_in_stuffer, cert_out_stuffer, key_in_stuffer, key_out_stuffer;
278  struct s2n_blob key_blob;
279  struct s2n_blob mem;
280 
281  /* Allocate the memory for the chain and key struct */
282  GUARD(s2n_alloc(&mem, sizeof(struct s2n_cert_chain_and_key)));
283  config->cert_and_key_pairs = (struct s2n_cert_chain_and_key *)(void *)mem.data;
284  config->cert_and_key_pairs->ocsp_status.data = NULL;
285  config->cert_and_key_pairs->ocsp_status.size = 0;
286 
287  /* Put the private key pem in a stuffer */
288  GUARD(s2n_stuffer_alloc_ro_from_string(&key_in_stuffer, private_key_pem));
289  GUARD(s2n_stuffer_growable_alloc(&key_out_stuffer, strlen(private_key_pem)));
290 
291  /* Convert pem to asn1 and asn1 to the private key */
292  GUARD(s2n_stuffer_rsa_private_key_from_pem(&key_in_stuffer, &key_out_stuffer));
293  GUARD(s2n_stuffer_free(&key_in_stuffer));
294  key_blob.size = s2n_stuffer_data_available(&key_out_stuffer);
295  key_blob.data = s2n_stuffer_raw_read(&key_out_stuffer, key_blob.size);
296  notnull_check(key_blob.data);
297  GUARD(s2n_asn1der_to_rsa_private_key(&config->cert_and_key_pairs->private_key, &key_blob));
298  GUARD(s2n_stuffer_free(&key_out_stuffer));
299 
300  /* Turn the chain into a stuffer */
301  GUARD(s2n_stuffer_alloc_ro_from_string(&chain_in_stuffer, cert_chain_pem));
302  GUARD(s2n_stuffer_growable_alloc(&cert_out_stuffer, 2048));
303 
304  struct s2n_cert_chain **insert = &config->cert_and_key_pairs->head;
305  uint32_t chain_size = 0;
306  do {
307  struct s2n_cert_chain *new_node;
308 
309  if (s2n_stuffer_certificate_from_pem(&chain_in_stuffer, &cert_out_stuffer) < 0) {
310  if (chain_size == 0) {
311  S2N_ERROR(S2N_ERR_NO_CERTIFICATE_IN_PEM);
312  }
313  break;
314  }
315 
316  GUARD(s2n_alloc(&mem, sizeof(struct s2n_cert_chain)));
317  new_node = (struct s2n_cert_chain *)(void *)mem.data;
318 
319  GUARD(s2n_alloc(&new_node->cert, s2n_stuffer_data_available(&cert_out_stuffer)));
320  GUARD(s2n_stuffer_read(&cert_out_stuffer, &new_node->cert));
321 
322  /* Additional 3 bytes for the length field in the protocol */
323  chain_size += new_node->cert.size + 3;
324  new_node->next = NULL;
325  *insert = new_node;
326  insert = &new_node->next;
327  } while (s2n_stuffer_data_available(&chain_in_stuffer));
328 
329  GUARD(s2n_stuffer_free(&chain_in_stuffer));
330  GUARD(s2n_stuffer_free(&cert_out_stuffer));
331 
332  config->cert_and_key_pairs->chain_size = chain_size;
333 
334  if (status && length > 0) {
335  GUARD(s2n_alloc(&config->cert_and_key_pairs->ocsp_status, length));
336  memcpy_check(config->cert_and_key_pairs->ocsp_status.data, status, length);
337  }
338 
339  return 0;
340 }
