75 /* The maximum size of a TLS record is 16389 bytes. This is; 1 byte for content
76  * type, 2 bytes for the protocol version, 2 bytes for the length field,
77  * and then up to 2^14 for the encrypted+compressed payload data.
78  */
79 #define S2N_TLS_RECORD_HEADER_LENGTH 5
80 #define S2N_TLS_MAXIMUM_FRAGMENT_LENGTH 16384
81 #define S2N_TLS_MAXIMUM_RECORD_LENGTH (S2N_TLS_MAXIMUM_FRAGMENT_LENGTH + S2N_TLS_RECORD_HEADER_LENGTH)
82 
83 /* The maximum size of an SSL2 message is 2^14 - 1, as neither of the first two
84  * bits in the length field are usable. Per;
85  * http://www-archive.mozilla.org/projects/security/pki/nss/ssl/draft02.html
86  * section 1.1
87  */
88 #define S2N_SSL2_RECORD_HEADER_LENGTH 2
89 #define S2N_SSL2_MAXIMUM_MESSAGE_LENGTH 16383
90 #define S2N_SSL2_MAXIMUM_RECORD_LENGTH (S2N_SSL2_MAXIMUM_MESSAGE_LENGTH + S2N_SSL2_RECORD_HEADER_LENGTH)
91 
92 /* s2n can use a "small" record length that is aligned to the dominant internet MTU;
93  * 1500 bytes, minus 20 bytes for an IP header, minus 20 bytes for a tcp
94  * header and 20 bytes for tcp/ip options (timestamp, sack etc) and a "large" record
95  * length that is designed to maximize throughput (fewer MACs per byte transferred
96  * and better efficiency of crypto engines).
97  */
98 #define S2N_SMALL_RECORD_LENGTH (1500 - 20 - 20 - 20)
99 #define S2N_SMALL_FRAGMENT_LENGTH (S2N_SMALL_RECORD_LENGTH - S2N_TLS_RECORD_HEADER_LENGTH)
100 #define S2N_LARGE_RECORD_LENGTH S2N_TLS_MAXIMUM_RECORD_LENGTH
101 #define S2N_LARGE_FRAGMENT_LENGTH S2N_TLS_MAXIMUM_FRAGMENT_LENGTH
102
