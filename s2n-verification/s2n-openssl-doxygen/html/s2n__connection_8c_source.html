45 struct s2n_connection *s2n_connection_new(s2n_mode mode)
46 {
47  struct s2n_blob blob;
48  struct s2n_connection *conn;
49 
50  GUARD_PTR(s2n_alloc(&blob, sizeof(struct s2n_connection)));
51 
52  GUARD_PTR(s2n_blob_zero(&blob));
53 
54  if (mode == S2N_CLIENT) {
55  /* At present s2n is not suitable for use in client mode, as it
56  * does not perform any certificate validation. However it is useful
57  * to use S2N in client mode for testing purposes. An environment
58  * variable is required to be set for the client mode to work.
59  */
60  if (getenv("S2N_ENABLE_CLIENT_MODE") == NULL) {
61  s2n_free(&blob);
62  S2N_ERROR_PTR(S2N_ERR_CLIENT_MODE_DISABLED);
63  }
64  }
65 
66  /* Cast 'through' void to acknowledge that we are changing alignment,
67  * which is ok, as blob.data is always aligned.
68  */
69  conn = (struct s2n_connection *)(void *)blob.data;
70  conn->mode = mode;
71  conn->blinding = S2N_BUILT_IN_BLINDING;
72  conn->config = &s2n_default_config;
73 
74  /* Allocate the fixed-size stuffers */
75  blob.data = conn->alert_in_data;
76  blob.size = S2N_ALERT_LENGTH;
77 
78  GUARD_PTR(s2n_stuffer_init(&conn->alert_in, &blob));
79 
80  blob.data = conn->reader_alert_out_data;
81  blob.size = S2N_ALERT_LENGTH;
82 
83  GUARD_PTR(s2n_stuffer_init(&conn->reader_alert_out, &blob));
84 
85  blob.data = conn->writer_alert_out_data;
86  blob.size = S2N_ALERT_LENGTH;
87 
88  GUARD_PTR(s2n_stuffer_init(&conn->writer_alert_out, &blob));
89  GUARD_PTR(s2n_stuffer_alloc(&conn->out, S2N_LARGE_RECORD_LENGTH));
90 
91  /* Initialize the growable stuffers. Zero length at first, but the resize
92  * in _wipe will fix that
93  */
94  blob.data = conn->header_in_data;
95  blob.size = S2N_TLS_RECORD_HEADER_LENGTH;
96 
97  GUARD_PTR(s2n_stuffer_init(&conn->header_in, &blob));
98  GUARD_PTR(s2n_stuffer_growable_alloc(&conn->in, 0));
99  GUARD_PTR(s2n_stuffer_growable_alloc(&conn->handshake.io, 0));
100  GUARD_PTR(s2n_connection_wipe(conn));
101  GUARD_PTR(s2n_timer_start(conn->config, &conn->write_timer));
102 
103  return conn;
104 }

106 static int s2n_connection_free_keys(struct s2n_connection *conn)
107 {
108  /* Destroy any keys - we call destroy on the pending object as that is where
109  * keys are allocated. */
110  if (conn->pending.cipher_suite && conn->pending.cipher_suite->cipher->destroy_key) {
111  GUARD(conn->pending.cipher_suite->cipher->destroy_key(&conn->pending.client_key));
112  GUARD(conn->pending.cipher_suite->cipher->destroy_key(&conn->pending.server_key));
113  }
114 
115  /* Free any pending server key received (we may not have completed a
116  * handshake, so this may not have been free'd yet) */
117  GUARD(s2n_rsa_public_key_free(&conn->pending.server_rsa_public_key));
118 
119  GUARD(s2n_dh_params_free(&conn->pending.server_dh_params));
120  GUARD(s2n_dh_params_free(&conn->active.server_dh_params));
121  GUARD(s2n_ecc_params_free(&conn->pending.server_ecc_params));
122  GUARD(s2n_ecc_params_free(&conn->active.server_ecc_params));
123 
124  GUARD(s2n_free(&conn->status_response));
125 
126  return 0;
127 }

129 int s2n_shutdown(struct s2n_connection *conn, s2n_blocked_status *more)
130 {
131  uint64_t elapsed;
132 
133  GUARD(s2n_timer_elapsed(conn->config, &conn->write_timer, &elapsed));
134  if (elapsed < conn->delay) {
135  S2N_ERROR(S2N_ERR_SHUTDOWN_PAUSED);
136  }
137 
138  /* Write any pending I/O */
139  GUARD(s2n_flush(conn, more));
140 
141  GUARD(s2n_queue_writer_close_alert(conn));
142 
143  /* Write the alert message out */
144  GUARD(s2n_flush(conn, more));
145 
146  /* Wipe the connection */
147  GUARD(s2n_connection_wipe(conn));
148 
149  return 0;
150 }

175 int s2n_connection_wipe(struct s2n_connection *conn)
176 {
177  /* First make a copy of everything we'd like to save, which isn't very
178  * much.
179  */
180  int mode = conn->mode;
181  struct s2n_config *config = conn->config;
182  struct s2n_stuffer alert_in;
183  struct s2n_stuffer reader_alert_out;
184  struct s2n_stuffer writer_alert_out;
185  struct s2n_stuffer handshake_io;
186  struct s2n_stuffer header_in;
187  struct s2n_stuffer in;
188  struct s2n_stuffer out;
189 
190  /* Wipe all of the sensitive stuff */
191  GUARD(s2n_connection_free_keys(conn));
192  GUARD(s2n_stuffer_wipe(&conn->alert_in));
193  GUARD(s2n_stuffer_wipe(&conn->reader_alert_out));
194  GUARD(s2n_stuffer_wipe(&conn->writer_alert_out));
195  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
196  GUARD(s2n_stuffer_wipe(&conn->header_in));
197  GUARD(s2n_stuffer_wipe(&conn->in));
198  GUARD(s2n_stuffer_wipe(&conn->out));
199 
200  /* Allocate or resize to their original sizes */
201  GUARD(s2n_stuffer_resize(&conn->in, S2N_LARGE_FRAGMENT_LENGTH));
202 
203  /* Allocate memory for handling handshakes */
204  GUARD(s2n_stuffer_resize(&conn->handshake.io, S2N_LARGE_RECORD_LENGTH));
205 
206  /* Clone the stuffers */
207  /* ignore gcc 4.7 address warnings because dest is allocated on the stack */
208  /* pragma gcc diagnostic was added in gcc 4.6 */
209 #if defined(__GNUC__) && GCC_VERSION >= 40600
210 #pragma GCC diagnostic push
211 #pragma GCC diagnostic ignored "-Waddress"
212 #endif
213  memcpy_check(&alert_in, &conn->alert_in, sizeof(struct s2n_stuffer));
214  memcpy_check(&reader_alert_out, &conn->reader_alert_out, sizeof(struct s2n_stuffer));
215  memcpy_check(&writer_alert_out, &conn->writer_alert_out, sizeof(struct s2n_stuffer));
216  memcpy_check(&handshake_io, &conn->handshake.io, sizeof(struct s2n_stuffer));
217  memcpy_check(&header_in, &conn->header_in, sizeof(struct s2n_stuffer));
218  memcpy_check(&in, &conn->in, sizeof(struct s2n_stuffer));
219  memcpy_check(&out, &conn->out, sizeof(struct s2n_stuffer));
220 #if defined(__GNUC__) && GCC_VERSION >= 40600
221 #pragma GCC diagnostic pop
222 #endif
223 
224  /* Zero the whole connection structure */
225  memset_check(conn, 0, sizeof(struct s2n_connection));
226 
227  conn->readfd = -1;
228  conn->writefd = -1;
229  conn->mode = mode;
230  conn->config = config;
231  conn->active.cipher_suite = &s2n_null_cipher_suite;
232  conn->pending.cipher_suite = &s2n_null_cipher_suite;
233  conn->server = &conn->active;
234  conn->client = &conn->active;
235  conn->max_fragment_length = S2N_SMALL_FRAGMENT_LENGTH;
236  conn->handshake.state = CLIENT_HELLO;
237  GUARD(s2n_hash_init(&conn->handshake.client_md5, S2N_HASH_MD5));
238  GUARD(s2n_hash_init(&conn->handshake.client_sha1, S2N_HASH_SHA1));
239  GUARD(s2n_hash_init(&conn->handshake.client_sha256, S2N_HASH_SHA256));
240  GUARD(s2n_hash_init(&conn->handshake.server_md5, S2N_HASH_MD5));
241  GUARD(s2n_hash_init(&conn->handshake.server_sha1, S2N_HASH_SHA1));
242  GUARD(s2n_hash_init(&conn->handshake.server_sha256, S2N_HASH_SHA256));
243  GUARD(s2n_hmac_init(&conn->client->client_record_mac, S2N_HMAC_NONE, NULL, 0));
244  GUARD(s2n_hmac_init(&conn->server->server_record_mac, S2N_HMAC_NONE, NULL, 0));
245 
246  memcpy_check(&conn->alert_in, &alert_in, sizeof(struct s2n_stuffer));
247  memcpy_check(&conn->reader_alert_out, &reader_alert_out, sizeof(struct s2n_stuffer));
248  memcpy_check(&conn->writer_alert_out, &writer_alert_out, sizeof(struct s2n_stuffer));
249  memcpy_check(&conn->handshake.io, &handshake_io, sizeof(struct s2n_stuffer));
250  memcpy_check(&conn->header_in, &header_in, sizeof(struct s2n_stuffer));
251  memcpy_check(&conn->in, &in, sizeof(struct s2n_stuffer));
252  memcpy_check(&conn->out, &out, sizeof(struct s2n_stuffer));
253 
254  /* Set everything to the highest version at first */
255  conn->server_protocol_version = s2n_highest_protocol_version;
256  conn->client_protocol_version = s2n_highest_protocol_version;
257  conn->actual_protocol_version = s2n_highest_protocol_version;
258 
259  return 0;
260 }

388 int s2n_connection_kill(struct s2n_connection *conn)
389 {
390  conn->closed = 1;
391 
392  /* Delay between 10 and 30 seconds in nanoseconds */
393  int64_t min = TEN_S, max = 3 * TEN_S;
394 
395  /* Keep track of the delay so that it can be enforced */
396  conn->delay = min + s2n_public_random(max - min);
397 
398  /* Restart the write timer */
399  GUARD(s2n_timer_start(conn->config, &conn->write_timer));
400 
401  if (conn->blinding == S2N_BUILT_IN_BLINDING) {
402  struct timespec sleep_time = { .tv_sec = conn->delay / ONE_S, .tv_nsec = conn->delay % ONE_S };
403  int r;
404 
405  do {
406  r = nanosleep(&sleep_time, &sleep_time);
407  }
408  while (r != 0);
409  }
410 
411  return 0;
412 }
