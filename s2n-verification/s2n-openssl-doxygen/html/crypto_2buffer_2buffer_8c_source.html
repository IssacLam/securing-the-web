97 int BUF_MEM_grow(BUF_MEM *str, size_t len)
98 {
99  char *ret;
100  size_t n;
101 
102  if (str->length >= len) {
103  str->length = len;
104  return (len);
105  }
106  if (str->max >= len) {
107  memset(&str->data[str->length], 0, len - str->length);
108  str->length = len;
109  return (len);
110  }
111  /* This limit is sufficient to ensure (len+3)/3*4 < 2**31 */
112  if (len > LIMIT_BEFORE_EXPANSION) {
113  BUFerr(BUF_F_BUF_MEM_GROW, ERR_R_MALLOC_FAILURE);
114  return 0;
115  }
116  n = (len + 3) / 3 * 4;
117  if (str->data == NULL)
118  ret = OPENSSL_malloc(n);
119  else
120  ret = OPENSSL_realloc(str->data, n);
121  if (ret == NULL) {
122  BUFerr(BUF_F_BUF_MEM_GROW, ERR_R_MALLOC_FAILURE);
123  len = 0;
124  } else {
125  str->data = ret;
126  str->max = n;
127  memset(&str->data[str->length], 0, len - str->length);
128  str->length = len;
129  }
130  return (len);
131 }
