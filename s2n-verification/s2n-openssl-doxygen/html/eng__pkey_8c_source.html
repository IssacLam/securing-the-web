99 EVP_PKEY *ENGINE_load_private_key(ENGINE *e, const char *key_id,
100  UI_METHOD *ui_method, void *callback_data)
101 {
102  EVP_PKEY *pkey;
103 
104  if (e == NULL) {
105  ENGINEerr(ENGINE_F_ENGINE_LOAD_PRIVATE_KEY,
106  ERR_R_PASSED_NULL_PARAMETER);
107  return 0;
108  }
109  CRYPTO_w_lock(CRYPTO_LOCK_ENGINE);
110  if (e->funct_ref == 0) {
111  CRYPTO_w_unlock(CRYPTO_LOCK_ENGINE);
112  ENGINEerr(ENGINE_F_ENGINE_LOAD_PRIVATE_KEY, ENGINE_R_NOT_INITIALISED);
113  return 0;
114  }
115  CRYPTO_w_unlock(CRYPTO_LOCK_ENGINE);
116  if (!e->load_privkey) {
117  ENGINEerr(ENGINE_F_ENGINE_LOAD_PRIVATE_KEY,
118  ENGINE_R_NO_LOAD_FUNCTION);
119  return 0;
120  }
121  pkey = e->load_privkey(e, key_id, ui_method, callback_data);
122  if (!pkey) {
123  ENGINEerr(ENGINE_F_ENGINE_LOAD_PRIVATE_KEY,
124  ENGINE_R_FAILED_LOADING_PRIVATE_KEY);
125  return 0;
126  }
127  return pkey;
128 }
