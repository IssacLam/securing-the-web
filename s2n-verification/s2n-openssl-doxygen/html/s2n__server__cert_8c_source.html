29 int s2n_server_cert_recv(struct s2n_connection *conn)
30 {
31  uint32_t size_of_all_certificates;
32 
33  GUARD(s2n_stuffer_read_uint24(&conn->handshake.io, &size_of_all_certificates));
34 
35  if (size_of_all_certificates > s2n_stuffer_data_available(&conn->handshake.io) || size_of_all_certificates < 3) {
36  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
37  }
38 
39  int certificate_count = 0;
40  while (s2n_stuffer_data_available(&conn->handshake.io)) {
41  uint32_t certificate_size;
42 
43  GUARD(s2n_stuffer_read_uint24(&conn->handshake.io, &certificate_size));
44 
45  if (certificate_size > s2n_stuffer_data_available(&conn->handshake.io) || certificate_size == 0) {
46  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
47  }
48 
49  struct s2n_blob asn1cert;
50  asn1cert.data = s2n_stuffer_raw_read(&conn->handshake.io, certificate_size);
51  asn1cert.size = certificate_size;
52  notnull_check(asn1cert.data);
53 
54  /* TODO: certificate validation goes here */
55  gt_check(certificate_size, 0);
56 
57  /* Pull the public key from the first certificate */
58  if (certificate_count == 0) {
59  GUARD(s2n_asn1der_to_rsa_public_key(&conn->pending.server_rsa_public_key, &asn1cert));
60  }
61 
62  certificate_count++;
63  }
64 
65  gte_check(certificate_count, 1);
66 
67  conn->handshake.next_state = SERVER_HELLO_DONE;
68 
69  if (conn->status_type == S2N_STATUS_REQUEST_OCSP) {
70  conn->handshake.next_state = SERVER_CERT_STATUS;
71  } else if (conn->pending.cipher_suite->key_exchange_alg->flags & S2N_KEY_EXCHANGE_EPH) {
72  conn->handshake.next_state = SERVER_KEY;
73  }
74 
75  return 0;
76 }

78 int s2n_server_cert_send(struct s2n_connection *conn)
79 {
80  struct s2n_cert_chain *head = conn->server->chosen_cert_chain->head;
81 
82  GUARD(s2n_stuffer_write_uint24(&conn->handshake.io, conn->server->chosen_cert_chain->chain_size));
83 
84  while (head) {
85  GUARD(s2n_stuffer_write_uint24(&conn->handshake.io, head->cert.size));
86  GUARD(s2n_stuffer_write_bytes(&conn->handshake.io, head->cert.data, head->cert.size));
87  head = head->next;
88  }
89 
90  conn->handshake.next_state = SERVER_HELLO_DONE;
91 
92  if (s2n_server_can_send_ocsp(conn)) {
93  conn->handshake.next_state = SERVER_CERT_STATUS;
94  } else if (conn->pending.cipher_suite->key_exchange_alg->flags & S2N_KEY_EXCHANGE_EPH) {
95  conn->handshake.next_state = SERVER_KEY;
96  }
97 
98  return 0;
99 }
