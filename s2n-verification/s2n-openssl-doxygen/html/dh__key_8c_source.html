97 int DH_compute_key_padded(unsigned char *key, const BIGNUM *pub_key, DH *dh)
98 {
99  int rv, pad;
100  rv = dh->meth->compute_key(key, pub_key, dh);
101  if (rv <= 0)
102  return rv;
103  pad = BN_num_bytes(dh->p) - rv;
104  if (pad > 0) {
105  memmove(key + pad, key, rv);
106  memset(key, 0, pad);
107  }
108  return rv + pad;
109 }

128 static int generate_key(DH *dh)
129 {
130  int ok = 0;
131  int generate_new_key = 0;
132  unsigned l;
133  BN_CTX *ctx;
134  BN_MONT_CTX *mont = NULL;
135  BIGNUM *pub_key = NULL, *priv_key = NULL;
136 
137  ctx = BN_CTX_new();
138  if (ctx == NULL)
139  goto err;
140 
141  if (dh->priv_key == NULL) {
142  priv_key = BN_new();
143  if (priv_key == NULL)
144  goto err;
145  generate_new_key = 1;
146  } else
147  priv_key = dh->priv_key;
148 
149  if (dh->pub_key == NULL) {
150  pub_key = BN_new();
151  if (pub_key == NULL)
152  goto err;
153  } else
154  pub_key = dh->pub_key;
155 
156  if (dh->flags & DH_FLAG_CACHE_MONT_P) {
157  mont = BN_MONT_CTX_set_locked(&dh->method_mont_p,
158  CRYPTO_LOCK_DH, dh->p, ctx);
159  if (!mont)
160  goto err;
161  }
162 
163  if (generate_new_key) {
164  if (dh->q) {
165  do {
166  if (!BN_rand_range(priv_key, dh->q))
167  goto err;
168  }
169  while (BN_is_zero(priv_key) || BN_is_one(priv_key));
170  } else {
171  /* secret exponent length */
172  l = dh->length ? dh->length : BN_num_bits(dh->p) - 1;
173  if (!BN_rand(priv_key, l, 0, 0))
174  goto err;
175  }
176  }
177 
178  {
179  BIGNUM local_prk;
180  BIGNUM *prk;
181 
182  if ((dh->flags & DH_FLAG_NO_EXP_CONSTTIME) == 0) {
183  BN_init(&local_prk);
184  prk = &local_prk;
185  BN_with_flags(prk, priv_key, BN_FLG_CONSTTIME);
186  } else
187  prk = priv_key;
188 
189  if (!dh->meth->bn_mod_exp(dh, pub_key, dh->g, prk, dh->p, ctx, mont))
190  goto err;
191  }
192 
193  dh->pub_key = pub_key;
194  dh->priv_key = priv_key;
195  ok = 1;
196  err:
197  if (ok != 1)
198  DHerr(DH_F_GENERATE_KEY, ERR_R_BN_LIB);
199 
200  if ((pub_key != NULL) && (dh->pub_key == NULL))
201  BN_free(pub_key);
202  if ((priv_key != NULL) && (dh->priv_key == NULL))
203  BN_free(priv_key);
204  BN_CTX_free(ctx);
205  return (ok);
206 }

208 static int compute_key(unsigned char *key, const BIGNUM *pub_key, DH *dh)
209 {
210  BN_CTX *ctx = NULL;
211  BN_MONT_CTX *mont = NULL;
212  BIGNUM *tmp;
213  int ret = -1;
214  int check_result;
215 
216  if (BN_num_bits(dh->p) > OPENSSL_DH_MAX_MODULUS_BITS) {
217  DHerr(DH_F_COMPUTE_KEY, DH_R_MODULUS_TOO_LARGE);
218  goto err;
219  }
220 
221  ctx = BN_CTX_new();
222  if (ctx == NULL)
223  goto err;
224  BN_CTX_start(ctx);
225  tmp = BN_CTX_get(ctx);
226 
227  if (dh->priv_key == NULL) {
228  DHerr(DH_F_COMPUTE_KEY, DH_R_NO_PRIVATE_VALUE);
229  goto err;
230  }
231 
232  if (dh->flags & DH_FLAG_CACHE_MONT_P) {
233  mont = BN_MONT_CTX_set_locked(&dh->method_mont_p,
234  CRYPTO_LOCK_DH, dh->p, ctx);
235  if ((dh->flags & DH_FLAG_NO_EXP_CONSTTIME) == 0) {
236  /* XXX */
237  BN_set_flags(dh->priv_key, BN_FLG_CONSTTIME);
238  }
239  if (!mont)
240  goto err;
241  }
242 
243  if (!DH_check_pub_key(dh, pub_key, &check_result) || check_result) {
244  DHerr(DH_F_COMPUTE_KEY, DH_R_INVALID_PUBKEY);
245  goto err;
246  }
247 
248  if (!dh->
249  meth->bn_mod_exp(dh, tmp, pub_key, dh->priv_key, dh->p, ctx, mont)) {
250  DHerr(DH_F_COMPUTE_KEY, ERR_R_BN_LIB);
251  goto err;
252  }
253 
254  ret = BN_bn2bin(tmp, key);
255  err:
256  if (ctx != NULL) {
257  BN_CTX_end(ctx);
258  BN_CTX_free(ctx);
259  }
260  return (ret);
261 }

263 static int dh_bn_mod_exp(const DH *dh, BIGNUM *r,
264  const BIGNUM *a, const BIGNUM *p,
265  const BIGNUM *m, BN_CTX *ctx, BN_MONT_CTX *m_ctx)
266 {
267  /*
268  * If a is only one word long and constant time is false, use the faster
269  * exponenentiation function.
270  */
271  if (a->top == 1 && ((dh->flags & DH_FLAG_NO_EXP_CONSTTIME) != 0)) {
272  BN_ULONG A = a->d[0];
273  return BN_mod_exp_mont_word(r, A, p, m, ctx, m_ctx);
274  } else
275  return BN_mod_exp_mont(r, a, p, m, ctx, m_ctx);
276 }
