70 int PKCS7_add_attrib_smimecap(PKCS7_SIGNER_INFO *si,
71  STACK_OF(X509_ALGOR) *cap)
72 {
73  ASN1_STRING *seq;
74  if (!(seq = ASN1_STRING_new())) {
75  PKCS7err(PKCS7_F_PKCS7_ADD_ATTRIB_SMIMECAP, ERR_R_MALLOC_FAILURE);
76  return 0;
77  }
78  seq->length = ASN1_item_i2d((ASN1_VALUE *)cap, &seq->data,
79  ASN1_ITEM_rptr(X509_ALGORS));
80  return PKCS7_add_signed_attribute(si, NID_SMIMECapabilities,
81  V_ASN1_SEQUENCE, seq);
82 }

84 STACK_OF(X509_ALGOR) *PKCS7_get_smimecap(PKCS7_SIGNER_INFO *si)
85 {
86  ASN1_TYPE *cap;
87  const unsigned char *p;
88 
89  cap = PKCS7_get_signed_attribute(si, NID_SMIMECapabilities);
90  if (!cap || (cap->type != V_ASN1_SEQUENCE))
91  return NULL;
92  p = cap->value.sequence->data;
93  return (STACK_OF(X509_ALGOR) *)
94  ASN1_item_d2i(NULL, &p, cap->value.sequence->length,
95  ASN1_ITEM_rptr(X509_ALGORS));
96 }

98 /* Basic smime-capabilities OID and optional integer arg */
99 int PKCS7_simple_smimecap(STACK_OF(X509_ALGOR) *sk, int nid, int arg)
100 {
101  X509_ALGOR *alg;
102 
103  if (!(alg = X509_ALGOR_new())) {
104  PKCS7err(PKCS7_F_PKCS7_SIMPLE_SMIMECAP, ERR_R_MALLOC_FAILURE);
105  return 0;
106  }
107  ASN1_OBJECT_free(alg->algorithm);
108  alg->algorithm = OBJ_nid2obj(nid);
109  if (arg > 0) {
110  ASN1_INTEGER *nbit;
111  if (!(alg->parameter = ASN1_TYPE_new())) {
112  PKCS7err(PKCS7_F_PKCS7_SIMPLE_SMIMECAP, ERR_R_MALLOC_FAILURE);
113  return 0;
114  }
115  if (!(nbit = ASN1_INTEGER_new())) {
116  PKCS7err(PKCS7_F_PKCS7_SIMPLE_SMIMECAP, ERR_R_MALLOC_FAILURE);
117  return 0;
118  }
119  if (!ASN1_INTEGER_set(nbit, arg)) {
120  PKCS7err(PKCS7_F_PKCS7_SIMPLE_SMIMECAP, ERR_R_MALLOC_FAILURE);
121  return 0;
122  }
123  alg->parameter->value.integer = nbit;
124  alg->parameter->type = V_ASN1_INTEGER;
125  }
126  sk_X509_ALGOR_push(sk, alg);
127  return 1;
128 }

151 int PKCS7_add1_attrib_digest(PKCS7_SIGNER_INFO *si,
152  const unsigned char *md, int mdlen)
153 {
154  ASN1_OCTET_STRING *os;
155  os = ASN1_OCTET_STRING_new();
156  if (!os)
157  return 0;
158  if (!ASN1_STRING_set(os, md, mdlen)
159  || !PKCS7_add_signed_attribute(si, NID_pkcs9_messageDigest,
160  V_ASN1_OCTET_STRING, os)) {
161  ASN1_OCTET_STRING_free(os);
162  return 0;
163  }
164  return 1;
165 }
