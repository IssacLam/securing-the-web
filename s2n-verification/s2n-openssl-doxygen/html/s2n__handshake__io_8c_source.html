83 /* Writing is relatively straight forward, simply write each message out as a record,
84  * we may fragment a message across multiple records, but we never coalesce multiple
85  * messages into single records.
86  * Precondition: pending outbound I/O has already been flushed
87  */
88 static int handshake_write_io(struct s2n_connection *conn)
89 {
90  uint8_t record_type = state_machine[conn->handshake.state].record_type;
91  s2n_blocked_status blocked = S2N_NOT_BLOCKED;
92  int max_payload_size;
93 
94  /* Populate handshake.io with header/payload for the current state, once.
95  * Check wiped instead of s2n_stuffer_data_available to differentiate between the initial call
96  * to handshake_write_io and a repeated call after an EWOULDBLOCK.
97  */
98  if (conn->handshake.io.wiped == 1) {
99  if (record_type == TLS_HANDSHAKE) {
100  GUARD(s2n_handshake_write_header(conn, state_machine[conn->handshake.state].message_type));
101  }
102  GUARD(state_machine[conn->handshake.state].handler[conn->mode] (conn));
103  if (record_type == TLS_HANDSHAKE) {
104  GUARD(s2n_handshake_finish_header(conn));
105  }
106  }
107 
108  /* Write the handshake data to records in fragment sized chunks */
109  struct s2n_blob out;
110  while(s2n_stuffer_data_available(&conn->handshake.io) > 0) {
111  out.size = s2n_stuffer_data_available(&conn->handshake.io);
112 
113  GUARD((max_payload_size = s2n_record_max_write_payload_size(conn)));
114  if (out.size > max_payload_size) {
115  out.size = max_payload_size;
116  }
117 
118  out.data = s2n_stuffer_raw_read(&conn->handshake.io, out.size);
119  notnull_check(out.data);
120 
121  /* Make the actual record */
122  GUARD(s2n_record_write(conn, record_type, &out));
123 
124  /* MD5 and SHA sum the handshake data too */
125  if (record_type == TLS_HANDSHAKE) {
126  GUARD(s2n_conn_update_handshake_hashes(conn, &out));
127  }
128 
129  /* Actually send the record. We could block here. Assume the caller will call flush before coming back. */
130  GUARD(s2n_flush(conn, &blocked));
131  }
132 
133  /* We're done sending the last record, reset everything */
134  GUARD(s2n_stuffer_wipe(&conn->out));
135  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
136 
137  /* Advance the state machine */
138  conn->handshake.state = conn->handshake.next_state;
139 
140  return 0;
141 }

143 /*
144  * Returns:
145  * 1 - more data is needed to complete the handshake message.
146  * 0 - we read the whole handshake message.
147  * -1 - error processing the handshake message.
148  */
149 static int read_full_handshake_message(struct s2n_connection *conn, uint8_t *message_type)
150 {
151  uint32_t current_handshake_data = s2n_stuffer_data_available(&conn->handshake.io);
152  if (current_handshake_data < TLS_HANDSHAKE_HEADER_LENGTH) {
153  /* The message may be so badly fragmented that we don't even read the full header, take
154  * what we can and then continue to the next record read iteration.
155  */
156  if (s2n_stuffer_data_available(&conn->in) < (TLS_HANDSHAKE_HEADER_LENGTH - current_handshake_data)) {
157  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, s2n_stuffer_data_available(&conn->in)));
158  return 1;
159  }
160 
161  /* Get the remainder of the header */
162  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, (TLS_HANDSHAKE_HEADER_LENGTH - current_handshake_data)));
163  }
164 
165  uint32_t handshake_message_length;
166  GUARD(s2n_handshake_parse_header(conn, message_type, &handshake_message_length));
167 
168  if (handshake_message_length > S2N_MAXIMUM_HANDSHAKE_MESSAGE_LENGTH) {
169  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
170  }
171 
172  uint32_t bytes_to_take = handshake_message_length - s2n_stuffer_data_available(&conn->handshake.io);;
173  if (bytes_to_take > s2n_stuffer_data_available(&conn->in)) {
174  bytes_to_take = s2n_stuffer_data_available(&conn->in);
175  }
176 
177  /* If the record is handshake data, add it to the handshake buffer */
178  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, bytes_to_take));
179 
180  /* If we have the whole handshake message, then success */
181  if (s2n_stuffer_data_available(&conn->handshake.io) == handshake_message_length) {
182  struct s2n_blob handshake;
183  handshake.data = conn->handshake.io.blob.data;
184  handshake.size = TLS_HANDSHAKE_HEADER_LENGTH + handshake_message_length;
185 
186  notnull_check(handshake.data);
187 
188  /* MD5 and SHA sum the handshake data too */
189  GUARD(s2n_conn_update_handshake_hashes(conn, &handshake));
190 
191  return 0;
192  }
193 
194  /* We don't have the whole message, so we'll need to go again */
195  GUARD(s2n_stuffer_reread(&conn->handshake.io));
196 
197  return 1;
198 }

200 /* Reading is a little more complicated than writing as the TLS RFCs allow content
201  * types to be interleaved at the record layer. We may get an alert message
202  * during the handshake phase, or messages of types that we don't support (e.g.
203  * HEARTBEAT messages), or during renegotiations we may even get application
204  * data messages that need to be handled by the application. The latter is punted
205  * for now (s2n does support renegotiations).
206  */
207 static int handshake_read_io(struct s2n_connection *conn)
208 {
209  uint8_t record_type;
210  int isSSLv2;
211 
212  int r = s2n_read_full_record(conn, &record_type, &isSSLv2);
213  if (r < 0) {
214  if (r == -2) {
215  conn->closed = 1;
216  S2N_ERROR(S2N_ERR_CLOSED);
217  }
218  return -1;
219  }
220 
221  if (isSSLv2) {
222  if (conn->handshake.state != CLIENT_HELLO) {
223  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
224  }
225 
226  /* Add the message to our handshake hashes */
227  struct s2n_blob hashed = {.data = conn->header_in.blob.data + 2,.size = 3 };
228  GUARD(s2n_conn_update_handshake_hashes(conn, &hashed));
229 
230  hashed.data = conn->in.blob.data;
231  hashed.size = s2n_stuffer_data_available(&conn->in);
232  GUARD(s2n_conn_update_handshake_hashes(conn, &hashed));
233 
234  /* Handle an SSLv2 client hello */
235  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, s2n_stuffer_data_available(&conn->in)));
236  GUARD(s2n_sslv2_client_hello_recv(conn));
237  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
238 
239  /* We're done with the record, wipe it */
240  GUARD(s2n_stuffer_wipe(&conn->header_in));
241  GUARD(s2n_stuffer_wipe(&conn->in));
242  conn->in_status = ENCRYPTED;
243 
244  /* Advance the state machine */
245  conn->handshake.state = conn->handshake.next_state;
246  }
247 
248  /* Now we have a record, but it could be a partial fragment of a message, or it might
249  * contain several messages.
250  */
251  if (record_type == TLS_APPLICATION_DATA) {
252  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
253  } else if (record_type == TLS_CHANGE_CIPHER_SPEC) {
254  if (s2n_stuffer_data_available(&conn->in) != 1) {
255  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
256  }
257 
258  GUARD(s2n_stuffer_copy(&conn->in, &conn->handshake.io, s2n_stuffer_data_available(&conn->in)));
259  GUARD(state_machine[conn->handshake.state].handler[conn->mode] (conn));
260  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
261 
262  /* We're done with the record, wipe it */
263  GUARD(s2n_stuffer_wipe(&conn->header_in));
264  GUARD(s2n_stuffer_wipe(&conn->in));
265  conn->in_status = ENCRYPTED;
266 
267  /* Advance the state machine */
268  conn->handshake.state = conn->handshake.next_state;
269 
270  return 0;
271  } else if (record_type != TLS_HANDSHAKE) {
272  if (record_type == TLS_ALERT) {
273  GUARD(s2n_process_alert_fragment(conn));
274  }
275 
276  /* Ignore record types that we don't support */
277 
278  /* We're done with the record, wipe it */
279  GUARD(s2n_stuffer_wipe(&conn->header_in));
280  GUARD(s2n_stuffer_wipe(&conn->in));
281  conn->in_status = ENCRYPTED;
282  return 0;
283  }
284 
285  /* Record is a handshake message */
286  while (s2n_stuffer_data_available(&conn->in)) {
287  uint8_t handshake_message_type;
288  GUARD((r = read_full_handshake_message(conn, &handshake_message_type)));
289 
290  /* Do we need more data? */
291  if (r == 1) {
292  /* Break out of this inner loop, but since we're not changing the state, the
293  * outer loop in s2n_handshake_io() will read another record.
294  */
295  GUARD(s2n_stuffer_wipe(&conn->header_in));
296  GUARD(s2n_stuffer_wipe(&conn->in));
297  conn->in_status = ENCRYPTED;
298  return 0;
299  }
300 
301  if (handshake_message_type != state_machine[conn->handshake.state].message_type) {
302  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
303  }
304 
305  /* Call the relevant handler */
306  r = state_machine[conn->handshake.state].handler[conn->mode](conn);
307  GUARD(s2n_stuffer_wipe(&conn->handshake.io));
308 
309  if (r < 0) {
310  GUARD(s2n_connection_kill(conn));
311 
312  return r;
313  }
314 
315  /* Advance the state machine */
316  conn->handshake.state = conn->handshake.next_state;
317  }
318 
319  /* We're done with the record, wipe it */
320  GUARD(s2n_stuffer_wipe(&conn->header_in));
321  GUARD(s2n_stuffer_wipe(&conn->in));
322  conn->in_status = ENCRYPTED;
323 
324  return 0;
325 }

327 int s2n_negotiate(struct s2n_connection *conn, s2n_blocked_status *blocked)
328 {
329  char this = 'S';
330  if (conn->mode == S2N_CLIENT) {
331  this = 'C';
332  }
333 
334  while (state_machine[conn->handshake.state].writer != 'B') {
335 
336  /* Flush any pending I/O or alert messages */
337  GUARD(s2n_flush(conn, blocked));
338 
339  if (state_machine[conn->handshake.state].writer == this) {
340  *blocked = S2N_BLOCKED_ON_WRITE;
341  GUARD(handshake_write_io(conn));
342  } else {
343  *blocked = S2N_BLOCKED_ON_READ;
344  GUARD(handshake_read_io(conn));
345  }
346 
347  /* If the handshake has just ended, free up memory */
348  if (state_machine[conn->handshake.state].writer == 'B') {
349  GUARD(s2n_stuffer_resize(&conn->handshake.io, 0));
350  }
351  }
352 
353  *blocked = S2N_NOT_BLOCKED;
354 
355  return 0;
356 }
