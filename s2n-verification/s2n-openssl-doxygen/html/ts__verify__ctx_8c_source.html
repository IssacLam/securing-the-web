112 TS_VERIFY_CTX *TS_REQ_to_TS_VERIFY_CTX(TS_REQ *req, TS_VERIFY_CTX *ctx)
113 {
114  TS_VERIFY_CTX *ret = ctx;
115  ASN1_OBJECT *policy;
116  TS_MSG_IMPRINT *imprint;
117  X509_ALGOR *md_alg;
118  ASN1_OCTET_STRING *msg;
119  const ASN1_INTEGER *nonce;
120 
121  OPENSSL_assert(req != NULL);
122  if (ret)
123  TS_VERIFY_CTX_cleanup(ret);
124  else if (!(ret = TS_VERIFY_CTX_new()))
125  return NULL;
126 
127  /* Setting flags. */
128  ret->flags = TS_VFY_ALL_IMPRINT & ~(TS_VFY_TSA_NAME | TS_VFY_SIGNATURE);
129 
130  /* Setting policy. */
131  if ((policy = TS_REQ_get_policy_id(req)) != NULL) {
132  if (!(ret->policy = OBJ_dup(policy)))
133  goto err;
134  } else
135  ret->flags &= ~TS_VFY_POLICY;
136 
137  /* Setting md_alg, imprint and imprint_len. */
138  imprint = TS_REQ_get_msg_imprint(req);
139  md_alg = TS_MSG_IMPRINT_get_algo(imprint);
140  if (!(ret->md_alg = X509_ALGOR_dup(md_alg)))
141  goto err;
142  msg = TS_MSG_IMPRINT_get_msg(imprint);
143  ret->imprint_len = ASN1_STRING_length(msg);
144  if (!(ret->imprint = OPENSSL_malloc(ret->imprint_len)))
145  goto err;
146  memcpy(ret->imprint, ASN1_STRING_data(msg), ret->imprint_len);
147 
148  /* Setting nonce. */
149  if ((nonce = TS_REQ_get_nonce(req)) != NULL) {
150  if (!(ret->nonce = ASN1_INTEGER_dup(nonce)))
151  goto err;
152  } else
153  ret->flags &= ~TS_VFY_NONCE;
154 
155  return ret;
156  err:
157  if (ctx)
158  TS_VERIFY_CTX_cleanup(ctx);
159  else
160  TS_VERIFY_CTX_free(ret);
161  return NULL;
162 }
