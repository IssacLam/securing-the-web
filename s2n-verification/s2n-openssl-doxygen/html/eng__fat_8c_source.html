86 #ifndef OPENSSL_NO_ECDSA

103 static int int_def_cb(const char *alg, int len, void *arg)
104 {
105  unsigned int *pflags = arg;
106  if (alg == NULL)
107  return 0;
108  if (!strncmp(alg, "ALL", len))
109  *pflags |= ENGINE_METHOD_ALL;
110  else if (!strncmp(alg, "RSA", len))
111  *pflags |= ENGINE_METHOD_RSA;
112  else if (!strncmp(alg, "DSA", len))
113  *pflags |= ENGINE_METHOD_DSA;
114  else if (!strncmp(alg, "ECDH", len))
115  *pflags |= ENGINE_METHOD_ECDH;
116  else if (!strncmp(alg, "ECDSA", len))
117  *pflags |= ENGINE_METHOD_ECDSA;
118  else if (!strncmp(alg, "DH", len))
119  *pflags |= ENGINE_METHOD_DH;
120  else if (!strncmp(alg, "RAND", len))
121  *pflags |= ENGINE_METHOD_RAND;
122  else if (!strncmp(alg, "CIPHERS", len))
123  *pflags |= ENGINE_METHOD_CIPHERS;
124  else if (!strncmp(alg, "DIGESTS", len))
125  *pflags |= ENGINE_METHOD_DIGESTS;
126  else if (!strncmp(alg, "PKEY", len))
127  *pflags |= ENGINE_METHOD_PKEY_METHS | ENGINE_METHOD_PKEY_ASN1_METHS;
128  else if (!strncmp(alg, "PKEY_CRYPTO", len))
129  *pflags |= ENGINE_METHOD_PKEY_METHS;
130  else if (!strncmp(alg, "PKEY_ASN1", len))
131  *pflags |= ENGINE_METHOD_PKEY_ASN1_METHS;
132  else
133  return 0;
134  return 1;
135 }

137 int ENGINE_set_default_string(ENGINE *e, const char *def_list)
138 {
139  unsigned int flags = 0;
140  if (!CONF_parse_list(def_list, ',', 1, int_def_cb, &flags)) {
141  ENGINEerr(ENGINE_F_ENGINE_SET_DEFAULT_STRING,
142  ENGINE_R_INVALID_STRING);
143  ERR_add_error_data(2, "str=", def_list);
144  return 0;
145  }
146  return ENGINE_set_default(e, flags);
147 }
