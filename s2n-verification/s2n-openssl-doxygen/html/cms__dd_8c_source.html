105 int cms_DigestedData_do_final(CMS_ContentInfo *cms, BIO *chain, int verify)
106 {
107  EVP_MD_CTX mctx;
108  unsigned char md[EVP_MAX_MD_SIZE];
109  unsigned int mdlen;
110  int r = 0;
111  CMS_DigestedData *dd;
112  EVP_MD_CTX_init(&mctx);
113 
114  dd = cms->d.digestedData;
115 
116  if (!cms_DigestAlgorithm_find_ctx(&mctx, chain, dd->digestAlgorithm))
117  goto err;
118 
119  if (EVP_DigestFinal_ex(&mctx, md, &mdlen) <= 0)
120  goto err;
121 
122  if (verify) {
123  if (mdlen != (unsigned int)dd->digest->length) {
124  CMSerr(CMS_F_CMS_DIGESTEDDATA_DO_FINAL,
125  CMS_R_MESSAGEDIGEST_WRONG_LENGTH);
126  goto err;
127  }
128 
129  if (memcmp(md, dd->digest->data, mdlen))
130  CMSerr(CMS_F_CMS_DIGESTEDDATA_DO_FINAL,
131  CMS_R_VERIFICATION_FAILURE);
132  else
133  r = 1;
134  } else {
135  if (!ASN1_STRING_set(dd->digest, md, mdlen))
136  goto err;
137  r = 1;
138  }
139 
140  err:
141  EVP_MD_CTX_cleanup(&mctx);
142 
143  return r;
144 
145 }
