86 {
87  switch (type) {
88  case EVP_CTRL_INIT:
89  data(c)->rounds = RC5_12_ROUNDS;
90  return 1;
91 
92  case EVP_CTRL_GET_RC5_ROUNDS:
93  *(int *)ptr = data(c)->rounds;
94  return 1;
95 
96  case EVP_CTRL_SET_RC5_ROUNDS:
97  switch (arg) {
98  case RC5_8_ROUNDS:
99  case RC5_12_ROUNDS:
100  case RC5_16_ROUNDS:
101  data(c)->rounds = arg;
102  return 1;
103 
104  default:
105  EVPerr(EVP_F_RC5_CTRL, EVP_R_UNSUPORTED_NUMBER_OF_ROUNDS);
106  return 0;
107  }
108 
109  default:
110  return -1;
111  }
