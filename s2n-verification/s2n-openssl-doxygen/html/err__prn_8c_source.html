77  CRYPTO_THREADID_current(&cur);
78  es = CRYPTO_THREADID_hash(&cur);
79  while ((l = ERR_get_error_line_data(&file, &line, &data, &flags)) != 0) {
80  ERR_error_string_n(l, buf, sizeof buf);
81  BIO_snprintf(buf2, sizeof(buf2), "%lu:%s:%s:%d:%s\n", es, buf,
82  file, line, (flags & ERR_TXT_STRING) ? data : "");
83  if (cb(buf2, strlen(buf2), u) <= 0)
84  break; /* abort outputting the error report */
85  }
