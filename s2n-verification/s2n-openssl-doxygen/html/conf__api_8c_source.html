273 CONF_VALUE *_CONF_new_section(CONF *conf, const char *section)
274 {
275  STACK_OF(CONF_VALUE) *sk = NULL;
276  int ok = 0, i;
277  CONF_VALUE *v = NULL, *vv;
278 
279  if ((sk = sk_CONF_VALUE_new_null()) == NULL)
280  goto err;
281  if ((v = OPENSSL_malloc(sizeof(CONF_VALUE))) == NULL)
282  goto err;
283  i = strlen(section) + 1;
284  if ((v->section = OPENSSL_malloc(i)) == NULL)
285  goto err;
286 
287  memcpy(v->section, section, i);
288  v->name = NULL;
289  v->value = (char *)sk;
290 
291  vv = lh_CONF_VALUE_insert(conf->data, v);
292  OPENSSL_assert(vv == NULL);
293  ok = 1;
294  err:
295  if (!ok) {
296  if (sk != NULL)
297  sk_CONF_VALUE_free(sk);
298  if (v != NULL)
299  OPENSSL_free(v);
300  v = NULL;
301  }
302  return (v);
303 }
