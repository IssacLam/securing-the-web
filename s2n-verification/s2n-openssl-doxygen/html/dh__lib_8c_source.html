95 int DH_set_method(DH *dh, const DH_METHOD *meth)
96 {
97  /*
98  * NB: The caller is specifically setting a method, so it's not up to us
99  * to deal with which ENGINE it comes from.
100  */
101  const DH_METHOD *mtmp;
102  mtmp = dh->meth;
103  if (mtmp->finish)
104  mtmp->finish(dh);
105 #ifndef OPENSSL_NO_ENGINE
106  if (dh->engine) {
107  ENGINE_finish(dh->engine);
108  dh->engine = NULL;
109  }
110 #endif
111  dh->meth = meth;
112  if (meth->init)
113  meth->init(dh);
114  return 1;
115 }

122 DH *DH_new_method(ENGINE *engine)
123 {
124  DH *ret;
125 
126  ret = (DH *)OPENSSL_malloc(sizeof(DH));
127  if (ret == NULL) {
128  DHerr(DH_F_DH_NEW_METHOD, ERR_R_MALLOC_FAILURE);
129  return (NULL);
130  }
131 
132  ret->meth = DH_get_default_method();
133 #ifndef OPENSSL_NO_ENGINE
134  if (engine) {
135  if (!ENGINE_init(engine)) {
136  DHerr(DH_F_DH_NEW_METHOD, ERR_R_ENGINE_LIB);
137  OPENSSL_free(ret);
138  return NULL;
139  }
140  ret->engine = engine;
141  } else
142  ret->engine = ENGINE_get_default_DH();
143  if (ret->engine) {
144  ret->meth = ENGINE_get_DH(ret->engine);
145  if (!ret->meth) {
146  DHerr(DH_F_DH_NEW_METHOD, ERR_R_ENGINE_LIB);
147  ENGINE_finish(ret->engine);
148  OPENSSL_free(ret);
149  return NULL;
150  }
151  }
152 #endif
153 
154  ret->pad = 0;
155  ret->version = 0;
156  ret->p = NULL;
157  ret->g = NULL;
158  ret->length = 0;
159  ret->pub_key = NULL;
160  ret->priv_key = NULL;
161  ret->q = NULL;
162  ret->j = NULL;
163  ret->seed = NULL;
164  ret->seedlen = 0;
165  ret->counter = NULL;
166  ret->method_mont_p = NULL;
167  ret->references = 1;
168  ret->flags = ret->meth->flags & ~DH_FLAG_NON_FIPS_ALLOW;
169  CRYPTO_new_ex_data(CRYPTO_EX_INDEX_DH, ret, &ret->ex_data);
170  if ((ret->meth->init != NULL) && !ret->meth->init(ret)) {
171 #ifndef OPENSSL_NO_ENGINE
172  if (ret->engine)
173  ENGINE_finish(ret->engine);
174 #endif
175  CRYPTO_free_ex_data(CRYPTO_EX_INDEX_DH, ret, &ret->ex_data);
176  OPENSSL_free(ret);
177  ret = NULL;
178  }
179  return (ret);
180 }

182 void DH_free(DH *r)
183 {
184  int i;
185  if (r == NULL)
186  return;
187  i = CRYPTO_add(&r->references, -1, CRYPTO_LOCK_DH);
188 #ifdef REF_PRINT
189  REF_PRINT("DH", r);
190 #endif
191  if (i > 0)
192  return;
193 #ifdef REF_CHECK
194  if (i < 0) {
195  fprintf(stderr, "DH_free, bad reference count\n");
196  abort();
197  }
198 #endif
199 
200  if (r->meth->finish)
201  r->meth->finish(r);
202 #ifndef OPENSSL_NO_ENGINE
203  if (r->engine)
204  ENGINE_finish(r->engine);
205 #endif
206 
207  CRYPTO_free_ex_data(CRYPTO_EX_INDEX_DH, r, &r->ex_data);
208 
209  if (r->p != NULL)
210  BN_clear_free(r->p);
211  if (r->g != NULL)
212  BN_clear_free(r->g);
213  if (r->q != NULL)
214  BN_clear_free(r->q);
215  if (r->j != NULL)
216  BN_clear_free(r->j);
217  if (r->seed)
218  OPENSSL_free(r->seed);
219  if (r->counter != NULL)
220  BN_clear_free(r->counter);
221  if (r->pub_key != NULL)
222  BN_clear_free(r->pub_key);
223  if (r->priv_key != NULL)
224  BN_clear_free(r->priv_key);
225  OPENSSL_free(r);
226 }
