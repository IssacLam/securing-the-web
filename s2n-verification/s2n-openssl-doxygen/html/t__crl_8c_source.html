102  ASN1_TIME_print(out, X509_CRL_get_lastUpdate(x));
103  BIO_printf(out, "\n%8sNext Update: ", "");
104  if (X509_CRL_get_nextUpdate(x))
105  ASN1_TIME_print(out, X509_CRL_get_nextUpdate(x));
106  else
107  BIO_printf(out, "NONE");
108  BIO_printf(out, "\n");
109 
110  X509V3_extensions_print(out, "CRL extensions", x->crl->extensions, 0, 8);
111 
112  rev = X509_CRL_get_REVOKED(x);
113 
114  if (sk_X509_REVOKED_num(rev) > 0)
115  BIO_printf(out, "Revoked Certificates:\n");
116  else
117  BIO_printf(out, "No Revoked Certificates.\n");
118 
119  for (i = 0; i < sk_X509_REVOKED_num(rev); i++) {
120  r = sk_X509_REVOKED_value(rev, i);
121  BIO_printf(out, " Serial Number: ");
122  i2a_ASN1_INTEGER(out, r->serialNumber);
123  BIO_printf(out, "\n Revocation Date: ");
124  ASN1_TIME_print(out, r->revocationDate);

127  r->extensions, 0, 8);
128  }
129  X509_signature_print(out, x->sig_alg, x->signature);
130 
131  return 1;
