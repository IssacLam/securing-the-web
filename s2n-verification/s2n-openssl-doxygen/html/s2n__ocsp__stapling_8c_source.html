44 int s2n_server_status_recv(struct s2n_connection *conn)
45 {
46  uint8_t type;
47  struct s2n_blob status = { .data = NULL, .size = 0 };
48 
49  GUARD(s2n_stuffer_read_uint8(&conn->handshake.io, &type));
50  GUARD(s2n_stuffer_read_uint24(&conn->handshake.io, &status.size));
51  status.data = s2n_stuffer_raw_read(&conn->handshake.io, status.size);
52  notnull_check(status.data);
53 
54  if (type == S2N_STATUS_REQUEST_OCSP) {
55  GUARD(s2n_alloc(&conn->status_response, status.size));
56  memcpy_check(conn->status_response.data, status.data, status.size);
57  conn->status_response.size = status.size;
58  }
59 
60  conn->handshake.next_state = SERVER_HELLO_DONE;
61 
62  if (conn->pending.cipher_suite->key_exchange_alg->flags & S2N_KEY_EXCHANGE_EPH) {
63  conn->handshake.next_state = SERVER_KEY;
64  }
65 
66  return 0;
67 }
