86 int X509v3_get_ext_by_OBJ(const STACK_OF(X509_EXTENSION) *sk,
87  ASN1_OBJECT *obj, int lastpos)
88 {
89  int n;
90  X509_EXTENSION *ex;
91 
92  if (sk == NULL)
93  return (-1);
94  lastpos++;
95  if (lastpos < 0)
96  lastpos = 0;
97  n = sk_X509_EXTENSION_num(sk);
98  for (; lastpos < n; lastpos++) {
99  ex = sk_X509_EXTENSION_value(sk, lastpos);
100  if (OBJ_cmp(ex->object, obj) == 0)
101  return (lastpos);
102  }
103  return (-1);
104 }

144 STACK_OF(X509_EXTENSION) *X509v3_add_ext(STACK_OF(X509_EXTENSION) **x,
145  X509_EXTENSION *ex, int loc)
146 {
147  X509_EXTENSION *new_ex = NULL;
148  int n;
149  STACK_OF(X509_EXTENSION) *sk = NULL;
150 
151  if (x == NULL) {
152  X509err(X509_F_X509V3_ADD_EXT, ERR_R_PASSED_NULL_PARAMETER);
153  goto err2;
154  }
155 
156  if (*x == NULL) {
157  if ((sk = sk_X509_EXTENSION_new_null()) == NULL)
158  goto err;
159  } else
160  sk = *x;
161 
162  n = sk_X509_EXTENSION_num(sk);
163  if (loc > n)
164  loc = n;
165  else if (loc < 0)
166  loc = n;
167 
168  if ((new_ex = X509_EXTENSION_dup(ex)) == NULL)
169  goto err2;
170  if (!sk_X509_EXTENSION_insert(sk, new_ex, loc))
171  goto err;
172  if (*x == NULL)
173  *x = sk;
174  return (sk);
175  err:
176  X509err(X509_F_X509V3_ADD_EXT, ERR_R_MALLOC_FAILURE);
177  err2:
178  if (new_ex != NULL)
179  X509_EXTENSION_free(new_ex);
180  if (sk != NULL)
181  sk_X509_EXTENSION_free(sk);
182  return (NULL);
183 }

203 X509_EXTENSION *X509_EXTENSION_create_by_OBJ(X509_EXTENSION **ex,
204  ASN1_OBJECT *obj, int crit,
205  ASN1_OCTET_STRING *data)
206 {
207  X509_EXTENSION *ret;
208 
209  if ((ex == NULL) || (*ex == NULL)) {
210  if ((ret = X509_EXTENSION_new()) == NULL) {
211  X509err(X509_F_X509_EXTENSION_CREATE_BY_OBJ,
212  ERR_R_MALLOC_FAILURE);
213  return (NULL);
214  }
215  } else
216  ret = *ex;
217 
218  if (!X509_EXTENSION_set_object(ret, obj))
219  goto err;
220  if (!X509_EXTENSION_set_critical(ret, crit))
221  goto err;
222  if (!X509_EXTENSION_set_data(ret, data))
223  goto err;
224 
225  if ((ex != NULL) && (*ex == NULL))
226  *ex = ret;
227  return (ret);
228  err:
229  if ((ex == NULL) || (ret != *ex))
230  X509_EXTENSION_free(ret);
231  return (NULL);
232 }
