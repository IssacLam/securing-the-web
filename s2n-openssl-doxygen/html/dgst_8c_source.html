82 static void list_md_fn(const EVP_MD *m,
83  const char *from, const char *to, void *arg)
84 {
85  const char *mname;
86  /* Skip aliases */
87  if (!m)
88  return;
89  mname = OBJ_nid2ln(EVP_MD_type(m));
90  /* Skip shortnames */
91  if (strcmp(from, mname))
92  return;
93  /* Skip clones */
94  if (EVP_MD_flags(m) & EVP_MD_FLAG_PKEY_DIGEST)
95  return;
96  if (strchr(mname, ' '))
97  mname = EVP_MD_name(m);
98  BIO_printf(arg, "-%-14s to use the %s message digest algorithm\n",
99  mname, mname);
100 }

539 int do_fp(BIO *out, unsigned char *buf, BIO *bp, int sep, int binout,
540  EVP_PKEY *key, unsigned char *sigin, int siglen,
541  const char *sig_name, const char *md_name,
542  const char *file, BIO *bmd)
543 {
544  size_t len;
545  int i;
546 
547  for (;;) {
548  i = BIO_read(bp, (char *)buf, BUFSIZE);
549  if (i < 0) {
550  BIO_printf(bio_err, "Read Error in %s\n", file);
551  ERR_print_errors(bio_err);
552  return 1;
553  }
554  if (i == 0)
555  break;
556  }
557  if (sigin) {
558  EVP_MD_CTX *ctx;
559  BIO_get_md_ctx(bp, &ctx);
560  i = EVP_DigestVerifyFinal(ctx, sigin, (unsigned int)siglen);
561  if (i > 0)
562  BIO_printf(out, "Verified OK\n");
563  else if (i == 0) {
564  BIO_printf(out, "Verification Failure\n");
565  return 1;
566  } else {
567  BIO_printf(bio_err, "Error Verifying Data\n");
568  ERR_print_errors(bio_err);
569  return 1;
570  }
571  return 0;
572  }
573  if (key) {
574  EVP_MD_CTX *ctx;
575  BIO_get_md_ctx(bp, &ctx);
576  len = BUFSIZE;
577  if (!EVP_DigestSignFinal(ctx, buf, &len)) {
578  BIO_printf(bio_err, "Error Signing Data\n");
579  ERR_print_errors(bio_err);
580  return 1;
581  }
582  } else {
583  len = BIO_gets(bp, (char *)buf, BUFSIZE);
584  if ((int)len < 0) {
585  ERR_print_errors(bio_err);
586  return 1;
587  }
588  }
589 
590  if (binout)
591  BIO_write(out, buf, len);
592  else if (sep == 2) {
593  for (i = 0; i < (int)len; i++)
594  BIO_printf(out, "%02x", buf[i]);
595  BIO_printf(out, " *%s\n", file);
596  } else {
597  if (sig_name) {
598  BIO_puts(out, sig_name);
599  if (md_name)
600  BIO_printf(out, "-%s", md_name);
601  BIO_printf(out, "(%s)= ", file);
602  } else if (md_name)
603  BIO_printf(out, "%s(%s)= ", md_name, file);
604  else
605  BIO_printf(out, "(%s)= ", file);
606  for (i = 0; i < (int)len; i++) {
607  if (sep && (i != 0))
608  BIO_printf(out, ":");
609  BIO_printf(out, "%02x", buf[i]);
610  }
611  BIO_printf(out, "\n");
612  }
613  return 0;
614 }
