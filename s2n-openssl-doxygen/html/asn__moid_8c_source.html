72 static int oid_module_init(CONF_IMODULE *md, const CONF *cnf)
73 {
74  int i;
75  const char *oid_section;
76  STACK_OF(CONF_VALUE) *sktmp;
77  CONF_VALUE *oval;
78  oid_section = CONF_imodule_get_value(md);
79  if (!(sktmp = NCONF_get_section(cnf, oid_section))) {
80  ASN1err(ASN1_F_OID_MODULE_INIT, ASN1_R_ERROR_LOADING_SECTION);
81  return 0;
82  }
83  for (i = 0; i < sk_CONF_VALUE_num(sktmp); i++) {
84  oval = sk_CONF_VALUE_value(sktmp, i);
85  if (!do_create(oval->value, oval->name)) {
86  ASN1err(ASN1_F_OID_MODULE_INIT, ASN1_R_ADDING_OBJECT);
87  return 0;
88  }
89  }
90  return 1;
91 }

103 /*-
104  * Create an OID based on a name value pair. Accept two formats.
105  * shortname = 1.2.3.4
106  * shortname = some long name, 1.2.3.4
107  */
108 
109 static int do_create(char *value, char *name)
110 {
111  int nid;
112  ASN1_OBJECT *oid;
113  char *ln, *ostr, *p, *lntmp;
114  p = strrchr(value, ',');
115  if (!p) {
116  ln = name;
117  ostr = value;
118  } else {
119  ln = NULL;
120  ostr = p + 1;
121  if (!*ostr)
122  return 0;
123  while (isspace((unsigned char)*ostr))
124  ostr++;
125  }
126 
127  nid = OBJ_create(ostr, name, ln);
128 
129  if (nid == NID_undef)
130  return 0;
131 
132  if (p) {
133  ln = value;
134  while (isspace((unsigned char)*ln))
135  ln++;
136  p--;
137  while (isspace((unsigned char)*p)) {
138  if (p == ln)
139  return 0;
140  p--;
141  }
142  p++;
143  lntmp = OPENSSL_malloc((p - ln) + 1);
144  if (lntmp == NULL)
145  return 0;
146  memcpy(lntmp, ln, p - ln);
147  lntmp[p - ln] = 0;
148  oid = OBJ_nid2obj(nid);
149  oid->ln = lntmp;
150  }
151 
152  return 1;
153 }
