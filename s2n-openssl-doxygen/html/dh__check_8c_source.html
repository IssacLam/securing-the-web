64 /*-
65  * Check that p is a safe prime and
66  * if g is 2, 3 or 5, check that it is a suitable generator
67  * where
68  * for 2, p mod 24 == 11
69  * for 3, p mod 12 == 5
70  * for 5, p mod 10 == 3 or 7
71  * should hold.
72  */

74 int DH_check(const DH *dh, int *ret)
75 {
76  int ok = 0;
77  BN_CTX *ctx = NULL;
78  BN_ULONG l;
79  BIGNUM *t1 = NULL, *t2 = NULL;
80 
81  *ret = 0;
82  ctx = BN_CTX_new();
83  if (ctx == NULL)
84  goto err;
85  BN_CTX_start(ctx);
86  t1 = BN_CTX_get(ctx);
87  if (t1 == NULL)
88  goto err;
89  t2 = BN_CTX_get(ctx);
90  if (t2 == NULL)
91  goto err;
92 
93  if (dh->q) {
94  if (BN_cmp(dh->g, BN_value_one()) <= 0)
95  *ret |= DH_NOT_SUITABLE_GENERATOR;
96  else if (BN_cmp(dh->g, dh->p) >= 0)
97  *ret |= DH_NOT_SUITABLE_GENERATOR;
98  else {
99  /* Check g^q == 1 mod p */
100  if (!BN_mod_exp(t1, dh->g, dh->q, dh->p, ctx))
101  goto err;
102  if (!BN_is_one(t1))
103  *ret |= DH_NOT_SUITABLE_GENERATOR;
104  }
105  if (!BN_is_prime_ex(dh->q, BN_prime_checks, ctx, NULL))
106  *ret |= DH_CHECK_Q_NOT_PRIME;
107  /* Check p == 1 mod q i.e. q divides p - 1 */
108  if (!BN_div(t1, t2, dh->p, dh->q, ctx))
109  goto err;
110  if (!BN_is_one(t2))
111  *ret |= DH_CHECK_INVALID_Q_VALUE;
112  if (dh->j && BN_cmp(dh->j, t1))
113  *ret |= DH_CHECK_INVALID_J_VALUE;
114 
115  } else if (BN_is_word(dh->g, DH_GENERATOR_2)) {
116  l = BN_mod_word(dh->p, 24);
117  if (l != 11)
118  *ret |= DH_NOT_SUITABLE_GENERATOR;
119  }
120 #if 0
121  else if (BN_is_word(dh->g, DH_GENERATOR_3)) {
122  l = BN_mod_word(dh->p, 12);
123  if (l != 5)
124  *ret |= DH_NOT_SUITABLE_GENERATOR;
125  }
126 #endif
127  else if (BN_is_word(dh->g, DH_GENERATOR_5)) {
128  l = BN_mod_word(dh->p, 10);
129  if ((l != 3) && (l != 7))
130  *ret |= DH_NOT_SUITABLE_GENERATOR;
131  } else
132  *ret |= DH_UNABLE_TO_CHECK_GENERATOR;
133 
134  if (!BN_is_prime_ex(dh->p, BN_prime_checks, ctx, NULL))
135  *ret |= DH_CHECK_P_NOT_PRIME;
136  else if (!dh->q) {
137  if (!BN_rshift1(t1, dh->p))
138  goto err;
139  if (!BN_is_prime_ex(t1, BN_prime_checks, ctx, NULL))
140  *ret |= DH_CHECK_P_NOT_SAFE_PRIME;
141  }
142  ok = 1;
143  err:
144  if (ctx != NULL) {
145  BN_CTX_end(ctx);
146  BN_CTX_free(ctx);
147  }
148  return (ok);
149 }

151 int DH_check_pub_key(const DH *dh, const BIGNUM *pub_key, int *ret)
152 {
153  int ok = 0;
154  BIGNUM *tmp = NULL;
155  BN_CTX *ctx = NULL;
156 
157  *ret = 0;
158  ctx = BN_CTX_new();
159  if (ctx == NULL)
160  goto err;
161  BN_CTX_start(ctx);
162  tmp = BN_CTX_get(ctx);
163  if (tmp == NULL || !BN_set_word(tmp, 1))
164  goto err;
165  if (BN_cmp(pub_key, tmp) <= 0)
166  *ret |= DH_CHECK_PUBKEY_TOO_SMALL;
167  if (BN_copy(tmp, dh->p) == NULL || !BN_sub_word(tmp, 1))
168  goto err;
169  if (BN_cmp(pub_key, tmp) >= 0)
170  *ret |= DH_CHECK_PUBKEY_TOO_LARGE;
171 
172  if (dh->q != NULL) {
173  /* Check pub_key^q == 1 mod p */
174  if (!BN_mod_exp(tmp, pub_key, dh->q, dh->p, ctx))
175  goto err;
176  if (!BN_is_one(tmp))
177  *ret |= DH_CHECK_PUBKEY_INVALID;
178  }
179 
180  ok = 1;
181  err:
182  if (ctx != NULL) {
183  BN_CTX_end(ctx);
184  BN_CTX_free(ctx);
185  }
186  return (ok);
