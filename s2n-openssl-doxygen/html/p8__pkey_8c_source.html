65 /* Minor tweak to operation: zero private key data */
66 static int pkey_cb(int operation, ASN1_VALUE **pval, const ASN1_ITEM *it,
67  void *exarg)
68 {
69  /* Since the structure must still be valid use ASN1_OP_FREE_PRE */
70  if (operation == ASN1_OP_FREE_PRE) {
71  PKCS8_PRIV_KEY_INFO *key = (PKCS8_PRIV_KEY_INFO *)*pval;
72  if (key->pkey && key->pkey->type == V_ASN1_OCTET_STRING
73  && key->pkey->value.octet_string != NULL)
74  OPENSSL_cleanse(key->pkey->value.octet_string->data,
75  key->pkey->value.octet_string->length);
76  }
77  return 1;
78 }

89 int PKCS8_pkey_set0(PKCS8_PRIV_KEY_INFO *priv, ASN1_OBJECT *aobj,
90  int version,
91  int ptype, void *pval, unsigned char *penc, int penclen)
92 {
93  unsigned char **ppenc = NULL;
94  if (version >= 0) {
95  if (!ASN1_INTEGER_set(priv->version, version))
96  return 0;
97  }
98  if (penc) {
99  int pmtype;
100  ASN1_OCTET_STRING *oct;
101  oct = ASN1_OCTET_STRING_new();
102  if (!oct)
103  return 0;
104  oct->data = penc;
105  ppenc = &oct->data;
106  oct->length = penclen;
107  if (priv->broken == PKCS8_NO_OCTET)
108  pmtype = V_ASN1_SEQUENCE;
109  else
110  pmtype = V_ASN1_OCTET_STRING;
111  ASN1_TYPE_set(priv->pkey, pmtype, oct);
112  }
113  if (!X509_ALGOR_set0(priv->pkeyalg, aobj, ptype, pval)) {
114  /* If call fails do not swallow 'enc' */
115  if (ppenc)
116  *ppenc = NULL;
117  return 0;
118  }
119  return 1;
120 }

122 int PKCS8_pkey_get0(ASN1_OBJECT **ppkalg,
123  const unsigned char **pk, int *ppklen,
124  X509_ALGOR **pa, PKCS8_PRIV_KEY_INFO *p8)
125 {
126  if (ppkalg)
127  *ppkalg = p8->pkeyalg->algorithm;
128  if (p8->pkey->type == V_ASN1_OCTET_STRING) {
129  p8->broken = PKCS8_OK;
130  if (pk) {
131  *pk = p8->pkey->value.octet_string->data;
132  *ppklen = p8->pkey->value.octet_string->length;
133  }
134  } else if (p8->pkey->type == V_ASN1_SEQUENCE) {
135  p8->broken = PKCS8_NO_OCTET;
136  if (pk) {
137  *pk = p8->pkey->value.sequence->data;
138  *ppklen = p8->pkey->value.sequence->length;
139  }
140  } else
141  return 0;
142  if (pa)
143  *pa = p8->pkeyalg;
144  return 1;
145 }
