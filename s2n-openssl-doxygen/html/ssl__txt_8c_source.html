106 int SSL_SESSION_print(BIO *bp, const SSL_SESSION *x)
107 {
108  unsigned int i;
109  const char *s;
110 
111  if (x == NULL)
112  goto err;
113  if (BIO_puts(bp, "SSL-Session:\n") <= 0)
114  goto err;
115  if (x->ssl_version == SSL2_VERSION)
116  s = "SSLv2";
117  else if (x->ssl_version == SSL3_VERSION)
118  s = "SSLv3";
119  else if (x->ssl_version == TLS1_2_VERSION)
120  s = "TLSv1.2";
121  else if (x->ssl_version == TLS1_1_VERSION)
122  s = "TLSv1.1";
123  else if (x->ssl_version == TLS1_VERSION)
124  s = "TLSv1";
125  else if (x->ssl_version == DTLS1_VERSION)
126  s = "DTLSv1";
127  else if (x->ssl_version == DTLS1_2_VERSION)
128  s = "DTLSv1.2";
129  else if (x->ssl_version == DTLS1_BAD_VER)
130  s = "DTLSv1-bad";
131  else
132  s = "unknown";
133  if (BIO_printf(bp, " Protocol : %s\n", s) <= 0)
134  goto err;
135 
136  if (x->cipher == NULL) {
137  if (((x->cipher_id) & 0xff000000) == 0x02000000) {
138  if (BIO_printf
139  (bp, " Cipher : %06lX\n", x->cipher_id & 0xffffff) <= 0)
140  goto err;
141  } else {
142  if (BIO_printf
143  (bp, " Cipher : %04lX\n", x->cipher_id & 0xffff) <= 0)
144  goto err;
145  }
146  } else {
147  if (BIO_printf
148  (bp, " Cipher : %s\n",
149  ((x->cipher == NULL) ? "unknown" : x->cipher->name)) <= 0)
150  goto err;
151  }
152  if (BIO_puts(bp, " Session-ID: ") <= 0)
153  goto err;
154  for (i = 0; i < x->session_id_length; i++) {
155  if (BIO_printf(bp, "%02X", x->session_id[i]) <= 0)
156  goto err;
157  }
158  if (BIO_puts(bp, "\n Session-ID-ctx: ") <= 0)
159  goto err;
160  for (i = 0; i < x->sid_ctx_length; i++) {
161  if (BIO_printf(bp, "%02X", x->sid_ctx[i]) <= 0)
162  goto err;
163  }
164  if (BIO_puts(bp, "\n Master-Key: ") <= 0)
165  goto err;
166  for (i = 0; i < (unsigned int)x->master_key_length; i++) {
167  if (BIO_printf(bp, "%02X", x->master_key[i]) <= 0)
168  goto err;
169  }
170  if (BIO_puts(bp, "\n Key-Arg : ") <= 0)
171  goto err;
172  if (x->key_arg_length == 0) {
173  if (BIO_puts(bp, "None") <= 0)
174  goto err;
175  } else
176  for (i = 0; i < x->key_arg_length; i++) {
177  if (BIO_printf(bp, "%02X", x->key_arg[i]) <= 0)
178  goto err;
179  }
180 #ifndef OPENSSL_NO_KRB5
181  if (BIO_puts(bp, "\n Krb5 Principal: ") <= 0)
182  goto err;
183  if (x->krb5_client_princ_len == 0) {
184  if (BIO_puts(bp, "None") <= 0)
185  goto err;
186  } else
187  for (i = 0; i < x->krb5_client_princ_len; i++) {
188  if (BIO_printf(bp, "%02X", x->krb5_client_princ[i]) <= 0)
189  goto err;
190  }
191 #endif /* OPENSSL_NO_KRB5 */
192 #ifndef OPENSSL_NO_PSK
193  if (BIO_puts(bp, "\n PSK identity: ") <= 0)
194  goto err;
195  if (BIO_printf(bp, "%s", x->psk_identity ? x->psk_identity : "None") <= 0)
196  goto err;
197  if (BIO_puts(bp, "\n PSK identity hint: ") <= 0)
198  goto err;
199  if (BIO_printf
200  (bp, "%s", x->psk_identity_hint ? x->psk_identity_hint : "None") <= 0)
201  goto err;
202 #endif
203 #ifndef OPENSSL_NO_SRP
204  if (BIO_puts(bp, "\n SRP username: ") <= 0)
205  goto err;
206  if (BIO_printf(bp, "%s", x->srp_username ? x->srp_username : "None") <= 0)
207  goto err;
208 #endif
209 #ifndef OPENSSL_NO_TLSEXT
210  if (x->tlsext_tick_lifetime_hint) {
211  if (BIO_printf(bp,
212  "\n TLS session ticket lifetime hint: %ld (seconds)",
213  x->tlsext_tick_lifetime_hint) <= 0)
214  goto err;
215  }
216  if (x->tlsext_tick) {
217  if (BIO_puts(bp, "\n TLS session ticket:\n") <= 0)
218  goto err;
219  if (BIO_dump_indent(bp, (char *)x->tlsext_tick, x->tlsext_ticklen, 4)
220  <= 0)
221  goto err;
222  }
223 #endif
224 
225 #ifndef OPENSSL_NO_COMP
226  if (x->compress_meth != 0) {
227  SSL_COMP *comp = NULL;
228 
229  ssl_cipher_get_evp(x, NULL, NULL, NULL, NULL, &comp);
230  if (comp == NULL) {
231  if (BIO_printf(bp, "\n Compression: %d", x->compress_meth) <=
232  0)
233  goto err;
234  } else {
235  if (BIO_printf
236  (bp, "\n Compression: %d (%s)", comp->id,
237  comp->method->name) <= 0)
238  goto err;
239  }
240  }
241 #endif
242  if (x->time != 0L) {
243  if (BIO_printf(bp, "\n Start Time: %ld", x->time) <= 0)
244  goto err;
245  }
246  if (x->timeout != 0L) {
247  if (BIO_printf(bp, "\n Timeout : %ld (sec)", x->timeout) <= 0)
248  goto err;
249  }
250  if (BIO_puts(bp, "\n") <= 0)
251  goto err;
252 
253  if (BIO_puts(bp, " Verify return code: ") <= 0)
254  goto err;
255  if (BIO_printf(bp, "%ld (%s)\n", x->verify_result,
256  X509_verify_cert_error_string(x->verify_result)) <= 0)
257  goto err;
258 
259  return (1);
260  err:
261  return (0);
262 }
