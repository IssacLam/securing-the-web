102 static ASN1_OCTET_STRING *s2i_skey_id(X509V3_EXT_METHOD *method,
103  X509V3_CTX *ctx, char *str)
104 {
105  ASN1_OCTET_STRING *oct;
106  ASN1_BIT_STRING *pk;
107  unsigned char pkey_dig[EVP_MAX_MD_SIZE];
108  unsigned int diglen;
109 
110  if (strcmp(str, "hash"))
111  return s2i_ASN1_OCTET_STRING(method, ctx, str);
112 
113  if (!(oct = M_ASN1_OCTET_STRING_new())) {
114  X509V3err(X509V3_F_S2I_SKEY_ID, ERR_R_MALLOC_FAILURE);
115  return NULL;
116  }
117 
118  if (ctx && (ctx->flags == CTX_TEST))
119  return oct;
120 
121  if (!ctx || (!ctx->subject_req && !ctx->subject_cert)) {
122  X509V3err(X509V3_F_S2I_SKEY_ID, X509V3_R_NO_PUBLIC_KEY);
123  goto err;
124  }
125 
126  if (ctx->subject_req)
127  pk = ctx->subject_req->req_info->pubkey->public_key;
128  else
129  pk = ctx->subject_cert->cert_info->key->public_key;
130 
131  if (!pk) {
132  X509V3err(X509V3_F_S2I_SKEY_ID, X509V3_R_NO_PUBLIC_KEY);
133  goto err;
134  }
135 
136  if (!EVP_Digest
137  (pk->data, pk->length, pkey_dig, &diglen, EVP_sha1(), NULL))
138  goto err;
139 
140  if (!M_ASN1_OCTET_STRING_set(oct, pkey_dig, diglen)) {
141  X509V3err(X509V3_F_S2I_SKEY_ID, ERR_R_MALLOC_FAILURE);
142  goto err;
143  }
144 
145  return oct;
146 
147  err:
148  M_ASN1_OCTET_STRING_free(oct);
149  return NULL;
150 }
