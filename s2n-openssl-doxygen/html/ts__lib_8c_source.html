71 int TS_ASN1_INTEGER_print_bio(BIO *bio, const ASN1_INTEGER *num)
72 {
73  BIGNUM num_bn;
74  int result = 0;
75  char *hex;
76 
77  BN_init(&num_bn);
78  ASN1_INTEGER_to_BN(num, &num_bn);
79  if ((hex = BN_bn2hex(&num_bn))) {
80  result = BIO_write(bio, "0x", 2) > 0;
81  result = result && BIO_write(bio, hex, strlen(hex)) > 0;
82  OPENSSL_free(hex);
83  }
84  BN_free(&num_bn);
85 
86  return result;
87 }

101 {

107  n = X509v3_get_ext_count(extensions);
108  for (i = 0; i < n; i++) {
109  ex = X509v3_get_ext(extensions, i);
110  obj = X509_EXTENSION_get_object(ex);
111  i2a_ASN1_OBJECT(bio, obj);
112  critical = X509_EXTENSION_get_critical(ex);
113  BIO_printf(bio, ": %s\n", critical ? "critical" : "");
114  if (!X509V3_EXT_print(bio, ex, 0, 4)) {
115  BIO_printf(bio, "%4s", "");
116  M_ASN1_OCTET_STRING_print(bio, ex->value);
117  }
118  BIO_write(bio, "\n", 1);
119  }
120 
121  return 1;
122 }
