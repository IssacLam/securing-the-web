77 int EVP_SignFinal(EVP_MD_CTX *ctx, unsigned char *sigret,
78  unsigned int *siglen, EVP_PKEY *pkey)
79 {
80  unsigned char m[EVP_MAX_MD_SIZE];
81  unsigned int m_len;
82  int i = 0, ok = 0, v;
83  EVP_MD_CTX tmp_ctx;
84  EVP_PKEY_CTX *pkctx = NULL;
85 
86  *siglen = 0;
87  EVP_MD_CTX_init(&tmp_ctx);
88  if (!EVP_MD_CTX_copy_ex(&tmp_ctx, ctx))
89  goto err;
90  if (!EVP_DigestFinal_ex(&tmp_ctx, &(m[0]), &m_len))
91  goto err;
92  EVP_MD_CTX_cleanup(&tmp_ctx);
93 
94  if (ctx->digest->flags & EVP_MD_FLAG_PKEY_METHOD_SIGNATURE) {
95  size_t sltmp = (size_t)EVP_PKEY_size(pkey);
96  i = 0;
97  pkctx = EVP_PKEY_CTX_new(pkey, NULL);
98  if (!pkctx)
99  goto err;
100  if (EVP_PKEY_sign_init(pkctx) <= 0)
101  goto err;
102  if (EVP_PKEY_CTX_set_signature_md(pkctx, ctx->digest) <= 0)
103  goto err;
104  if (EVP_PKEY_sign(pkctx, sigret, &sltmp, m, m_len) <= 0)
105  goto err;
106  *siglen = sltmp;
107  i = 1;
108  err:
109  EVP_PKEY_CTX_free(pkctx);
110  return i;
111  }
112 
113  for (i = 0; i < 4; i++) {
114  v = ctx->digest->required_pkey_type[i];
115  if (v == 0)
116  break;
117  if (pkey->type == v) {
118  ok = 1;
119  break;
120  }
121  }
122  if (!ok) {
123  EVPerr(EVP_F_EVP_SIGNFINAL, EVP_R_WRONG_PUBLIC_KEY_TYPE);
124  return (0);
125  }
126 
127  if (ctx->digest->sign == NULL) {
128  EVPerr(EVP_F_EVP_SIGNFINAL, EVP_R_NO_SIGN_FUNCTION_CONFIGURED);
129  return (0);
130  }
131  return (ctx->digest->sign(ctx->digest->type, m, m_len, sigret, siglen,
132  pkey->pkey.ptr));
133 }
