38 int s2n_client_hello_recv(struct s2n_connection *conn)
39 {
40  struct s2n_stuffer *in = &conn->handshake.io;
41  uint8_t session_id[S2N_TLS_SESSION_ID_LEN];
42  uint8_t session_id_len;
43  uint8_t compression_methods;
44  uint16_t extensions_size;
45  uint16_t cipher_suites_length;
46  uint8_t *cipher_suites;
47  uint8_t client_protocol_version[S2N_TLS_PROTOCOL_VERSION_LEN];
48 
49  GUARD(s2n_stuffer_read_bytes(in, client_protocol_version, S2N_TLS_PROTOCOL_VERSION_LEN));
50  GUARD(s2n_stuffer_read_bytes(in, conn->pending.client_random, S2N_TLS_RANDOM_DATA_LEN));
51  GUARD(s2n_stuffer_read_uint8(in, &session_id_len));
52 
53  conn->client_protocol_version = (client_protocol_version[0] * 10) + client_protocol_version[1];
54  if (conn->client_protocol_version < conn->config->cipher_preferences->minimum_protocol_version || conn->client_protocol_version > S2N_TLS12) {
55  GUARD(s2n_queue_reader_unsupported_protocol_version_alert(conn));
56  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
57  }
58  conn->client_hello_version = conn->client_protocol_version;
59 
60  if (session_id_len > S2N_TLS_SESSION_ID_LEN) {
61  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
62  }
63 
64  GUARD(s2n_stuffer_read_bytes(in, session_id, session_id_len));
65 
66  GUARD(s2n_stuffer_read_uint16(in, &cipher_suites_length));
67  if (cipher_suites_length % S2N_TLS_CIPHER_SUITE_LEN) {
68  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
69  }
70  cipher_suites = s2n_stuffer_raw_read(in, cipher_suites_length);
71  notnull_check(cipher_suites);
72  /* Don't choose the cipher yet, read the extensions first */
73 
74  GUARD(s2n_stuffer_read_uint8(in, &compression_methods));
75  GUARD(s2n_stuffer_skip_read(in, compression_methods));
76 
77  /* This is going to be our default if the client has no preference. */
78  conn->pending.server_ecc_params.negotiated_curve = &s2n_ecc_supported_curves[0];
79 
80  if (s2n_stuffer_data_available(in) >= 2) {
81  /* Read extensions if they are present */
82  GUARD(s2n_stuffer_read_uint16(in, &extensions_size));
83 
84  if (extensions_size > s2n_stuffer_data_available(in)) {
85  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
86  }
87 
88  struct s2n_blob extensions;
89  extensions.size = extensions_size;
90  extensions.data = s2n_stuffer_raw_read(in, extensions.size);
91 
92  GUARD(s2n_client_extensions_recv(conn, &extensions));
93  }
94 
95  /* Now choose the ciphers and the cert chain. */
96  GUARD(s2n_set_cipher_as_tls_server(conn, cipher_suites, cipher_suites_length / 2));
97  conn->server->chosen_cert_chain = conn->config->cert_and_key_pairs;
98 
99  conn->handshake.next_state = SERVER_HELLO;
100 
101  return 0;
102 }

104 int s2n_client_hello_send(struct s2n_connection *conn)
105 {
106  uint32_t gmt_unix_time = time(NULL);
107  struct s2n_stuffer *out = &conn->handshake.io;
108  struct s2n_stuffer client_random;
109  struct s2n_blob b, r;
110  uint8_t session_id_len = 0;
111  uint8_t client_protocol_version[S2N_TLS_PROTOCOL_VERSION_LEN];
112 
113  b.data = conn->pending.client_random;
114  b.size = S2N_TLS_RANDOM_DATA_LEN;
115 
116  /* Create the client random data */
117  GUARD(s2n_stuffer_init(&client_random, &b));
118  GUARD(s2n_stuffer_write_uint32(&client_random, gmt_unix_time));
119 
120  r.data = s2n_stuffer_raw_write(&client_random, S2N_TLS_RANDOM_DATA_LEN - 4);
121  r.size = S2N_TLS_RANDOM_DATA_LEN - 4;
122  notnull_check(r.data);
123  GUARD(s2n_get_public_random_data(&r));
124 
125  client_protocol_version[0] = conn->client_protocol_version / 10;
126  client_protocol_version[1] = conn->client_protocol_version % 10;
127  conn->client_hello_version = conn->client_protocol_version;
128 
129  GUARD(s2n_stuffer_write_bytes(out, client_protocol_version, S2N_TLS_PROTOCOL_VERSION_LEN));
130  GUARD(s2n_stuffer_copy(&client_random, out, S2N_TLS_RANDOM_DATA_LEN));
131  GUARD(s2n_stuffer_write_uint8(out, session_id_len));
132  GUARD(s2n_stuffer_write_uint16(out, conn->config->cipher_preferences->count * S2N_TLS_CIPHER_SUITE_LEN));
133  GUARD(s2n_stuffer_write_bytes(out, conn->config->cipher_preferences->wire_format, conn->config->cipher_preferences->count * S2N_TLS_CIPHER_SUITE_LEN));
134 
135  /* Zero compression methods */
136  GUARD(s2n_stuffer_write_uint8(out, 1));
137  GUARD(s2n_stuffer_write_uint8(out, 0));
138 
139  /* Write the extensions */
140  GUARD(s2n_client_extensions_send(conn, out));
141 
142  conn->handshake.next_state = SERVER_HELLO;
143 
144  return 0;
145 }

147 /* See http://www-archive.mozilla.org/projects/security/pki/nss/ssl/draft02.html 2.5 */
148 int s2n_sslv2_client_hello_recv(struct s2n_connection *conn)
149 {
150  struct s2n_stuffer *in = &conn->handshake.io;
151  uint16_t session_id_length;
152  uint16_t cipher_suites_length;
153  uint16_t challenge_length;
154  uint8_t *cipher_suites;
155 
156  if (conn->client_protocol_version < conn->config->cipher_preferences->minimum_protocol_version || conn->client_protocol_version > S2N_TLS12) {
157  GUARD(s2n_queue_reader_unsupported_protocol_version_alert(conn));
158  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
159  }
160 
161  /* We start 5 bytes into the record */
162  GUARD(s2n_stuffer_read_uint16(in, &cipher_suites_length));
163 
164  if (cipher_suites_length % S2N_SSLv2_CIPHER_SUITE_LEN) {
165  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
166  }
167 
168  GUARD(s2n_stuffer_read_uint16(in, &session_id_length));
169 
170  if (session_id_length) {
171  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
172  }
173 
174  GUARD(s2n_stuffer_read_uint16(in, &challenge_length));
175 
176  if (challenge_length > S2N_TLS_RANDOM_DATA_LEN) {
177  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
178  }
179 
180  cipher_suites = s2n_stuffer_raw_read(in, cipher_suites_length);
181  notnull_check(cipher_suites);
182  GUARD(s2n_set_cipher_as_sslv2_server(conn, cipher_suites, cipher_suites_length / S2N_SSLv2_CIPHER_SUITE_LEN));
183 
184  struct s2n_blob b;
185  b.data = conn->pending.client_random;
186  b.size = S2N_TLS_RANDOM_DATA_LEN;
187 
188  b.data += S2N_TLS_RANDOM_DATA_LEN - challenge_length;
189  b.size -= S2N_TLS_RANDOM_DATA_LEN - challenge_length;
190 
191  GUARD(s2n_stuffer_read(in, &b));
192 
193  conn->server->chosen_cert_chain = conn->config->cert_and_key_pairs;
194  conn->client_hello_version = S2N_SSLv2;
195  conn->handshake.next_state = SERVER_HELLO;
196 
197  return 0;
198 }
