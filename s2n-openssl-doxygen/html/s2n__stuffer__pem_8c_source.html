29 static int s2n_stuffer_data_from_pem(struct s2n_stuffer *pem, struct s2n_stuffer *asn1, const char *keyword)
30 {
31  uint8_t linepad[S2N_PEM_LINE_LENGTH + 1];
32  struct s2n_blob line_blob = {.data = linepad,.size = S2N_PEM_LINE_LENGTH + 1 };
33  struct s2n_stuffer line;
34  uint8_t *field;
35 
36  GUARD(s2n_stuffer_init(&line, &line_blob));
37  GUARD(s2n_stuffer_read_token(pem, &line, '\n'));
38 
39  /* Check that the line matches the header */
40  field = s2n_stuffer_raw_read(&line, sizeof(S2N_PEM_BEGIN_TOKEN) - 1);
41  notnull_check(field);
42  if (memcmp(field, S2N_PEM_BEGIN_TOKEN, sizeof(S2N_PEM_BEGIN_TOKEN) - 1)) {
43  S2N_ERROR(S2N_ERR_INVALID_PEM);
44  }
45 
46  field = s2n_stuffer_raw_read(&line, strlen(keyword));
47  notnull_check(field);
48  if (memcmp(field, keyword, strlen(keyword))) {
49  S2N_ERROR(S2N_ERR_INVALID_PEM);
50  }
51 
52  field = s2n_stuffer_raw_read(&line, sizeof(S2N_PEM_LINE) - 1);
53  notnull_check(field);
54  if (memcmp(field, S2N_PEM_LINE, sizeof(S2N_PEM_LINE) - 1)) {
55  S2N_ERROR(S2N_ERR_INVALID_PEM);
56  }
57 
58  /* Get the actual base64 data */
59  do {
60  GUARD(s2n_stuffer_rewrite(&line));
61  GUARD(s2n_stuffer_read_token(pem, &line, '\n'));
62 
63  char c;
64  GUARD(s2n_stuffer_peek_char(&line, &c));
65  if (c == '-') {
66  GUARD(s2n_stuffer_reread(&line));
67  break;
68  }
69 
70  if (s2n_stuffer_read_base64(&line, asn1) < 0) {
71  GUARD(s2n_stuffer_reread(&line));
72  break;
73  }
74 
75  } while (1);
76 
77  /* Check that the line matches the trailer */
78  field = s2n_stuffer_raw_read(&line, sizeof(S2N_PEM_END_TOKEN) - 1);
79  notnull_check(field);
80  if (memcmp(field, S2N_PEM_END_TOKEN, sizeof(S2N_PEM_END_TOKEN) - 1)) {
81  S2N_ERROR(S2N_ERR_INVALID_PEM);
82  }
83 
84  field = s2n_stuffer_raw_read(&line, strlen(keyword));
85  notnull_check(field);
86  if (memcmp(field, keyword, strlen(keyword))) {
87  S2N_ERROR(S2N_ERR_INVALID_PEM);
88  }
89 
90  field = s2n_stuffer_raw_read(&line, sizeof(S2N_PEM_LINE) - 1);
91  notnull_check(field);
92  if (memcmp(field, S2N_PEM_LINE, sizeof(S2N_PEM_LINE) - 1)) {
93  S2N_ERROR(S2N_ERR_INVALID_PEM);
94  }
95 
96  return 0;
97 }
