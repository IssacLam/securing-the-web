35 /* How much overhead does the IV, MAC, TAG and padding bytes introduce ? */
36 static uint16_t overhead(struct s2n_connection *conn)
37 {
38  struct s2n_crypto_parameters *active = conn->server;
39 
40  if (conn->mode == S2N_CLIENT) {
41  active = conn->client;
42  }
43 
44  uint16_t extra = s2n_hmac_digest_size(active->cipher_suite->hmac_alg);
45 
46  if (active->cipher_suite->cipher->type == S2N_CBC) {
47  /* Subtract one for the padding length byte */
48  extra += 1;
49 
50  if (conn->actual_protocol_version > S2N_TLS10) {
51  extra += active->cipher_suite->cipher->io.cbc.record_iv_size;
52  }
53  } else if (active->cipher_suite->cipher->type == S2N_AEAD) {
54  extra += active->cipher_suite->cipher->io.aead.tag_size;
55  extra += active->cipher_suite->cipher->io.aead.record_iv_size;
56  }
57 
58  return extra;
59 }

61 int s2n_record_max_write_payload_size(struct s2n_connection *conn)
62 {
63  uint16_t max_fragment_size = conn->max_fragment_length;
64  struct s2n_crypto_parameters *active = conn->server;
65 
66  if (conn->mode == S2N_CLIENT) {
67  active = conn->client;
68  }
69 
70  /* Round the fragment size down to be block aligned */
71  if (active->cipher_suite->cipher->type == S2N_CBC) {
72  max_fragment_size -= max_fragment_size % active->cipher_suite->cipher->io.cbc.block_size;
73  }
74 
75  return max_fragment_size - overhead(conn);
76 }

78 int s2n_record_write(struct s2n_connection *conn, uint8_t content_type, struct s2n_blob *in)
79 {
80  struct s2n_blob out, iv, aad;
81  uint8_t padding = 0;
82  uint16_t block_size = 0;
83  uint8_t protocol_version[S2N_TLS_PROTOCOL_VERSION_LEN];
84  uint8_t aad_gen[S2N_TLS_MAX_AAD_LEN] = { 0 };
85  uint8_t aad_iv[S2N_TLS_MAX_IV_LEN] = { 0 };
86 
87  uint8_t *sequence_number = conn->server->server_sequence_number;
88  struct s2n_hmac_state *mac = &conn->server->server_record_mac;
89  struct s2n_session_key *session_key = &conn->server->server_key;
90  struct s2n_cipher_suite *cipher_suite = conn->server->cipher_suite;
91  uint8_t *implicit_iv = conn->server->server_implicit_iv;
92 
93  if (conn->mode == S2N_CLIENT) {
94  sequence_number = conn->client->client_sequence_number;
95  mac = &conn->client->client_record_mac;
96  session_key = &conn->client->client_key;
97  cipher_suite = conn->client->cipher_suite;
98  implicit_iv = conn->client->client_implicit_iv;
99  }
100 
101  if (s2n_stuffer_data_available(&conn->out)) {
102  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
103  }
104 
105  int mac_digest_size = s2n_hmac_digest_size(mac->alg);
106  gte_check(mac_digest_size, 0);
107 
108  /* Before we do anything, we need to figure out what the length of the
109  * fragment is going to be.
110  */
111  uint16_t data_bytes_to_take = in->size;
112  if (data_bytes_to_take > s2n_record_max_write_payload_size(conn)) {
113  data_bytes_to_take = s2n_record_max_write_payload_size(conn);
114  }
115  uint16_t extra = overhead(conn);
116 
117  /* If we have padding to worry about, figure that out too */
118  if (cipher_suite->cipher->type == S2N_CBC) {
119  block_size = cipher_suite->cipher->io.cbc.block_size;
120  if (((data_bytes_to_take + extra) % block_size)) {
121  padding = block_size - ((data_bytes_to_take + extra) % block_size);
122  }
123  }
124 
125  /* Start the MAC with the sequence number */
126  GUARD(s2n_hmac_update(mac, sequence_number, S2N_TLS_SEQUENCE_NUM_LEN));
127 
128  /* Now that we know the length, start writing the record */
129  protocol_version[0] = conn->actual_protocol_version / 10;
130  protocol_version[1] = conn->actual_protocol_version % 10;
131  GUARD(s2n_stuffer_write_uint8(&conn->out, content_type));
132  GUARD(s2n_stuffer_write_bytes(&conn->out, protocol_version, S2N_TLS_PROTOCOL_VERSION_LEN));
133 
134  /* First write a header that has the payload length, this is for the MAC */
135  GUARD(s2n_stuffer_write_uint16(&conn->out, data_bytes_to_take));
136 
137  if (conn->actual_protocol_version > S2N_SSLv3) {
138  GUARD(s2n_hmac_update(mac, conn->out.blob.data, S2N_TLS_RECORD_HEADER_LENGTH));
139  } else {
140  /* SSLv3 doesn't include the protocol version in the MAC */
141  GUARD(s2n_hmac_update(mac, conn->out.blob.data, 1));
142  GUARD(s2n_hmac_update(mac, conn->out.blob.data + 3, 2));
143  }
144 
145  /* Rewrite the length to be the actual fragment length */
146  uint16_t actual_fragment_length = data_bytes_to_take + padding + extra;
147  GUARD(s2n_stuffer_wipe_n(&conn->out, 2));
148  GUARD(s2n_stuffer_write_uint16(&conn->out, actual_fragment_length));
149 
150  /* If we're AEAD, write the sequence number as an IV, and generate the AAD */
151  if (cipher_suite->cipher->type == S2N_AEAD) {
152  GUARD(s2n_stuffer_write_bytes(&conn->out, sequence_number, S2N_TLS_SEQUENCE_NUM_LEN));
153 
154  struct s2n_stuffer iv_stuffer;
155  iv.data = aad_iv;
156  iv.size = sizeof(aad_iv);
157 
158  GUARD(s2n_stuffer_init(&iv_stuffer, &iv));
159  GUARD(s2n_stuffer_write_bytes(&iv_stuffer, implicit_iv, cipher_suite->cipher->io.aead.fixed_iv_size));
160  GUARD(s2n_stuffer_write_bytes(&iv_stuffer, sequence_number, S2N_TLS_SEQUENCE_NUM_LEN));
161 
162  /* Set the IV size to the amount of data written */
163  iv.size = s2n_stuffer_data_available(&iv_stuffer);
164 
165  aad.data = aad_gen;
166  aad.size = sizeof(aad_gen);
167 
168  struct s2n_stuffer ad_stuffer;
169  GUARD(s2n_stuffer_init(&ad_stuffer, &aad));
170  GUARD(s2n_aead_aad_init(conn, sequence_number, content_type, data_bytes_to_take, &ad_stuffer));
171  } else if (cipher_suite->cipher->type == S2N_CBC) {
172  iv.size = block_size;
173  iv.data = implicit_iv;
174 
175  /* For TLS1.1/1.2; write the IV with random data */
176  if (conn->actual_protocol_version > S2N_TLS10) {
177  GUARD(s2n_get_public_random_data(&iv));
178  GUARD(s2n_stuffer_write(&conn->out, &iv));
179  }
180  }
181 
182  /* We are done with this sequence number, so we can increment it */
183  struct s2n_blob seq = {.data = sequence_number, .size = S2N_TLS_SEQUENCE_NUM_LEN };
184  GUARD(s2n_increment_sequence_number(&seq));
185 
186  /* Write the plaintext data */
187  out.data = in->data;
188  out.size = data_bytes_to_take;
189  GUARD(s2n_stuffer_write(&conn->out, &out));
190  GUARD(s2n_hmac_update(mac, out.data, out.size));
191 
192  /* Write the digest */
193  uint8_t *digest = s2n_stuffer_raw_write(&conn->out, mac_digest_size);
194  notnull_check(digest);
195 
196  GUARD(s2n_hmac_digest(mac, digest, mac_digest_size));
197  GUARD(s2n_hmac_reset(mac));
198 
199  if (cipher_suite->cipher->type == S2N_CBC) {
200  /* Include padding bytes, each with the value 'p', and
201  * include an extra padding length byte, also with the value 'p'.
202  */
203  for (int i = 0; i <= padding; i++) {
204  GUARD(s2n_stuffer_write_uint8(&conn->out, padding));
205  }
206  }
207 
208  /* Rewind to rewrite/encrypt the packet */
209  GUARD(s2n_stuffer_rewrite(&conn->out));
210 
211  /* Skip the header */
212  GUARD(s2n_stuffer_skip_write(&conn->out, S2N_TLS_RECORD_HEADER_LENGTH));
213 
214  struct s2n_blob en;
215  uint16_t encrypted_length = data_bytes_to_take + mac_digest_size;
216 
217  if (cipher_suite->cipher->type == S2N_AEAD) {
218  encrypted_length += cipher_suite->cipher->io.aead.record_iv_size;
219  encrypted_length += cipher_suite->cipher->io.aead.tag_size;
220  }
221 
222  if (cipher_suite->cipher->type == S2N_CBC) {
223  if (conn->actual_protocol_version > S2N_TLS10) {
224  /* Leave the IV alone and unencrypted */
225  GUARD(s2n_stuffer_skip_write(&conn->out, iv.size));
226  }
227 
228  /* Encrypt the padding and the padding length byte too */
229  encrypted_length += padding + 1;
230  }
231 
232  /* Do the encryption */
233  en.size = encrypted_length;
234  en.data = s2n_stuffer_raw_write(&conn->out, en.size);
235  notnull_check(en.data);
236 
237  switch (cipher_suite->cipher->type) {
238  case S2N_STREAM:
239  GUARD(cipher_suite->cipher->io.stream.encrypt(session_key, &en, &en));
240  break;
241  case S2N_CBC:
242  GUARD(cipher_suite->cipher->io.cbc.encrypt(session_key, &iv, &en, &en));
243 
244  /* Copy the last encrypted block to be the next IV */
245  gte_check(en.size, block_size);
246  memcpy_check(implicit_iv, en.data + en.size - block_size, block_size);
247  break;
248  case S2N_AEAD:
249  GUARD(cipher_suite->cipher->io.aead.encrypt(session_key, &iv, &aad, &en, &en));
250  break;
251  default:
252  return -1;
253  break;
254  }
255 
256  conn->wire_bytes_out += actual_fragment_length + S2N_TLS_RECORD_HEADER_LENGTH;
257 
258  return data_bytes_to_take;
259 }
