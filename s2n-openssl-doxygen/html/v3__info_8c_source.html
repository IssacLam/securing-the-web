110 static STACK_OF(CONF_VALUE) *i2v_AUTHORITY_INFO_ACCESS(X509V3_EXT_METHOD
111  *method, AUTHORITY_INFO_ACCESS
112  *ainfo, STACK_OF(CONF_VALUE)
113  *ret)
114 {
115  ACCESS_DESCRIPTION *desc;
116  int i, nlen;
117  char objtmp[80], *ntmp;
118  CONF_VALUE *vtmp;
119  for (i = 0; i < sk_ACCESS_DESCRIPTION_num(ainfo); i++) {
120  desc = sk_ACCESS_DESCRIPTION_value(ainfo, i);
121  ret = i2v_GENERAL_NAME(method, desc->location, ret);
122  if (!ret)
123  break;
124  vtmp = sk_CONF_VALUE_value(ret, i);
125  i2t_ASN1_OBJECT(objtmp, sizeof objtmp, desc->method);
126  nlen = strlen(objtmp) + strlen(vtmp->name) + 5;
127  ntmp = OPENSSL_malloc(nlen);
128  if (!ntmp) {
129  X509V3err(X509V3_F_I2V_AUTHORITY_INFO_ACCESS,
130  ERR_R_MALLOC_FAILURE);
131  return NULL;
132  }
133  BUF_strlcpy(ntmp, objtmp, nlen);
134  BUF_strlcat(ntmp, " - ", nlen);
135  BUF_strlcat(ntmp, vtmp->name, nlen);
136  OPENSSL_free(vtmp->name);
137  vtmp->name = ntmp;
138 
139  }
140  if (!ret)
141  return sk_CONF_VALUE_new_null();
142  return ret;
143 }

145 static AUTHORITY_INFO_ACCESS *v2i_AUTHORITY_INFO_ACCESS(X509V3_EXT_METHOD
146  *method,
147  X509V3_CTX *ctx,
148  STACK_OF(CONF_VALUE)
149  *nval)
150 {
151  AUTHORITY_INFO_ACCESS *ainfo = NULL;
152  CONF_VALUE *cnf, ctmp;
153  ACCESS_DESCRIPTION *acc;
154  int i, objlen;
155  char *objtmp, *ptmp;
156  if (!(ainfo = sk_ACCESS_DESCRIPTION_new_null())) {
157  X509V3err(X509V3_F_V2I_AUTHORITY_INFO_ACCESS, ERR_R_MALLOC_FAILURE);
158  return NULL;
159  }
160  for (i = 0; i < sk_CONF_VALUE_num(nval); i++) {
161  cnf = sk_CONF_VALUE_value(nval, i);
162  if (!(acc = ACCESS_DESCRIPTION_new())
163  || !sk_ACCESS_DESCRIPTION_push(ainfo, acc)) {
164  X509V3err(X509V3_F_V2I_AUTHORITY_INFO_ACCESS,
165  ERR_R_MALLOC_FAILURE);
166  goto err;
167  }
168  ptmp = strchr(cnf->name, ';');
169  if (!ptmp) {
170  X509V3err(X509V3_F_V2I_AUTHORITY_INFO_ACCESS,
171  X509V3_R_INVALID_SYNTAX);
172  goto err;
173  }
174  objlen = ptmp - cnf->name;
175  ctmp.name = ptmp + 1;
176  ctmp.value = cnf->value;
177  if (!v2i_GENERAL_NAME_ex(acc->location, method, ctx, &ctmp, 0))
178  goto err;
179  if (!(objtmp = OPENSSL_malloc(objlen + 1))) {
180  X509V3err(X509V3_F_V2I_AUTHORITY_INFO_ACCESS,
181  ERR_R_MALLOC_FAILURE);
182  goto err;
183  }
184  strncpy(objtmp, cnf->name, objlen);
185  objtmp[objlen] = 0;
186  acc->method = OBJ_txt2obj(objtmp, 0);
187  if (!acc->method) {
188  X509V3err(X509V3_F_V2I_AUTHORITY_INFO_ACCESS,
189  X509V3_R_BAD_OBJECT);
190  ERR_add_error_data(2, "value=", objtmp);
191  OPENSSL_free(objtmp);
192  goto err;
193  }
194  OPENSSL_free(objtmp);
195 
196  }
197  return ainfo;
198  err:
199  sk_ACCESS_DESCRIPTION_pop_free(ainfo, ACCESS_DESCRIPTION_free);
200  return NULL;
201 }
