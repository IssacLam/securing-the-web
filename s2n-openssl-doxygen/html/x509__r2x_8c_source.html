68 X509 *X509_REQ_to_X509(X509_REQ *r, int days, EVP_PKEY *pkey)
69 {
70  X509 *ret = NULL;
71  X509_CINF *xi = NULL;
72  X509_NAME *xn;
73 
74  if ((ret = X509_new()) == NULL) {
75  X509err(X509_F_X509_REQ_TO_X509, ERR_R_MALLOC_FAILURE);
76  goto err;
77  }
78 
79  /* duplicate the request */
80  xi = ret->cert_info;
81 
82  if (sk_X509_ATTRIBUTE_num(r->req_info->attributes) != 0) {
83  if ((xi->version = M_ASN1_INTEGER_new()) == NULL)
84  goto err;
85  if (!ASN1_INTEGER_set(xi->version, 2))
86  goto err;
87 /*- xi->extensions=ri->attributes; <- bad, should not ever be done
88  ri->attributes=NULL; */
89  }
90 
91  xn = X509_REQ_get_subject_name(r);
92  if (X509_set_subject_name(ret, X509_NAME_dup(xn)) == 0)
93  goto err;
94  if (X509_set_issuer_name(ret, X509_NAME_dup(xn)) == 0)
95  goto err;
96 
97  if (X509_gmtime_adj(xi->validity->notBefore, 0) == NULL)
98  goto err;
99  if (X509_gmtime_adj(xi->validity->notAfter, (long)60 * 60 * 24 * days) ==
100  NULL)
101  goto err;
102 
103  X509_set_pubkey(ret, X509_REQ_get_pubkey(r));
104 
105  if (!X509_sign(ret, pkey, EVP_md5()))
106  goto err;
107  if (0) {
108  err:
109  X509_free(ret);
110  ret = NULL;
111  }
112  return (ret);
113 }
