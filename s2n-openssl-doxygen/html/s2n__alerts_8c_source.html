58 int s2n_process_alert_fragment(struct s2n_connection *conn)
59 {
60  if (s2n_stuffer_data_available(&conn->alert_in) == 2) {
61  S2N_ERROR(S2N_ERR_ALERT_PRESENT);
62  }
63 
64  while (s2n_stuffer_data_available(&conn->in)) {
65  uint8_t bytes_required = 2;
66 
67  /* Alerts are two bytes long, but can still be fragmented or coalesced */
68  if (s2n_stuffer_data_available(&conn->alert_in) == 1) {
69  bytes_required = 1;
70  }
71 
72  int bytes_to_read = bytes_required;
73  if (bytes_to_read > s2n_stuffer_data_available(&conn->in)) {
74  bytes_to_read = s2n_stuffer_data_available(&conn->in);
75  }
76 
77  GUARD(s2n_stuffer_copy(&conn->in, &conn->alert_in, bytes_to_read));
78 
79  if (s2n_stuffer_data_available(&conn->alert_in) == 2) {
80  conn->closed = 1;
81 
82  /* Close notifications are handled as shutdowns */
83  if (conn->alert_in_data[1] == S2N_TLS_ALERT_CLOSE_NOTIFY) {
84  return 0;
85  }
86 
87  /* All other alerts are treated as fatal errors (even warnings) */
88  S2N_ERROR(S2N_ERR_ALERT);
89  }
90  }
91 
92  return 0;
93 }

95 int s2n_queue_writer_close_alert(struct s2n_connection *conn)
96 {
97  uint8_t alert[2];
98  struct s2n_blob out = {.data = alert,.size = sizeof(alert) };
99 
100  /* If there is an alert pending, do nothing */
101  if (s2n_stuffer_data_available(&conn->writer_alert_out)) {
102  return 0;
103  }
104 
105  alert[0] = S2N_TLS_ALERT_LEVEL_FATAL;
106  alert[1] = S2N_TLS_ALERT_CLOSE_NOTIFY;
107 
108  GUARD(s2n_stuffer_write(&conn->writer_alert_out, &out));
109 
110  return 0;
111 }
