65 int i2d_PrivateKey(EVP_PKEY *a, unsigned char **pp)
66 {
67  if (a->ameth && a->ameth->old_priv_encode) {
68  return a->ameth->old_priv_encode(a, pp);
69  }
70  if (a->ameth && a->ameth->priv_encode) {
71  PKCS8_PRIV_KEY_INFO *p8 = EVP_PKEY2PKCS8(a);
72  int ret = 0;
73  if (p8 != NULL) {
74  ret = i2d_PKCS8_PRIV_KEY_INFO(p8, pp);
75  PKCS8_PRIV_KEY_INFO_free(p8);
76  }
77  return ret;
78  }
79  ASN1err(ASN1_F_I2D_PRIVATEKEY, ASN1_R_UNSUPPORTED_PUBLIC_KEY_TYPE);
80  return -1;
81 }
