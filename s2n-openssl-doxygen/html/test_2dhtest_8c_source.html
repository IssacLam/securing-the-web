101 int main(int argc, char *argv[])
102 {
103  BN_GENCB _cb;
104  DH *a;
105  DH *b = NULL;
106  char buf[12];
107  unsigned char *abuf = NULL, *bbuf = NULL;
108  int i, alen, blen, aout, bout, ret = 1;
109  BIO *out;
110 
111  CRYPTO_malloc_debug_init();
112  CRYPTO_dbg_set_options(V_CRYPTO_MDEBUG_ALL);
113  CRYPTO_mem_ctrl(CRYPTO_MEM_CHECK_ON);
114 
115 # ifdef OPENSSL_SYS_WIN32
116  CRYPTO_malloc_init();
117 # endif
118 
119  RAND_seed(rnd_seed, sizeof rnd_seed);
120 
121  out = BIO_new(BIO_s_file());
122  if (out == NULL)
123  EXIT(1);
124  BIO_set_fp(out, stdout, BIO_NOCLOSE);
125 
126  BN_GENCB_set(&_cb, &cb, out);
127  if (((a = DH_new()) == NULL) || !DH_generate_parameters_ex(a, 64,
128  DH_GENERATOR_5,
129  &_cb))
130  goto err;
131 
132  if (!DH_check(a, &i))
133  goto err;
134  if (i & DH_CHECK_P_NOT_PRIME)
135  BIO_puts(out, "p value is not prime\n");
136  if (i & DH_CHECK_P_NOT_SAFE_PRIME)
137  BIO_puts(out, "p value is not a safe prime\n");
138  if (i & DH_UNABLE_TO_CHECK_GENERATOR)
139  BIO_puts(out, "unable to check the generator value\n");
140  if (i & DH_NOT_SUITABLE_GENERATOR)
141  BIO_puts(out, "the g value is not a generator\n");
142 
143  BIO_puts(out, "\np =");
144  BN_print(out, a->p);
145  BIO_puts(out, "\ng =");
146  BN_print(out, a->g);
147  BIO_puts(out, "\n");
148 
149  b = DH_new();
150  if (b == NULL)
151  goto err;
152 
153  b->p = BN_dup(a->p);
154  b->g = BN_dup(a->g);
155  if ((b->p == NULL) || (b->g == NULL))
156  goto err;
157 
158  /* Set a to run with normal modexp and b to use constant time */
159  a->flags &= ~DH_FLAG_NO_EXP_CONSTTIME;
160  b->flags |= DH_FLAG_NO_EXP_CONSTTIME;
161 
162  if (!DH_generate_key(a))
163  goto err;
164  BIO_puts(out, "pri 1=");
165  BN_print(out, a->priv_key);
166  BIO_puts(out, "\npub 1=");
167  BN_print(out, a->pub_key);
168  BIO_puts(out, "\n");
169 
170  if (!DH_generate_key(b))
171  goto err;
172  BIO_puts(out, "pri 2=");
173  BN_print(out, b->priv_key);
174  BIO_puts(out, "\npub 2=");
175  BN_print(out, b->pub_key);
176  BIO_puts(out, "\n");
177 
178  alen = DH_size(a);
179  abuf = (unsigned char *)OPENSSL_malloc(alen);
180  aout = DH_compute_key(abuf, b->pub_key, a);
181 
182  BIO_puts(out, "key1 =");
183  for (i = 0; i < aout; i++) {
184  sprintf(buf, "%02X", abuf[i]);
185  BIO_puts(out, buf);
186  }
187  BIO_puts(out, "\n");
188 
189  blen = DH_size(b);
190  bbuf = (unsigned char *)OPENSSL_malloc(blen);
191  bout = DH_compute_key(bbuf, a->pub_key, b);
192 
193  BIO_puts(out, "key2 =");
194  for (i = 0; i < bout; i++) {
195  sprintf(buf, "%02X", bbuf[i]);
196  BIO_puts(out, buf);
197  }
198  BIO_puts(out, "\n");
199  if ((aout < 4) || (bout != aout) || (memcmp(abuf, bbuf, aout) != 0)) {
200  fprintf(stderr, "Error in DH routines\n");
201  ret = 1;
202  } else
203  ret = 0;
204  if (!run_rfc5114_tests())
205  ret = 1;
206  err:
207  ERR_print_errors_fp(stderr);
208 
209  if (abuf != NULL)
210  OPENSSL_free(abuf);
211  if (bbuf != NULL)
212  OPENSSL_free(bbuf);
213  if (b != NULL)
214  DH_free(b);
215  if (a != NULL)
216  DH_free(a);
217  BIO_free(out);
218 # ifdef OPENSSL_SYS_NETWARE
219  if (ret)
220  printf("ERROR: %d\n", ret);
221 # endif
222  EXIT(ret);
223  return (ret);
224 }

226 static int MS_CALLBACK cb(int p, int n, BN_GENCB *arg)
227 {
228  char c = '*';
229 
230  if (p == 0)
231  c = '.';
232  if (p == 1)
233  c = '+';
234  if (p == 2)
235  c = '*';
236  if (p == 3)
237  c = '\n';
238  BIO_write(arg->arg, &c, 1);
239  (void)BIO_flush(arg->arg);
240 # ifdef LINT
241  p = n;
242 # endif
243  return 1;
244 }
