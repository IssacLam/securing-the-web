65 int X509_certificate_type(X509 *x, EVP_PKEY *pkey)
66 {
67  EVP_PKEY *pk;
68  int ret = 0, i;
69 
70  if (x == NULL)
71  return (0);
72 
73  if (pkey == NULL)
74  pk = X509_get_pubkey(x);
75  else
76  pk = pkey;
77 
78  if (pk == NULL)
79  return (0);
80 
81  switch (pk->type) {
82  case EVP_PKEY_RSA:
83  ret = EVP_PK_RSA | EVP_PKT_SIGN;
84 /* if (!sign only extension) */
85  ret |= EVP_PKT_ENC;
86  break;
87  case EVP_PKEY_DSA:
88  ret = EVP_PK_DSA | EVP_PKT_SIGN;
89  break;
90  case EVP_PKEY_EC:
91  ret = EVP_PK_EC | EVP_PKT_SIGN | EVP_PKT_EXCH;
92  break;
93  case EVP_PKEY_DH:
94  ret = EVP_PK_DH | EVP_PKT_EXCH;
95  break;
96  case NID_id_GostR3410_94:
97  case NID_id_GostR3410_2001:
98  ret = EVP_PKT_EXCH | EVP_PKT_SIGN;
99  break;
100  default:
101  break;
102  }
103 
104  i = OBJ_obj2nid(x->sig_alg->algorithm);
105  if (i && OBJ_find_sigid_algs(i, NULL, &i)) {
106 
107  switch (i) {
108  case NID_rsaEncryption:
109  case NID_rsa:
110  ret |= EVP_PKS_RSA;
111  break;
112  case NID_dsa:
113  case NID_dsa_2:
114  ret |= EVP_PKS_DSA;
115  break;
116  case NID_X9_62_id_ecPublicKey:
117  ret |= EVP_PKS_EC;
118  break;
119  default:
120  break;
121  }
122  }
123 
124  if (pkey == NULL)
125  EVP_PKEY_free(pk);
126  return (ret);
127 }
