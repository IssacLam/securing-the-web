64 int ASN1_TYPE_set_octetstring(ASN1_TYPE *a, unsigned char *data, int len)
65 {
66  ASN1_STRING *os;
67 
68  if ((os = M_ASN1_OCTET_STRING_new()) == NULL)
69  return (0);
70  if (!M_ASN1_OCTET_STRING_set(os, data, len)) {
71  M_ASN1_OCTET_STRING_free(os);
72  return 0;
73  }
74  ASN1_TYPE_set(a, V_ASN1_OCTET_STRING, os);
75  return (1);
76 }

79 int ASN1_TYPE_get_octetstring(ASN1_TYPE *a, unsigned char *data, int max_len)
80 {
81  int ret, num;
82  unsigned char *p;
83 
84  if ((a->type != V_ASN1_OCTET_STRING) || (a->value.octet_string == NULL)) {
85  ASN1err(ASN1_F_ASN1_TYPE_GET_OCTETSTRING, ASN1_R_DATA_IS_WRONG);
86  return (-1);
87  }
88  p = M_ASN1_STRING_data(a->value.octet_string);
89  ret = M_ASN1_STRING_length(a->value.octet_string);
90  if (ret < max_len)
91  num = ret;
92  else
93  num = max_len;
94  memcpy(data, p, num);
95  return (ret);
96 }

98 int ASN1_TYPE_set_int_octetstring(ASN1_TYPE *a, long num, unsigned char *data,
99  int len)
100 {
101  int n, size;
102  ASN1_OCTET_STRING os, *osp;
103  ASN1_INTEGER in;
104  unsigned char *p;
105  unsigned char buf[32]; /* when they have 256bit longs, I'll be in
106  * trouble */
107  in.data = buf;
108  in.length = 32;
109  os.data = data;
110  os.type = V_ASN1_OCTET_STRING;
111  os.length = len;
112  ASN1_INTEGER_set(&in, num);
113  n = i2d_ASN1_INTEGER(&in, NULL);
114  n += M_i2d_ASN1_OCTET_STRING(&os, NULL);
115 
116  size = ASN1_object_size(1, n, V_ASN1_SEQUENCE);
117 
118  if ((osp = ASN1_STRING_new()) == NULL)
119  return (0);
120  /* Grow the 'string' */
121  if (!ASN1_STRING_set(osp, NULL, size)) {
122  ASN1_STRING_free(osp);
123  return (0);
124  }
125 
126  M_ASN1_STRING_length_set(osp, size);
127  p = M_ASN1_STRING_data(osp);
128 
129  ASN1_put_object(&p, 1, n, V_ASN1_SEQUENCE, V_ASN1_UNIVERSAL);
130  i2d_ASN1_INTEGER(&in, &p);
131  M_i2d_ASN1_OCTET_STRING(&os, &p);
132 
133  ASN1_TYPE_set(a, V_ASN1_SEQUENCE, osp);
134  return (1);
135 }

141 /* int max_len: for returned value */
142 int ASN1_TYPE_get_int_octetstring(ASN1_TYPE *a, long *num,
143  unsigned char *data, int max_len)
144 {
145  int ret = -1, n;
146  ASN1_INTEGER *ai = NULL;
147  ASN1_OCTET_STRING *os = NULL;
148  const unsigned char *p;
149  long length;
150  ASN1_const_CTX c;
151 
152  if ((a->type != V_ASN1_SEQUENCE) || (a->value.sequence == NULL)) {
153  goto err;
154  }
155  p = M_ASN1_STRING_data(a->value.sequence);
156  length = M_ASN1_STRING_length(a->value.sequence);
157 
158  c.pp = &p;
159  c.p = p;
160  c.max = p + length;
161  c.error = ASN1_R_DATA_IS_WRONG;
162 
163  M_ASN1_D2I_start_sequence();
164  c.q = c.p;
165  if ((ai = d2i_ASN1_INTEGER(NULL, &c.p, c.slen)) == NULL)
166  goto err;
167  c.slen -= (c.p - c.q);
168  c.q = c.p;
169  if ((os = d2i_ASN1_OCTET_STRING(NULL, &c.p, c.slen)) == NULL)
170  goto err;
171  c.slen -= (c.p - c.q);
172  if (!M_ASN1_D2I_end_sequence())
173  goto err;
174 
175  if (num != NULL)
176  *num = ASN1_INTEGER_get(ai);
177 
178  ret = M_ASN1_STRING_length(os);
179  if (max_len > ret)
180  n = ret;
181  else
182  n = max_len;
183 
184  if (data != NULL)
185  memcpy(data, M_ASN1_STRING_data(os), n);
186  if (0) {
187  err:
188  ASN1err(ASN1_F_ASN1_TYPE_GET_INT_OCTETSTRING, ASN1_R_DATA_IS_WRONG);
189  }
190  if (os != NULL)
191  M_ASN1_OCTET_STRING_free(os);
192  if (ai != NULL)
193  M_ASN1_INTEGER_free(ai);
194  return (ret);
195 }
