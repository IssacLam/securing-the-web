70 /*
71  * SetBlobCmp This function compares two elements of SET_OF block
72  */
73 static int SetBlobCmp(const void *elem1, const void *elem2)
74 {
75  const MYBLOB *b1 = (const MYBLOB *)elem1;
76  const MYBLOB *b2 = (const MYBLOB *)elem2;
77  int r;
78 
79  r = memcmp(b1->pbData, b2->pbData,
80  b1->cbData < b2->cbData ? b1->cbData : b2->cbData);
81  if (r != 0)
82  return r;
83  return b1->cbData - b2->cbData;
84 }

89 int i2d_ASN1_SET(STACK_OF(OPENSSL_BLOCK) *a, unsigned char **pp,
90  i2d_of_void *i2d, int ex_tag, int ex_class, int is_set)
91 {
92  int ret = 0, r;
93  int i;
94  unsigned char *p;
95  unsigned char *pStart, *pTempMem;
96  MYBLOB *rgSetBlob;
97  int totSize;
98 
99  if (a == NULL)
100  return (0);
101  for (i = sk_OPENSSL_BLOCK_num(a) - 1; i >= 0; i--)
102  ret += i2d(sk_OPENSSL_BLOCK_value(a, i), NULL);
103  r = ASN1_object_size(1, ret, ex_tag);
104  if (pp == NULL)
105  return (r);
106 
107  p = *pp;
108  ASN1_put_object(&p, 1, ret, ex_tag, ex_class);
109 
110 /* Modified by gp@nsj.co.jp */
111  /* And then again by Ben */
112  /* And again by Steve */
113 
114  if (!is_set || (sk_OPENSSL_BLOCK_num(a) < 2)) {
115  for (i = 0; i < sk_OPENSSL_BLOCK_num(a); i++)
116  i2d(sk_OPENSSL_BLOCK_value(a, i), &p);
117 
118  *pp = p;
119  return (r);
120  }
121 
122  pStart = p; /* Catch the beg of Setblobs */
123  /* In this array we will store the SET blobs */
124  rgSetBlob = OPENSSL_malloc(sk_OPENSSL_BLOCK_num(a) * sizeof(MYBLOB));
125  if (rgSetBlob == NULL) {
126  ASN1err(ASN1_F_I2D_ASN1_SET, ERR_R_MALLOC_FAILURE);
127  return (0);
128  }
129 
130  for (i = 0; i < sk_OPENSSL_BLOCK_num(a); i++) {
131  rgSetBlob[i].pbData = p; /* catch each set encode blob */
132  i2d(sk_OPENSSL_BLOCK_value(a, i), &p);
133  rgSetBlob[i].cbData = p - rgSetBlob[i].pbData; /* Length of this
134  * SetBlob */
135  }
136  *pp = p;
137  totSize = p - pStart; /* This is the total size of all set blobs */
138 
139  /*
140  * Now we have to sort the blobs. I am using a simple algo. *Sort ptrs
141  * *Copy to temp-mem *Copy from temp-mem to user-mem
142  */
143  qsort(rgSetBlob, sk_OPENSSL_BLOCK_num(a), sizeof(MYBLOB), SetBlobCmp);
144  if (!(pTempMem = OPENSSL_malloc(totSize))) {
145  ASN1err(ASN1_F_I2D_ASN1_SET, ERR_R_MALLOC_FAILURE);
146  return (0);
147  }
148 
149 /* Copy to temp mem */
150  p = pTempMem;
151  for (i = 0; i < sk_OPENSSL_BLOCK_num(a); ++i) {
152  memcpy(p, rgSetBlob[i].pbData, rgSetBlob[i].cbData);
153  p += rgSetBlob[i].cbData;
154  }
155 
156 /* Copy back to user mem*/
157  memcpy(pStart, pTempMem, totSize);
158  OPENSSL_free(pTempMem);
159  OPENSSL_free(rgSetBlob);
160 
161  return (r);
162 }
