113 void lh_node_usage_stats(LHASH *lh, FILE *out)
114 {
115  LHASH_NODE *n;
116  unsigned long num;
117  unsigned int i;
118  unsigned long total = 0, n_used = 0;
119 
120  for (i = 0; i < lh->num_nodes; i++) {
121  for (n = lh->b[i], num = 0; n != NULL; n = n->next)
122  num++;
123  if (num != 0) {
124  n_used++;
125  total += num;
126  }
127  }
128  fprintf(out, "%lu nodes used out of %u\n", n_used, lh->num_nodes);
129  fprintf(out, "%lu items\n", total);
130  if (n_used == 0)
131  return;
132  fprintf(out, "load %d.%02d actual load %d.%02d\n",
133  (int)(total / lh->num_nodes),
134  (int)((total % lh->num_nodes) * 100 / lh->num_nodes),
135  (int)(total / n_used), (int)((total % n_used) * 100 / n_used));
136 }
