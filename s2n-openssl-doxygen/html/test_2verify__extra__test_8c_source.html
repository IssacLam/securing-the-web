65 static STACK_OF(X509) *load_certs_from_file(const char *filename)
66 {
67  STACK_OF(X509) *certs;
68  BIO *bio;
69  X509 *x;
70 
71  bio = BIO_new_file(filename, "r");
72 
73  if (bio == NULL) {
74  return NULL;
75  }
76 
77  certs = sk_X509_new_null();
78  if (certs == NULL) {
79  BIO_free(bio);
80  return NULL;
81  }
82 
83  ERR_set_mark();
84  do {
85  x = PEM_read_bio_X509(bio, NULL, 0, NULL);
86  if (x != NULL && !sk_X509_push(certs, x)) {
87  sk_X509_pop_free(certs, X509_free);
88  BIO_free(bio);
89  return NULL;
90  } else if (x == NULL) {
91  /*
92  * We probably just ran out of certs, so ignore any errors
93  * generated
94  */
95  ERR_pop_to_mark();
96  }
97  } while (x != NULL);
98 
99  BIO_free(bio);
100 
101  return certs;
102 }

104 /*
105  * Test for CVE-2015-1793 (Alternate Chains Certificate Forgery)
106  *
107  * Chain is as follows:
108  *
109  * rootCA (self-signed)
110  * |
111  * interCA
112  * |
113  * subinterCA subinterCA (self-signed)
114  * | |
115  * leaf ------------------
116  * |
117  * bad
118  *
119  * rootCA, interCA, subinterCA, subinterCA (ss) all have CA=TRUE
120  * leaf and bad have CA=FALSE
121  *
122  * subinterCA and subinterCA (ss) have the same subject name and keys
123  *
124  * interCA (but not rootCA) and subinterCA (ss) are in the trusted store
125  * (roots.pem)
126  * leaf and subinterCA are in the untrusted list (untrusted.pem)
127  * bad is the certificate being verified (bad.pem)
128  *
129  * Versions vulnerable to CVE-2015-1793 will fail to detect that leaf has
130  * CA=FALSE, and will therefore incorrectly verify bad
131  *
132  */
133 static int test_alt_chains_cert_forgery(void)
134 {
135  int ret = 0;
136  int i;
137  X509 *x = NULL;
138  STACK_OF(X509) *untrusted = NULL;
139  BIO *bio = NULL;
140  X509_STORE_CTX *sctx = NULL;
141  X509_STORE *store = NULL;
142  X509_LOOKUP *lookup = NULL;
143 
144  store = X509_STORE_new();
145  if (store == NULL)
146  goto err;
147 
148  lookup = X509_STORE_add_lookup(store, X509_LOOKUP_file());
149  if (lookup == NULL)
150  goto err;
151  if(!X509_LOOKUP_load_file(lookup, "certs/roots.pem", X509_FILETYPE_PEM))
152  goto err;
153 
154  untrusted = load_certs_from_file("certs/untrusted.pem");
155 
156  if ((bio = BIO_new_file("certs/bad.pem", "r")) == NULL)
157  goto err;
158 
159  if((x = PEM_read_bio_X509(bio, NULL, 0, NULL)) == NULL)
160  goto err;
161 
162  sctx = X509_STORE_CTX_new();
163  if (sctx == NULL)
164  goto err;
165 
166  if (!X509_STORE_CTX_init(sctx, store, x, untrusted))
167  goto err;
168 
169  i = X509_verify_cert(sctx);
170 
171  if(i == 0 && X509_STORE_CTX_get_error(sctx) == X509_V_ERR_INVALID_CA) {
172  /* This is the result we were expecting: Test passed */
173  ret = 1;
174  }
175  err:
176  X509_STORE_CTX_free(sctx);
177  X509_free(x);
178  BIO_free(bio);
179  sk_X509_pop_free(untrusted, X509_free);
180  X509_STORE_free(store);
181  if (ret != 1)
182  ERR_print_errors_fp(stderr);
183  return ret;
184 }
