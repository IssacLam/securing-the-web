97 int EVP_read_pw_string_min(char *buf, int min, int len, const char *prompt,
98  int verify)
99 {
100  int ret;
101  char buff[BUFSIZ];
102  UI *ui;
103 
104  if ((prompt == NULL) && (prompt_string[0] != '\0'))
105  prompt = prompt_string;
106  ui = UI_new();
107  if (ui == NULL)
108  return -1;
109  UI_add_input_string(ui, prompt, 0, buf, min,

111  if (verify)
112  UI_add_verify_string(ui, prompt, 0,
113  buff, min, (len >= BUFSIZ) ? BUFSIZ - 1 : len,
114  buf);
115  ret = UI_process(ui);
116  UI_free(ui);
117  OPENSSL_cleanse(buff, BUFSIZ);

119 }

121 int EVP_BytesToKey(const EVP_CIPHER *type, const EVP_MD *md,
122  const unsigned char *salt, const unsigned char *data,
123  int datal, int count, unsigned char *key,
124  unsigned char *iv)
125 {
126  EVP_MD_CTX c;
127  unsigned char md_buf[EVP_MAX_MD_SIZE];
128  int niv, nkey, addmd = 0;
129  unsigned int mds = 0, i;
130  int rv = 0;
131  nkey = type->key_len;
132  niv = type->iv_len;
133  OPENSSL_assert(nkey <= EVP_MAX_KEY_LENGTH);
134  OPENSSL_assert(niv <= EVP_MAX_IV_LENGTH);
135 
136  if (data == NULL)
137  return (nkey);
138 
139  EVP_MD_CTX_init(&c);
140  for (;;) {
141  if (!EVP_DigestInit_ex(&c, md, NULL))
142  goto err;
143  if (addmd++)
144  if (!EVP_DigestUpdate(&c, &(md_buf[0]), mds))
145  goto err;
146  if (!EVP_DigestUpdate(&c, data, datal))
147  goto err;
148  if (salt != NULL)
149  if (!EVP_DigestUpdate(&c, salt, PKCS5_SALT_LEN))
150  goto err;
151  if (!EVP_DigestFinal_ex(&c, &(md_buf[0]), &mds))
152  goto err;
153 
154  for (i = 1; i < (unsigned int)count; i++) {
155  if (!EVP_DigestInit_ex(&c, md, NULL))
156  goto err;
157  if (!EVP_DigestUpdate(&c, &(md_buf[0]), mds))
158  goto err;
159  if (!EVP_DigestFinal_ex(&c, &(md_buf[0]), &mds))
160  goto err;
161  }
162  i = 0;
163  if (nkey) {
164  for (;;) {
165  if (nkey == 0)
166  break;
167  if (i == mds)
168  break;
169  if (key != NULL)
170  *(key++) = md_buf[i];
171  nkey--;
172  i++;
173  }
174  }
175  if (niv && (i != mds)) {
176  for (;;) {
177  if (niv == 0)
178  break;
179  if (i == mds)
180  break;
181  if (iv != NULL)
182  *(iv++) = md_buf[i];
183  niv--;
184  i++;
185  }
186  }
187  if ((nkey == 0) && (niv == 0))
188  break;
189  }
190  rv = type->key_len;
191  err:
192  EVP_MD_CTX_cleanup(&c);
193  OPENSSL_cleanse(md_buf, sizeof(md_buf));
194  return rv;
195 }
