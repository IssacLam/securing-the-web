64 int i2a_ASN1_INTEGER(BIO *bp, ASN1_INTEGER *a)
65 {
66  int i, n = 0;
67  static const char *h = "0123456789ABCDEF";
68  char buf[2];
69 
70  if (a == NULL)
71  return (0);
72 
73  if (a->type & V_ASN1_NEG) {
74  if (BIO_write(bp, "-", 1) != 1)
75  goto err;
76  n = 1;
77  }
78 
79  if (a->length == 0) {
80  if (BIO_write(bp, "00", 2) != 2)
81  goto err;
82  n += 2;
83  } else {
84  for (i = 0; i < a->length; i++) {
85  if ((i != 0) && (i % 35 == 0)) {
86  if (BIO_write(bp, "\\\n", 2) != 2)
87  goto err;
88  n += 2;
89  }
90  buf[0] = h[((unsigned char)a->data[i] >> 4) & 0x0f];
91  buf[1] = h[((unsigned char)a->data[i]) & 0x0f];
92  if (BIO_write(bp, buf, 2) != 2)
93  goto err;
94  n += 2;
95  }
96  }
97  return (n);
98  err:
99  return (-1);
100 }

102 int a2i_ASN1_INTEGER(BIO *bp, ASN1_INTEGER *bs, char *buf, int size)
103 {
104  int ret = 0;
105  int i, j, k, m, n, again, bufsize;
106  unsigned char *s = NULL, *sp;
107  unsigned char *bufp;
108  int num = 0, slen = 0, first = 1;
109 
110  bs->type = V_ASN1_INTEGER;
111 
112  bufsize = BIO_gets(bp, buf, size);
113  for (;;) {
114  if (bufsize < 1)
115  goto err_sl;
116  i = bufsize;
117  if (buf[i - 1] == '\n')
118  buf[--i] = '\0';
119  if (i == 0)
120  goto err_sl;
121  if (buf[i - 1] == '\r')
122  buf[--i] = '\0';
123  if (i == 0)
124  goto err_sl;
125  again = (buf[i - 1] == '\\');
126 
127  for (j = 0; j < i; j++) {
128 #ifndef CHARSET_EBCDIC
129  if (!(((buf[j] >= '0') && (buf[j] <= '9')) ||
130  ((buf[j] >= 'a') && (buf[j] <= 'f')) ||
131  ((buf[j] >= 'A') && (buf[j] <= 'F'))))
132 #else
133  /*
134  * This #ifdef is not strictly necessary, since the characters
135  * A...F a...f 0...9 are contiguous (yes, even in EBCDIC - but
136  * not the whole alphabet). Nevertheless, isxdigit() is faster.
137  */
138  if (!isxdigit(buf[j]))
139 #endif
140  {
141  i = j;
142  break;
143  }
144  }
145  buf[i] = '\0';
146  /*
147  * We have now cleared all the crap off the end of the line
148  */
149  if (i < 2)
150  goto err_sl;
151 
152  bufp = (unsigned char *)buf;
153  if (first) {
154  first = 0;
155  if ((bufp[0] == '0') && (buf[1] == '0')) {
156  bufp += 2;
157  i -= 2;
158  }
159  }
160  k = 0;
161  i -= again;
162  if (i % 2 != 0) {
163  ASN1err(ASN1_F_A2I_ASN1_INTEGER, ASN1_R_ODD_NUMBER_OF_CHARS);
164  goto err;
165  }
166  i /= 2;
167  if (num + i > slen) {
168  if (s == NULL)
169  sp = (unsigned char *)OPENSSL_malloc((unsigned int)num +
170  i * 2);
171  else
172  sp = OPENSSL_realloc_clean(s, slen, num + i * 2);
173  if (sp == NULL) {
174  ASN1err(ASN1_F_A2I_ASN1_INTEGER, ERR_R_MALLOC_FAILURE);
175  if (s != NULL)
176  OPENSSL_free(s);
177  goto err;
178  }
179  s = sp;
180  slen = num + i * 2;
181  }
182  for (j = 0; j < i; j++, k += 2) {
183  for (n = 0; n < 2; n++) {
184  m = bufp[k + n];
185  if ((m >= '0') && (m <= '9'))
186  m -= '0';
187  else if ((m >= 'a') && (m <= 'f'))
188  m = m - 'a' + 10;
189  else if ((m >= 'A') && (m <= 'F'))
190  m = m - 'A' + 10;
191  else {
192  ASN1err(ASN1_F_A2I_ASN1_INTEGER,
193  ASN1_R_NON_HEX_CHARACTERS);
194  goto err;
195  }
196  s[num + j] <<= 4;
197  s[num + j] |= m;
198  }
199  }
200  num += i;
201  if (again)
202  bufsize = BIO_gets(bp, buf, size);
203  else
204  break;
205  }
206  bs->length = num;
207  bs->data = s;
208  ret = 1;
209  err:
210  if (0) {
211  err_sl:
212  ASN1err(ASN1_F_A2I_ASN1_INTEGER, ASN1_R_SHORT_LINE);
213  }
214  return (ret);
215 }
