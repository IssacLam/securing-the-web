91 static int dsa_builtin_keygen(DSA *dsa)
92 {
93  int ok = 0;
94  BN_CTX *ctx = NULL;
95  BIGNUM *pub_key = NULL, *priv_key = NULL;
96 
97  if ((ctx = BN_CTX_new()) == NULL)
98  goto err;
99 
100  if (dsa->priv_key == NULL) {
101  if ((priv_key = BN_new()) == NULL)
102  goto err;
103  } else
104  priv_key = dsa->priv_key;
105 
106  do
107  if (!BN_rand_range(priv_key, dsa->q))
108  goto err;
109  while (BN_is_zero(priv_key)) ;
110 
111  if (dsa->pub_key == NULL) {
112  if ((pub_key = BN_new()) == NULL)
113  goto err;
114  } else
115  pub_key = dsa->pub_key;
116 
117  {
118  BIGNUM local_prk;
119  BIGNUM *prk;
120 
121  if ((dsa->flags & DSA_FLAG_NO_EXP_CONSTTIME) == 0) {
122  BN_init(&local_prk);
123  prk = &local_prk;
124  BN_with_flags(prk, priv_key, BN_FLG_CONSTTIME);
125  } else
126  prk = priv_key;
127 
128  if (!BN_mod_exp(pub_key, dsa->g, prk, dsa->p, ctx))
129  goto err;
130  }
131 
132  dsa->priv_key = priv_key;
133  dsa->pub_key = pub_key;
134  ok = 1;
135 
136  err:
137  if ((pub_key != NULL) && (dsa->pub_key == NULL))
138  BN_free(pub_key);
139  if ((priv_key != NULL) && (dsa->priv_key == NULL))
140  BN_free(priv_key);
141  if (ctx != NULL)
142  BN_CTX_free(ctx);
143  return (ok);
144 }
