64 static int get_random_bytes(unsigned char *buf, int num)
65 {
66  size_t rnd;
67 
68  while (num >= (int)sizeof(size_t)) {
69  if ((rnd = OPENSSL_ia32_rdrand()) == 0)
70  return 0;
71 
72  *((size_t *)buf) = rnd;
73  buf += sizeof(size_t);
74  num -= sizeof(size_t);
75  }
76  if (num) {
77  if ((rnd = OPENSSL_ia32_rdrand()) == 0)
78  return 0;
79 
80  memcpy(buf, &rnd, num);
81  }
82 
83  return 1;
84 }

120 static ENGINE *ENGINE_rdrand(void)
