102  char obj_tmp1[80];
103  char obj_tmp2[80];
104  for (i = 0; i < sk_POLICY_MAPPING_num(pmaps); i++) {
105  pmap = sk_POLICY_MAPPING_value(pmaps, i);
106  i2t_ASN1_OBJECT(obj_tmp1, 80, pmap->issuerDomainPolicy);
107  i2t_ASN1_OBJECT(obj_tmp2, 80, pmap->subjectDomainPolicy);

113 static void *v2i_POLICY_MAPPINGS(const X509V3_EXT_METHOD *method,
114  X509V3_CTX *ctx, STACK_OF(CONF_VALUE) *nval)
115 {
116  POLICY_MAPPINGS *pmaps;
117  POLICY_MAPPING *pmap;
118  ASN1_OBJECT *obj1, *obj2;
119  CONF_VALUE *val;
120  int i;
121 
122  if (!(pmaps = sk_POLICY_MAPPING_new_null())) {
123  X509V3err(X509V3_F_V2I_POLICY_MAPPINGS, ERR_R_MALLOC_FAILURE);
124  return NULL;
125  }
126 
127  for (i = 0; i < sk_CONF_VALUE_num(nval); i++) {
128  val = sk_CONF_VALUE_value(nval, i);
129  if (!val->value || !val->name) {
130  sk_POLICY_MAPPING_pop_free(pmaps, POLICY_MAPPING_free);
131  X509V3err(X509V3_F_V2I_POLICY_MAPPINGS,
132  X509V3_R_INVALID_OBJECT_IDENTIFIER);
133  X509V3_conf_err(val);
134  return NULL;
135  }
136  obj1 = OBJ_txt2obj(val->name, 0);
137  obj2 = OBJ_txt2obj(val->value, 0);
138  if (!obj1 || !obj2) {
139  sk_POLICY_MAPPING_pop_free(pmaps, POLICY_MAPPING_free);
140  X509V3err(X509V3_F_V2I_POLICY_MAPPINGS,
141  X509V3_R_INVALID_OBJECT_IDENTIFIER);
142  X509V3_conf_err(val);
143  return NULL;
144  }
145  pmap = POLICY_MAPPING_new();
146  if (!pmap) {
147  sk_POLICY_MAPPING_pop_free(pmaps, POLICY_MAPPING_free);
148  X509V3err(X509V3_F_V2I_POLICY_MAPPINGS, ERR_R_MALLOC_FAILURE);
149  return NULL;
150  }
151  pmap->issuerDomainPolicy = obj1;
152  pmap->subjectDomainPolicy = obj2;
153  sk_POLICY_MAPPING_push(pmaps, pmap);
154  }
155  return pmaps;
156 }
