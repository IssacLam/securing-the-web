84 {
85  char *b;
86  unsigned char *p;
87  int i, j = 0, n, ret = 1;
88 
89  n = i2d(x, NULL);
90  b = (char *)OPENSSL_malloc(n);
91  if (b == NULL) {
92  ASN1err(ASN1_F_ASN1_I2D_BIO, ERR_R_MALLOC_FAILURE);
93  return (0);
94  }
95 
96  p = (unsigned char *)b;
97  i2d(x, &p);
98 
99  for (;;) {
100  i = BIO_write(out, &(b[j]), n);
101  if (i == n)
102  break;
103  if (i <= 0) {
104  ret = 0;
105  break;
106  }
107  j += i;
108  n -= i;
109  }
110  OPENSSL_free(b);
111  return (ret);
