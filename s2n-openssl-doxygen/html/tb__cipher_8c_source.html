76 int ENGINE_register_ciphers(ENGINE *e)
77 {
78  if (e->ciphers) {
79  const int *nids;
80  int num_nids = e->ciphers(e, NULL, &nids, 0);
81  if (num_nids > 0)
82  return engine_table_register(&cipher_table,
83  engine_unregister_all_ciphers, e,
84  nids, num_nids, 0);
85  }
86  return 1;
87 }

120 /* Obtains a cipher implementation from an ENGINE functional reference */
121 const EVP_CIPHER *ENGINE_get_cipher(ENGINE *e, int nid)
122 {
123  const EVP_CIPHER *ret;
124  ENGINE_CIPHERS_PTR fn = ENGINE_get_ciphers(e);
125  if (!fn || !fn(e, &ret, NULL, nid)) {
126  ENGINEerr(ENGINE_F_ENGINE_GET_CIPHER, ENGINE_R_UNIMPLEMENTED_CIPHER);
127  return NULL;
128  }
129  return ret;
130 }
