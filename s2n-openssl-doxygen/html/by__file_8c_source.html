91 static int by_file_ctrl(X509_LOOKUP *ctx, int cmd, const char *argp,
92  long argl, char **ret)
93 {
94  int ok = 0;
95  char *file;
96 
97  switch (cmd) {
98  case X509_L_FILE_LOAD:
99  if (argl == X509_FILETYPE_DEFAULT) {
100  file = (char *)getenv(X509_get_default_cert_file_env());
101  if (file)
102  ok = (X509_load_cert_crl_file(ctx, file,
103  X509_FILETYPE_PEM) != 0);
104 
105  else
106  ok = (X509_load_cert_crl_file
107  (ctx, X509_get_default_cert_file(),
108  X509_FILETYPE_PEM) != 0);
109 
110  if (!ok) {
111  X509err(X509_F_BY_FILE_CTRL, X509_R_LOADING_DEFAULTS);
112  }
113  } else {
114  if (argl == X509_FILETYPE_PEM)
115  ok = (X509_load_cert_crl_file(ctx, argp,
116  X509_FILETYPE_PEM) != 0);
117  else
118  ok = (X509_load_cert_file(ctx, argp, (int)argl) != 0);
119  }
120  break;
121  }
122  return (ok);
123 }

125 int X509_load_cert_file(X509_LOOKUP *ctx, const char *file, int type)
126 {
127  int ret = 0;
128  BIO *in = NULL;
129  int i, count = 0;
130  X509 *x = NULL;
131 
132  if (file == NULL)
133  return (1);
134  in = BIO_new(BIO_s_file_internal());
135 
136  if ((in == NULL) || (BIO_read_filename(in, file) <= 0)) {
137  X509err(X509_F_X509_LOAD_CERT_FILE, ERR_R_SYS_LIB);
138  goto err;
139  }
140 
141  if (type == X509_FILETYPE_PEM) {
142  for (;;) {
143  x = PEM_read_bio_X509_AUX(in, NULL, NULL, NULL);
144  if (x == NULL) {
145  if ((ERR_GET_REASON(ERR_peek_last_error()) ==
146  PEM_R_NO_START_LINE) && (count > 0)) {
147  ERR_clear_error();
148  break;
149  } else {
150  X509err(X509_F_X509_LOAD_CERT_FILE, ERR_R_PEM_LIB);
151  goto err;
152  }
153  }
154  i = X509_STORE_add_cert(ctx->store_ctx, x);
155  if (!i)
156  goto err;
157  count++;
158  X509_free(x);
159  x = NULL;
160  }
161  ret = count;
162  } else if (type == X509_FILETYPE_ASN1) {
163  x = d2i_X509_bio(in, NULL);
164  if (x == NULL) {
165  X509err(X509_F_X509_LOAD_CERT_FILE, ERR_R_ASN1_LIB);
166  goto err;
167  }
168  i = X509_STORE_add_cert(ctx->store_ctx, x);
169  if (!i)
170  goto err;
171  ret = i;
172  } else {
173  X509err(X509_F_X509_LOAD_CERT_FILE, X509_R_BAD_X509_FILETYPE);
174  goto err;
175  }
176  err:
177  if (x != NULL)
178  X509_free(x);
179  if (in != NULL)
180  BIO_free(in);
181  return (ret);
182 }

243 int X509_load_cert_crl_file(X509_LOOKUP *ctx, const char *file, int type)
244 {
245  STACK_OF(X509_INFO) *inf;
246  X509_INFO *itmp;
247  BIO *in;
248  int i, count = 0;
249  if (type != X509_FILETYPE_PEM)
250  return X509_load_cert_file(ctx, file, type);
251  in = BIO_new_file(file, "r");
252  if (!in) {
253  X509err(X509_F_X509_LOAD_CERT_CRL_FILE, ERR_R_SYS_LIB);
254  return 0;
255  }
256  inf = PEM_X509_INFO_read_bio(in, NULL, NULL, NULL);
257  BIO_free(in);
258  if (!inf) {
259  X509err(X509_F_X509_LOAD_CERT_CRL_FILE, ERR_R_PEM_LIB);
260  return 0;
261  }
262  for (i = 0; i < sk_X509_INFO_num(inf); i++) {
263  itmp = sk_X509_INFO_value(inf, i);
264  if (itmp->x509) {
265  X509_STORE_add_cert(ctx->store_ctx, itmp->x509);
266  count++;
267  }
268  if (itmp->crl) {
269  X509_STORE_add_crl(ctx->store_ctx, itmp->crl);
270  count++;
271  }
272  }
273  sk_X509_INFO_pop_free(inf, X509_INFO_free);
274  return count;
275 }
