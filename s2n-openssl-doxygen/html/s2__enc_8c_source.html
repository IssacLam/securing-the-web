63 int ssl2_enc_init(SSL *s, int client)
64 {
65  /* Max number of bytes needed */
66  EVP_CIPHER_CTX *rs, *ws;
67  const EVP_CIPHER *c;
68  const EVP_MD *md;
69  int num;
70 
71  if (!ssl_cipher_get_evp(s->session, &c, &md, NULL, NULL, NULL)) {
72  ssl2_return_error(s, SSL2_PE_NO_CIPHER);
73  SSLerr(SSL_F_SSL2_ENC_INIT, SSL_R_PROBLEMS_MAPPING_CIPHER_FUNCTIONS);
74  return (0);
75  }
76  ssl_replace_hash(&s->read_hash, md);
77  ssl_replace_hash(&s->write_hash, md);
78 
79  if ((s->enc_read_ctx == NULL) && ((s->enc_read_ctx = (EVP_CIPHER_CTX *)
80  OPENSSL_malloc(sizeof(EVP_CIPHER_CTX)))
81  == NULL))
82  goto err;
83 
84  /*
85  * make sure it's intialized in case the malloc for enc_write_ctx fails
86  * and we exit with an error
87  */
88  rs = s->enc_read_ctx;
89  EVP_CIPHER_CTX_init(rs);
90 
91  if ((s->enc_write_ctx == NULL) && ((s->enc_write_ctx = (EVP_CIPHER_CTX *)
92  OPENSSL_malloc(sizeof
93  (EVP_CIPHER_CTX))) ==
94  NULL))
95  goto err;
96 
97  ws = s->enc_write_ctx;
98  EVP_CIPHER_CTX_init(ws);
99 
100  num = c->key_len;
101  s->s2->key_material_length = num * 2;
102  OPENSSL_assert(s->s2->key_material_length <= sizeof s->s2->key_material);
103 
104  if (ssl2_generate_key_material(s) <= 0)
105  return 0;
106 
107  OPENSSL_assert(c->iv_len <= (int)sizeof(s->session->key_arg));
108  EVP_EncryptInit_ex(ws, c, NULL,
109  &(s->s2->key_material[(client) ? num : 0]),
110  s->session->key_arg);
111  EVP_DecryptInit_ex(rs, c, NULL,
112  &(s->s2->key_material[(client) ? 0 : num]),
113  s->session->key_arg);
114  s->s2->read_key = &(s->s2->key_material[(client) ? 0 : num]);
115  s->s2->write_key = &(s->s2->key_material[(client) ? num : 0]);
116  return (1);
117  err:
118  SSLerr(SSL_F_SSL2_ENC_INIT, ERR_R_MALLOC_FAILURE);
119  return (0);
120 }

122 /*
123  * read/writes from s->s2->mac_data using length for encrypt and decrypt.
124  * It sets s->s2->padding and s->[rw]length if we are encrypting Returns 0 on
125  * error and 1 on success
126  */
127 int ssl2_enc(SSL *s, int send)
128 {
129  EVP_CIPHER_CTX *ds;
130  unsigned long l;
131  int bs;
132 
133  if (send) {
134  ds = s->enc_write_ctx;
135  l = s->s2->wlength;
136  } else {
137  ds = s->enc_read_ctx;
138  l = s->s2->rlength;
139  }
140 
141  /* check for NULL cipher */
142  if (ds == NULL)
143  return 1;
144 
145  bs = ds->cipher->block_size;
146  /*
147  * This should be using (bs-1) and bs instead of 7 and 8, but what the
148  * hell.
149  */
150  if (bs == 8)
151  l = (l + 7) / 8 * 8;
152 
153  if (EVP_Cipher(ds, s->s2->mac_data, s->s2->mac_data, l) < 1)
154  return 0;
155 
156  return 1;
157 }

159 void ssl2_mac(SSL *s, unsigned char *md, int send)
160 {
161  EVP_MD_CTX c;
162  unsigned char sequence[4], *p, *sec, *act;
163  unsigned long seq;
164  unsigned int len;
165 
166  if (send) {
167  seq = s->s2->write_sequence;
168  sec = s->s2->write_key;
169  len = s->s2->wact_data_length;
170  act = s->s2->wact_data;
171  } else {
172  seq = s->s2->read_sequence;
173  sec = s->s2->read_key;
174  len = s->s2->ract_data_length;
175  act = s->s2->ract_data;
176  }
177 
178  p = &(sequence[0]);
179  l2n(seq, p);
180 
181  /* There has to be a MAC algorithm. */
182  EVP_MD_CTX_init(&c);
183  EVP_MD_CTX_copy(&c, s->read_hash);
184  EVP_DigestUpdate(&c, sec, EVP_CIPHER_CTX_key_length(s->enc_read_ctx));
185  EVP_DigestUpdate(&c, act, len);
186  /* the above line also does the pad data */
187  EVP_DigestUpdate(&c, sequence, 4);
188  EVP_DigestFinal_ex(&c, md, NULL);
189  EVP_MD_CTX_cleanup(&c);
190 }
