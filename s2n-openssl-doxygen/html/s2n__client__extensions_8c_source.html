36 int s2n_client_extensions_send(struct s2n_connection *conn, struct s2n_stuffer *out)
37 {
38  uint16_t total_size = 0;
39 
40  /* Signature algorithms */
41  if (conn->actual_protocol_version == S2N_TLS12) {
42  total_size += 8;
43  }
44 
45  uint16_t application_protocols_len = conn->config->application_protocols.size;
46  uint16_t server_name_len = strlen(conn->server_name);
47 
48  if (server_name_len) {
49  total_size += 9 + server_name_len;
50  }
51  if (application_protocols_len) {
52  total_size += 6 + application_protocols_len;
53  }
54  if (conn->config->status_request_type != S2N_STATUS_REQUEST_NONE) {
55  total_size += 9;
56  }
57 
58  /* Write ECC extensions: Supported Curves and Supported Point Formats */
59  int ec_curves_count = sizeof(s2n_ecc_supported_curves) / sizeof(s2n_ecc_supported_curves[0]);
60  total_size += 12 + ec_curves_count * 2;
61 
62  GUARD(s2n_stuffer_write_uint16(out, total_size));
63 
64  if (conn->actual_protocol_version == S2N_TLS12) {
65  /* The extension header */
66  GUARD(s2n_stuffer_write_uint16(out, TLS_EXTENSION_SIGNATURE_ALGORITHMS));
67  GUARD(s2n_stuffer_write_uint16(out, 4));
68 
69  /* Just one signature/hash pair, so 2 bytes */
70  GUARD(s2n_stuffer_write_uint16(out, 2));
71  GUARD(s2n_stuffer_write_uint8(out, TLS_HASH_ALGORITHM_SHA1));
72  GUARD(s2n_stuffer_write_uint8(out, TLS_SIGNATURE_ALGORITHM_RSA));
73  }
74 
75  if (server_name_len) {
76  /* Write the server name */
77  GUARD(s2n_stuffer_write_uint16(out, TLS_EXTENSION_SERVER_NAME));
78  GUARD(s2n_stuffer_write_uint16(out, server_name_len + 5));
79 
80  /* Size of all of the server names */
81  GUARD(s2n_stuffer_write_uint16(out, server_name_len + 3));
82 
83  /* Name type - host name, RFC3546 */
84  GUARD(s2n_stuffer_write_uint8(out, 0));
85 
86  struct s2n_blob server_name;
87  server_name.data = (uint8_t *) conn->server_name;
88  server_name.size = server_name_len;
89  GUARD(s2n_stuffer_write_uint16(out, server_name_len));
90  GUARD(s2n_stuffer_write(out, &server_name));
91  }
92 
93  /* Write ALPN extension */
94  if (application_protocols_len) {
95  GUARD(s2n_stuffer_write_uint16(out, TLS_EXTENSION_ALPN));
96  GUARD(s2n_stuffer_write_uint16(out, application_protocols_len + 2));
97  GUARD(s2n_stuffer_write_uint16(out, application_protocols_len));
98  GUARD(s2n_stuffer_write(out, &conn->config->application_protocols));
99  }
100 
101  if (conn->config->status_request_type != S2N_STATUS_REQUEST_NONE) {
102  /* We only support OCSP */
103  eq_check(conn->config->status_request_type, S2N_STATUS_REQUEST_OCSP);
104  GUARD(s2n_stuffer_write_uint16(out, TLS_EXTENSION_STATUS_REQUEST));
105  GUARD(s2n_stuffer_write_uint16(out, 5));
106  GUARD(s2n_stuffer_write_uint8(out, (uint8_t)conn->config->status_request_type));
107  GUARD(s2n_stuffer_write_uint16(out, 0));
108  GUARD(s2n_stuffer_write_uint16(out, 0));
109  }
110 
111  /*
112  * RFC 4492: Clients SHOULD send both the Supported Elliptic Curves Extension
113  * and the Supported Point Formats Extension.
114  */
115  {
116  GUARD(s2n_stuffer_write_uint16(out, TLS_EXTENSION_ELLIPTIC_CURVES));
117  GUARD(s2n_stuffer_write_uint16(out, 2 + ec_curves_count * 2));
118  /* Curve list len */
119  GUARD(s2n_stuffer_write_uint16(out, ec_curves_count * 2));
120  /* Curve list */
121  for (int i = 0; i < ec_curves_count; i++) {
122  GUARD(s2n_stuffer_write_uint16(out, s2n_ecc_supported_curves[i].iana_id));
123  }
124 
125  GUARD(s2n_stuffer_write_uint16(out, TLS_EXTENSION_EC_POINT_FORMATS));
126  GUARD(s2n_stuffer_write_uint16(out, 2));
127  /* Point format list len */
128  GUARD(s2n_stuffer_write_uint8(out, 1));
129  /* Only allow uncompressed format */
130  GUARD(s2n_stuffer_write_uint8(out, 0));
131  }
132 
133  return 0;
134 }

136 int s2n_client_extensions_recv(struct s2n_connection *conn, struct s2n_blob *extensions)
137 {
138  struct s2n_stuffer in;
139 
140  GUARD(s2n_stuffer_init(&in, extensions));
141  GUARD(s2n_stuffer_write(&in, extensions));
142 
143  while (s2n_stuffer_data_available(&in)) {
144  struct s2n_blob ext;
145  uint16_t extension_type, extension_size;
146  struct s2n_stuffer extension;
147 
148  GUARD(s2n_stuffer_read_uint16(&in, &extension_type));
149  GUARD(s2n_stuffer_read_uint16(&in, &extension_size));
150 
151  ext.size = extension_size;
152  lte_check(extension_size, s2n_stuffer_data_available(&in));
153  ext.data = s2n_stuffer_raw_read(&in, ext.size);
154  notnull_check(ext.data);
155 
156  GUARD(s2n_stuffer_init(&extension, &ext));
157  GUARD(s2n_stuffer_write(&extension, &ext));
158 
159  switch (extension_type) {
160  case TLS_EXTENSION_SERVER_NAME:
161  GUARD(s2n_server_name_rcv(conn, &extension));
162  break;
163  case TLS_EXTENSION_SIGNATURE_ALGORITHMS:
164  GUARD(s2n_signature_algorithms_rcv(conn, &extension));
165  break;
166  case TLS_EXTENSION_ALPN:
167  GUARD(s2n_alpn_rcv(conn, &extension));
168  break;
169  case TLS_EXTENSION_STATUS_REQUEST:
170  GUARD(s2n_status_request_rcv(conn, &extension));
171  break;
172  case TLS_EXTENSION_ELLIPTIC_CURVES:
173  GUARD(s2n_elliptic_curves_rcv(conn, &extension));
174  break;
175  case TLS_EXTENSION_EC_POINT_FORMATS:
176  GUARD(s2n_ec_point_formats_rcv(conn, &extension));
177  break;
178  }
179  }
180 
181  return 0;
182 }

184 static int s2n_server_name_rcv(struct s2n_connection *conn, struct s2n_stuffer *extension)
185 {
186  uint16_t size_of_all;
187  uint8_t server_name_type;
188  uint16_t server_name_len;
189  uint8_t *server_name;
190 
191  GUARD(s2n_stuffer_read_uint16(extension, &size_of_all));
192  if (size_of_all > s2n_stuffer_data_available(extension) || size_of_all < 3) {
193  /* the size of all server names is incorrect, ignore the extension */
194  return 0;
195  }
196 
197  GUARD(s2n_stuffer_read_uint8(extension, &server_name_type));
198  if (server_name_type != 0) {
199  /* unknown server name type, ignore the extension */
200  return 0;
201  }
202 
203  GUARD(s2n_stuffer_read_uint16(extension, &server_name_len));
204  if (server_name_len + 3 > size_of_all) {
205  /* the server name length is incorrect, ignore the extension */
206  return 0;
207  }
208 
209  if (server_name_len > sizeof(conn->server_name) - 1) {
210  /* the server name is too long, ignore the extension */
211  return 0;
212  }
213 
214  notnull_check(server_name = s2n_stuffer_raw_read(extension, server_name_len));
215 
216  /* copy the first server name */
217  memcpy_check(conn->server_name, server_name, server_name_len);
218  return 0;
219 }

221 static int s2n_signature_algorithms_rcv(struct s2n_connection *conn, struct s2n_stuffer *extension)
222 {
223  uint16_t length_of_all_pairs;
224  GUARD(s2n_stuffer_read_uint16(extension, &length_of_all_pairs));
225  if (length_of_all_pairs > s2n_stuffer_data_available(extension)) {
226  /* Malformed length, ignore the extension */
227  return 0;
228  }
229 
230  if (length_of_all_pairs % 2 || s2n_stuffer_data_available(extension) % 2) {
231  /* Pairs occur in two byte lengths. Malformed length, ignore the extension. */
232  return 0;
233  }
234 
235  while (s2n_stuffer_data_available(extension)) {
236  uint8_t hash_alg;
237  uint8_t sig_alg;
238 
239  GUARD(s2n_stuffer_read_uint8(extension, &hash_alg));
240  GUARD(s2n_stuffer_read_uint8(extension, &sig_alg));
241 
242  if (hash_alg == TLS_HASH_ALGORITHM_SHA1 && sig_alg == TLS_SIGNATURE_ALGORITHM_RSA) {
243  /* Supported pair found, return success */
244  return 0;
245  }
246  }
247 
248  /* No supported signature algorithms, fail the handshake */
249  S2N_ERROR(S2N_ERR_INVALID_SIGNATURE_ALGORITHM);
250 }

252 static int s2n_alpn_rcv(struct s2n_connection *conn, struct s2n_stuffer *extension)
253 {
254  uint16_t size_of_all;
255  struct s2n_stuffer client_protos;
256  struct s2n_stuffer server_protos;
257 
258  if (!conn->config->application_protocols.size) {
259  /* No protocols configured, nothing to do */
260  return 0;
261  }
262 
263  GUARD(s2n_stuffer_read_uint16(extension, &size_of_all));
264  if (size_of_all > s2n_stuffer_data_available(extension) || size_of_all < 3) {
265  /* Malformed length, ignore the extension */
266  return 0;
267  }
268 
269  struct s2n_blob application_protocols = {
270  .data = s2n_stuffer_raw_read(extension, size_of_all),
271  .size = size_of_all
272  };
273  notnull_check(application_protocols.data);
274 
275  /* Find a matching protocol */
276  GUARD(s2n_stuffer_init(&client_protos, &application_protocols));
277  GUARD(s2n_stuffer_write(&client_protos, &application_protocols));
278  GUARD(s2n_stuffer_init(&server_protos, &conn->config->application_protocols));
279  GUARD(s2n_stuffer_write(&server_protos, &conn->config->application_protocols));
280 
281  while (s2n_stuffer_data_available(&server_protos)) {
282  uint8_t length;
283  uint8_t protocol[255];
284  GUARD(s2n_stuffer_read_uint8(&server_protos, &length));
285  GUARD(s2n_stuffer_read_bytes(&server_protos, protocol, length));
286 
287  while (s2n_stuffer_data_available(&client_protos)) {
288  uint8_t client_length;
289  uint8_t client_protocol[255];
290  GUARD(s2n_stuffer_read_uint8(&client_protos, &client_length));
291  if (client_length > s2n_stuffer_data_available(&client_protos)) {
292  S2N_ERROR(S2N_ERR_BAD_MESSAGE);
293  }
294  if (client_length != length) {
295  GUARD(s2n_stuffer_skip_read(&client_protos, client_length));
296  } else {
297  GUARD(s2n_stuffer_read_bytes(&client_protos, client_protocol, client_length));
298  if (memcmp(client_protocol, protocol, client_length) == 0) {
299  memcpy_check(conn->application_protocol, client_protocol, client_length);
300  conn->application_protocol[client_length] = '\0';
301  return 0;
302  }
303  }
304  }
305 
306  GUARD(s2n_stuffer_reread(&client_protos));
307  }
308 
309  S2N_ERROR(S2N_ERR_NO_APPLICATION_PROTOCOL);
310 }

312 static int s2n_status_request_rcv(struct s2n_connection *conn, struct s2n_stuffer *extension)
313 {
314  uint16_t size_of_all;
315  GUARD(s2n_stuffer_read_uint16(extension, &size_of_all));
316  if (size_of_all > s2n_stuffer_data_available(extension) || size_of_all < 5) {
317  /* Malformed length, ignore the extension */
318  return 0;
319  }
320  uint8_t type;
321  GUARD(s2n_stuffer_read_uint8(extension, &type));
322  if (type != (uint8_t)S2N_STATUS_REQUEST_OCSP) {
323  /* We only support OCSP (type 1), ignore the extension */
324  return 0;
325  }
326  conn->status_type = (s2n_status_request_type)type;
327  return 0;
328 }

330 static int s2n_elliptic_curves_rcv(struct s2n_connection *conn, struct s2n_stuffer *extension)
331 {
332  uint16_t size_of_all;
333  struct s2n_blob proposed_curves;
334 
335  GUARD(s2n_stuffer_read_uint16(extension, &size_of_all));
336  if (size_of_all > s2n_stuffer_data_available(extension) || size_of_all % 2) {
337  /* Malformed length, ignore the extension */
338  return 0;
339  }
340 
341  proposed_curves.size = size_of_all;
342  proposed_curves.data = s2n_stuffer_raw_read(extension, proposed_curves.size);
343  notnull_check(proposed_curves.data);
344 
345  if (s2n_ecc_find_supported_curve(&proposed_curves, &conn->pending.server_ecc_params.negotiated_curve) != 0) {
346  /* Can't agree on a curve, ECC is not allowed. Return success to proceed with the handhsake. */
347  conn->pending.server_ecc_params.negotiated_curve = NULL;
348  }
349  return 0;
350 }
