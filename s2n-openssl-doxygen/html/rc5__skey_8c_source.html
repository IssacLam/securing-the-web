62 void RC5_32_set_key(RC5_32_KEY *key, int len, const unsigned char *data,
63  int rounds)
64 {
65  RC5_32_INT L[64], l, ll, A, B, *S, k;
66  int i, j, m, c, t, ii, jj;
67 
68  if ((rounds != RC5_16_ROUNDS) &&
69  (rounds != RC5_12_ROUNDS) && (rounds != RC5_8_ROUNDS))
70  rounds = RC5_16_ROUNDS;
71 
72  key->rounds = rounds;
73  S = &(key->data[0]);
74  j = 0;
75  for (i = 0; i <= (len - 8); i += 8) {
76  c2l(data, l);
77  L[j++] = l;
78  c2l(data, l);
79  L[j++] = l;
80  }
81  ii = len - i;
82  if (ii) {
83  k = len & 0x07;
84  c2ln(data, l, ll, k);
85  L[j + 0] = l;
86  L[j + 1] = ll;
87  }
88 
89  c = (len + 3) / 4;
90  t = (rounds + 1) * 2;
91  S[0] = RC5_32_P;
92  for (i = 1; i < t; i++)
93  S[i] = (S[i - 1] + RC5_32_Q) & RC5_32_MASK;
94 
95  j = (t > c) ? t : c;
96  j *= 3;
97  ii = jj = 0;
98  A = B = 0;
99  for (i = 0; i < j; i++) {
100  k = (S[ii] + A + B) & RC5_32_MASK;
101  A = S[ii] = ROTATE_l32(k, 3);
102  m = (int)(A + B);
103  k = (L[jj] + A + B) & RC5_32_MASK;
104  B = L[jj] = ROTATE_l32(k, m);
105  if (++ii >= t)
106  ii = 0;
107  if (++jj >= c)
108  jj = 0;
109  }
110 }
