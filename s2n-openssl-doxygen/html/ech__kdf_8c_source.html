61 /* Way more than we will ever need */
62 #define ECDH_KDF_MAX (1 << 30)

64 int ECDH_KDF_X9_62(unsigned char *out, size_t outlen,
65  const unsigned char *Z, size_t Zlen,
66  const unsigned char *sinfo, size_t sinfolen,
67  const EVP_MD *md)
68 {
69  EVP_MD_CTX mctx;
70  int rv = 0;
71  unsigned int i;
72  size_t mdlen;
73  unsigned char ctr[4];
74  if (sinfolen > ECDH_KDF_MAX || outlen > ECDH_KDF_MAX
75  || Zlen > ECDH_KDF_MAX)
76  return 0;
77  mdlen = EVP_MD_size(md);
78  EVP_MD_CTX_init(&mctx);
79  for (i = 1;; i++) {
80  unsigned char mtmp[EVP_MAX_MD_SIZE];
81  EVP_DigestInit_ex(&mctx, md, NULL);
82  ctr[3] = i & 0xFF;
83  ctr[2] = (i >> 8) & 0xFF;
84  ctr[1] = (i >> 16) & 0xFF;
85  ctr[0] = (i >> 24) & 0xFF;
86  if (!EVP_DigestUpdate(&mctx, Z, Zlen))
87  goto err;
88  if (!EVP_DigestUpdate(&mctx, ctr, sizeof(ctr)))
89  goto err;
90  if (!EVP_DigestUpdate(&mctx, sinfo, sinfolen))
91  goto err;
92  if (outlen >= mdlen) {
93  if (!EVP_DigestFinal(&mctx, out, NULL))
94  goto err;
95  outlen -= mdlen;
96  if (outlen == 0)
97  break;
98  out += mdlen;
99  } else {
100  if (!EVP_DigestFinal(&mctx, mtmp, NULL))
101  goto err;
102  memcpy(out, mtmp, outlen);
103  OPENSSL_cleanse(mtmp, mdlen);
104  break;
105  }
106  }
107  rv = 1;
108  err:
109  EVP_MD_CTX_cleanup(&mctx);
110  return rv;
111 }
