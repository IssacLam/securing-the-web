105 int main(int argc, const char *argv[])
106 {
107  struct addrinfo hints, *ai;
108  int r, sockfd = 0;
109 
110  if (argc != 3) {
111  usage();
112  }
113 
114  memset(&hints, 0, sizeof(hints));
115 
116  hints.ai_family = AF_UNSPEC;
117  hints.ai_socktype = SOCK_STREAM;
118 
119  if (signal(SIGPIPE, SIG_IGN) == SIG_ERR) {
120  fprintf(stderr, "Error disabling SIGPIPE\n");
121  exit(1);
122  }
123 
124  if ((r = getaddrinfo(argv[1], argv[2], &hints, &ai)) < 0) {
125  fprintf(stderr, "getaddrinfo error: %s\n", gai_strerror(r));
126  exit(1);
127  }
128 
129  if ((sockfd = socket(ai->ai_family, ai->ai_socktype, ai->ai_protocol)) == -1) {
130  fprintf(stderr, "socket error: %s\n", strerror(errno));
131  exit(1);
132  }
133 
134  r = 1;
135  if (setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, &r, sizeof(int)) < 0) {
136  fprintf(stderr, "setsockopt error: %s\n", strerror(errno));
137  exit(1);
138  }
139 
140  if (bind(sockfd, ai->ai_addr, ai->ai_addrlen) < 0) {
141  fprintf(stderr, "bind error: %s\n", strerror(errno));
142  exit(1);
143  }
144 
145  if (listen(sockfd, 1) == -1) {
146  fprintf(stderr, "listen error: %s\n", strerror(errno));
147  exit(1);
148  }
149 
150  if (s2n_init() < 0) {
151  fprintf(stderr, "Error running s2n_init(): '%s'\n", s2n_strerror(s2n_errno, "EN"));
152  }
153 
154  printf("Listening on %s:%s\n", argv[1], argv[2]);
155 
156  struct s2n_config *config = s2n_config_new();
157  if (!config) {
158  fprintf(stderr, "Error getting new s2n config: '%s'\n", s2n_strerror(s2n_errno, "EN"));
159  exit(1);
160  }
161 
162  if (s2n_config_add_cert_chain_and_key(config, certificate, private_key) < 0) {
163  fprintf(stderr, "Error getting certificate/key: '%s'\n", s2n_strerror(s2n_errno, "EN"));
164  exit(1);
165  }
166 
167  if (s2n_config_add_dhparams(config, dhparams) < 0) {
168  fprintf(stderr, "Error adding DH parameters: '%s'\n", s2n_strerror(s2n_errno, "EN"));
169  exit(1);
170  }
171 
172  struct s2n_connection *conn = s2n_connection_new(S2N_SERVER);
173  if (!conn) {
174  fprintf(stderr, "Error getting new s2n connection: '%s'\n", s2n_strerror(s2n_errno, "EN"));
175  exit(1);
176  }
177 
178  if (s2n_connection_set_config(conn, config) < 0) {
179  fprintf(stderr, "Error setting configuration: '%s'\n", s2n_strerror(s2n_errno, "EN"));
180  exit(1);
181  }
182 
183  int fd;
184  while ((fd = accept(sockfd, ai->ai_addr, &ai->ai_addrlen)) > 0) {
185  if (s2n_connection_set_fd(conn, fd) < 0) {
186  fprintf(stderr, "Error setting file descriptor: '%s'\n", s2n_strerror(s2n_errno, "EN"));
187  exit(1);
188  }
189 
190  echo(conn, fd);
191 
192  if (s2n_connection_wipe(conn) < 0) {
193  fprintf(stderr, "Error wiping connection: '%s'\n", s2n_strerror(s2n_errno, "EN"));
194  exit(1);
195  }
196  }
197 
198  if (s2n_connection_free(conn) < 0) {
199  fprintf(stderr, "Error freeing connection: '%s'\n", s2n_strerror(s2n_errno, "EN"));
200  exit(1);
201  }
202 
203  if (s2n_cleanup() < 0) {
204  fprintf(stderr, "Error running s2n_cleanup(): '%s'\n", s2n_strerror(s2n_errno, "EN"));
205  }
206 
207  return 0;
208 }
